346 int
347 net_route_v4_best_gw(openvpn_net_ctx_t *ctx, const in_addr_t *dst,
348  in_addr_t *best_gw, char *best_iface)
349 {
350  best_iface[0] = '\0';
351 
352  FILE *fp = fopen("/proc/net/route", "r");
353  if (!fp)
354  {
355  return -1;
356  }
357 
358  char line[256];
359  int count = 0;
360  unsigned int lowest_metric = UINT_MAX;
361  while (fgets(line, sizeof(line), fp) != NULL)
362  {
363  if (count)
364  {
365  unsigned int net_x = 0;
366  unsigned int mask_x = 0;
367  unsigned int gw_x = 0;
368  unsigned int metric = 0;
369  unsigned int flags = 0;
370  char name[16];
371  name[0] = '\0';
372 
373  const int np = sscanf(line, "%15s\t%x\t%x\t%x\t%*s\t%*s\t%d\t%x",
374  name, &net_x, &gw_x, &flags, &metric,
375  &mask_x);
376 
377  if (np == 6 && (flags & IFF_UP))
378  {
379  const in_addr_t net = ntohl(net_x);
380  const in_addr_t mask = ntohl(mask_x);
381  const in_addr_t gw = ntohl(gw_x);
382 
383  if (!net && !mask && metric < lowest_metric)
384  {
385  *best_gw = gw;
386  strcpy(best_iface, name);
387  lowest_metric = metric;
388  }
389  }
390  }
391  ++count;
392  }
393  fclose(fp);
394 
395  return 0;
396 }
