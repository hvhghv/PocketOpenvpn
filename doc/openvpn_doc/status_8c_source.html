62 struct status_output *
63 status_open(const char *filename,
64  const int refresh_freq,
65  const int msglevel,
66  const struct virtual_output *vout,
67  const unsigned int flags)
68 {
69  struct status_output *so = NULL;
70  if (filename || msglevel >= 0 || vout)
71  {
72  ALLOC_OBJ_CLEAR(so, struct status_output);
73  so->flags = flags;
74  so->msglevel = msglevel;
75  so->vout = vout;
76  so->fd = -1;
77  buf_reset(&so->read_buf);
78  event_timeout_clear(&so->et);
79  if (filename)
80  {
81  switch (so->flags)
82  {
83  case STATUS_OUTPUT_WRITE:
84  so->fd = platform_open(filename,
85  O_CREAT | O_TRUNC | O_WRONLY,
86  S_IRUSR | S_IWUSR);
87  break;
88 
89  case STATUS_OUTPUT_READ:
90  so->fd = platform_open(filename,
91  O_RDONLY,
92  S_IRUSR | S_IWUSR);
93  break;
94 
95  case STATUS_OUTPUT_READ|STATUS_OUTPUT_WRITE:
96  so->fd = platform_open(filename,
97  O_CREAT | O_RDWR,
98  S_IRUSR | S_IWUSR);
99  break;
100 
101  default:
102  ASSERT(0);
103  }
104  if (so->fd >= 0)
105  {
106  so->filename = string_alloc(filename, NULL);
107  set_cloexec(so->fd);
108 
109  /* allocate read buffer */
110  if (so->flags & STATUS_OUTPUT_READ)
111  {
112  so->read_buf = alloc_buf(512);
113  }
114  }
115  else
116  {
117  msg(M_WARN, "Note: cannot open %s for %s", filename, print_status_mode(so->flags));
118  so->errors = true;
119  }
120  }
121  else
122  {
123  so->flags = STATUS_OUTPUT_WRITE;
124  }
125 
126  if ((so->flags & STATUS_OUTPUT_WRITE) && refresh_freq > 0)
127  {
128  event_timeout_init(&so->et, refresh_freq, 0);
129  }
130  }
131  return so;
132 }

160 {

162  {
163 #if defined(HAVE_FTRUNCATE)
164  {

166  if (ftruncate(so->fd, off) != 0)

169  }
170  }
171 #elif defined(HAVE_CHSIZE)
172  {
173  const long off = (long) lseek(so->fd, (off_t)0, SEEK_CUR);
174  chsize(so->fd, off);
175  }
176 #else /* if defined(HAVE_FTRUNCATE) */
177 #warning both ftruncate and chsize functions appear to be missing from this OS
178 #endif
179 
180  /* clear read buffer */
181  if (buf_defined(&so->read_buf))
182  {
183  ASSERT(buf_init(&so->read_buf, 0));
184  }
185  }

225 void
226 status_printf(struct status_output *so, const char *format, ...)
227 {
228  if (so && (so->flags & STATUS_OUTPUT_WRITE))
229  {
230  char buf[STATUS_PRINTF_MAXLEN+2]; /* leave extra bytes for CR, LF */
231  va_list arglist;
232  int stat;
233 
234  va_start(arglist, format);
235  stat = vsnprintf(buf, STATUS_PRINTF_MAXLEN, format, arglist);
236  va_end(arglist);
237  buf[STATUS_PRINTF_MAXLEN - 1] = 0;
238 
239  if (stat < 0 || stat >= STATUS_PRINTF_MAXLEN)
240  {
241  so->errors = true;
242  }
243 
244  if (so->msglevel >= 0 && !so->errors)
245  {
246  msg(so->msglevel, "%s", buf);
247  }
248 
249  if (so->fd >= 0 && !so->errors)
250  {
251  int len;
252  strcat(buf, "\n");
253  len = strlen(buf);
254  if (len > 0)
255  {
256  if (write(so->fd, buf, len) != len)
257  {
258  so->errors = true;
259  }
260  }
261  }
262 
263  if (so->vout && !so->errors)
264  {
265  chomp(buf);
266  (*so->vout->func)(so->vout->arg, so->vout->flags_default, buf);
267  }
268  }

271 bool
272 status_read(struct status_output *so, struct buffer *buf)
273 {
274  bool ret = false;
275 
276  if (so && so->fd >= 0 && (so->flags & STATUS_OUTPUT_READ))
277  {
278  ASSERT(buf_defined(&so->read_buf));
279  ASSERT(buf_defined(buf));
280  while (true)
281  {
282  const int c = buf_read_u8(&so->read_buf);
283 
284  /* read more of file into buffer */
285  if (c == -1)
286  {
287  int len;
288 
289  ASSERT(buf_init(&so->read_buf, 0));
290  len = read(so->fd, BPTR(&so->read_buf), BCAP(&so->read_buf));
291  if (len <= 0)
292  {
293  break;
294  }
295 
296  ASSERT(buf_inc_len(&so->read_buf, len));
297  continue;
298  }
299 
300  ret = true;
301 
302  if (c == '\r')
303  {
304  continue;
305  }
306 
307  if (c == '\n')
308  {
309  break;
310  }
311 
312  buf_write_u8(buf, c);
313  }
314 
315  buf_null_terminate(buf);
316  }
317 
318  return ret;
