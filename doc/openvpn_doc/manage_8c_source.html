309 static void
310 virtual_output_callback_func(void *arg, const unsigned int flags, const char *str)
311 {
312  struct management *man = (struct management *) arg;
313  static int recursive_level = 0; /* GLOBAL */
314 
315 #define AF_DID_PUSH (1<<0)
316 #define AF_DID_RESET (1<<1)
317  if (recursive_level < 5) /* limit recursion */
318  {
319  struct gc_arena gc = gc_new();
320  struct log_entry e;
321  const char *out = NULL;
322  unsigned int action_flags = 0;
323 
324  ++recursive_level;
325 
326  CLEAR(e);
327  update_time();
328  e.timestamp = now;
329  e.u.msg_flags = flags;
330  e.string = str;
331 
332  if (flags & M_FATAL)
333  {
334  man->persist.standalone_disabled = false;
335  }
336 
337  if (flags != M_CLIENT)
338  {
339  log_history_add(man->persist.log, &e);
340  }
341 
342  if (!man_password_needed(man))
343  {
344  if (flags == M_CLIENT)
345  {
346  out = log_entry_print(&e, LOG_PRINT_CRLF, &gc);
347  }
348  else if (man->connection.log_realtime)
349  {
350  out = log_entry_print(&e, LOG_PRINT_INT_DATE
351  | LOG_PRINT_MSG_FLAGS
352  | LOG_PRINT_LOG_PREFIX
353  | LOG_PRINT_CRLF, &gc);
354  }
355  if (out)
356  {
357  man_output_list_push_str(man, out);
358  action_flags |= AF_DID_PUSH;
359  }
360  if (flags & M_FATAL)
361  {
362  out = log_entry_print(&e, LOG_FATAL_NOTIFY|LOG_PRINT_CRLF, &gc);
363  if (out)
364  {
365  man_output_list_push_str(man, out);
366  action_flags |= (AF_DID_PUSH|AF_DID_RESET);
367  }
368  }
369  }
370 
371  gc_free(&gc);
372 
373  if (action_flags & AF_DID_PUSH)
374  {
375  man_output_list_push_finalize(man);
376  }
377  if (action_flags & AF_DID_RESET)
378  {
379  man_reset_client_socket(man, true);
380  }
381 
382  --recursive_level;
383  }
384  else
385  {
386  /* cannot use msg here */
387  printf("virtual_output: message to management interface "
388  "dropped due to recursion: <%s>\n", str);
389  }
390 }

396 static int
397 man_mod_signal(const struct management *man, const int signum)
398 {
399  const unsigned int flags = man->settings.mansig;
400  int s = signum;
401  if (s == SIGUSR1)
402  {
403  if (flags & MANSIG_MAP_USR1_TO_HUP)
404  {
405  s = SIGHUP;
406  }
407  if (flags & MANSIG_MAP_USR1_TO_TERM)
408  {
409  s = SIGTERM;
410  }
411  }
412  if (flags & MANSIG_IGNORE_USR1_HUP)
413  {
414  if (s == SIGHUP || s == SIGUSR1)
415  {
416  s = -1;
417  }
418  }
419  return s;
420 }

507 static void
508 man_kill(struct management *man, const char *victim)
509 {
510  struct gc_arena gc = gc_new();
511 
512  if (man->persist.callback.kill_by_cn && man->persist.callback.kill_by_addr)
513  {
514  struct buffer buf;
515  char p1[128];
516  char p2[128];
517  int n_killed;
518 
519  buf_set_read(&buf, (uint8_t *) victim, strlen(victim) + 1);
520  buf_parse(&buf, ':', p1, sizeof(p1));
521  buf_parse(&buf, ':', p2, sizeof(p2));
522 
523  if (strlen(p1) && strlen(p2))
524  {
525  /* IP:port specified */
526  bool status;
527  const in_addr_t addr = getaddr(GETADDR_HOST_ORDER|GETADDR_MSG_VIRT_OUT, p1, 0, &status, NULL);
528  if (status)
529  {
530  const int port = atoi(p2);
531  if (port > 0 && port < 65536)
532  {
533  n_killed = (*man->persist.callback.kill_by_addr)(man->persist.callback.arg, addr, port);
534  if (n_killed > 0)
535  {
536  msg(M_CLIENT, "SUCCESS: %d client(s) at address %s:%d killed",
537  n_killed,
538  print_in_addr_t(addr, 0, &gc),
539  port);
540  }
541  else
542  {
543  msg(M_CLIENT, "ERROR: client at address %s:%d not found",
544  print_in_addr_t(addr, 0, &gc),
545  port);
546  }
547  }
548  else
549  {
550  msg(M_CLIENT, "ERROR: port number is out of range: %s", p2);
551  }
552  }
553  else
554  {
555  msg(M_CLIENT, "ERROR: error parsing IP address: %s", p1);
556  }
557  }
558  else if (strlen(p1))
559  {
560  /* common name specified */
561  n_killed = (*man->persist.callback.kill_by_cn)(man->persist.callback.arg, p1);
562  if (n_killed > 0)
563  {
564  msg(M_CLIENT, "SUCCESS: common name '%s' found, %d client(s) killed", p1, n_killed);
565  }
566  else
567  {
568  msg(M_CLIENT, "ERROR: common name '%s' not found", p1);
569  }
570  }
571  else
572  {
573  msg(M_CLIENT, "ERROR: kill parse");
574  }
575  }
576  else
577  {
578  msg(M_CLIENT, "ERROR: The 'kill' command is not supported by the current daemon mode");
579  }
580 
581  gc_free(&gc);
582 }

589 man_history(struct management *man,
590  const char *parm,
591  const char *type,
592  struct log_history *log,
593  bool *realtime,
594  const unsigned int lep_flags)
595 {
596  struct gc_arena gc = gc_new();
597  int n = 0;
598 
599  if (streq(parm, "on"))
600  {
601  *realtime = true;

603  }
604  else if (streq(parm, "off"))
605  {
606  *realtime = false;
607  msg(M_CLIENT, "SUCCESS: real-time %s notification set to OFF", type);
608  }
609  else if (streq(parm, "all") || (n = atoi(parm)) > 0)

612  const int start = (n ? n : size) - 1;
613  int i;
614 
615  for (i = start; i >= 0; --i)
616  {
617  const struct log_entry *e = log_history_ref(log, i);
618  if (e)
619  {
620  const char *out = log_entry_print(e, lep_flags, &gc);
621  virtual_output_callback_func(man, M_CLIENT, out);
622  }
623  }
624  msg(M_CLIENT, "END");
625  }
626  else
627  {
628  msg(M_CLIENT, "ERROR: %s parameter must be 'on' or 'off' or some number n or 'all'", type);
629  }
630 
631  gc_free(&gc);
632 }
