95 char *
96 mutate_ncp_cipher_list(const char *list, struct gc_arena *gc)
97 {
98  bool error_found = false;
99 
100  struct buffer new_list = alloc_buf(MAX_NCP_CIPHERS_LENGTH);
101 
102  char *const tmp_ciphers = string_alloc(list, NULL);
103  const char *token = strtok(tmp_ciphers, ":");
104  while (token)
105  {
106  /*
107  * Going through a roundtrip by using cipher_kt_get/cipher_kt_name
108  * (and translate_cipher_name_from_openvpn/
109  * translate_cipher_name_to_openvpn) also normalises the cipher name,
110  * e.g. replacing AeS-128-gCm with AES-128-GCM
111  *
112  * ciphers that have ? in front of them are considered optional and
113  * OpenVPN will only warn if they are not found (and remove them from
114  * the list)
115  */
116 
117  bool optional = false;
118  if (token[0] == '?')
119  {
120  token++;
121  optional = true;
122  }
123  const cipher_kt_t *ktc = cipher_kt_get(token);
124  if (strcmp(token, "none") == 0)
125  {
126  msg(M_WARN, "WARNING: cipher 'none' specified for --data-ciphers. "
127  "This allows negotiation of NO encryption and "
128  "tunnelled data WILL then be transmitted in clear text "
129  "over the network! "
130  "PLEASE DO RECONSIDER THIS SETTING!");
131  }
132  if (!ktc && strcmp(token, "none") != 0)
133  {
134  const char* optstr = optional ? "optional ": "";
135  msg(M_WARN, "Unsupported %scipher in --data-ciphers: %s", optstr, token);
136  error_found = error_found || !optional;
137  }
138  else
139  {
140  const char *ovpn_cipher_name = cipher_kt_name(ktc);
141  if (ktc == NULL)
142  {
143  /* NULL resolves to [null-cipher] but we need none for
144  * data-ciphers */
145  ovpn_cipher_name = "none";
146  }
147 
148  if (buf_len(&new_list)> 0)
149  {
150  /* The next if condition ensure there is always space for
151  * a :
152  */
153  buf_puts(&new_list, ":");
154  }
155 
156  /* Ensure buffer has capacity for cipher name + : + \0 */
157  if (!(buf_forward_capacity(&new_list) >
158  strlen(ovpn_cipher_name) + 2))
159  {
160  msg(M_WARN, "Length of --data-ciphers is over the "
161  "limit of 127 chars");
162  error_found = true;
163  }
164  else
165  {
166  buf_puts(&new_list, ovpn_cipher_name);
167  }
168  }
169  token = strtok(NULL, ":");
170  }
171 
172 
173 
174  char *ret = NULL;
175  if (!error_found && buf_len(&new_list) > 0)
176  {
177  buf_null_terminate(&new_list);
178  ret = string_alloc(buf_str(&new_list), gc);
179  }
180  free(tmp_ciphers);
181  free_buf(&new_list);
182 
183  return ret;
184 }

188 {

190  char *tmp_ciphers_orig = tmp_ciphers;

206 const char *
207 tls_peer_ncp_list(const char *peer_info, struct gc_arena *gc)
208 {
209  /* Check if the peer sends the IV_CIPHERS list */
210  const char *ncp_ciphers_start;
211  if (peer_info && (ncp_ciphers_start = strstr(peer_info, "IV_CIPHERS=")))
212  {
213  ncp_ciphers_start += strlen("IV_CIPHERS=");
214  const char *ncp_ciphers_end = strstr(ncp_ciphers_start, "\n");
215  if (!ncp_ciphers_end)
216  {
217  /* IV_CIPHERS is at end of the peer_info list and no '\n'
218  * follows */
219  ncp_ciphers_end = ncp_ciphers_start + strlen(ncp_ciphers_start);
220  }
221 
222  char *ncp_ciphers_peer = string_alloc(ncp_ciphers_start, gc);
223  /* NULL terminate the copy at the right position */
224  ncp_ciphers_peer[ncp_ciphers_end - ncp_ciphers_start] = '\0';
225  return ncp_ciphers_peer;
226 
227  }
228  else if (tls_peer_info_ncp_ver(peer_info)>=2)
229  {
230  /* If the peer announces IV_NCP=2 then it supports the AES GCM
231  * ciphers */
232  return "AES-256-GCM:AES-128-GCM";
233  }
234  else
235  {
236  return "";
237  }
238 }

240 char *
241 ncp_get_best_cipher(const char *server_list, const char *peer_info,
242  const char *remote_cipher, struct gc_arena *gc)
243 {
244  /*
245  * The gc of the parameter is tied to the VPN session, create a
246  * short lived gc arena that is only valid for the duration of
247  * this function
248  */
249 
250  struct gc_arena gc_tmp = gc_new();
251 
252  const char *peer_ncp_list = tls_peer_ncp_list(peer_info, &gc_tmp);
253 
254  /* non-NCP client without OCC? "assume nothing" */
255  /* For client doing the newer version of NCP (that send IV_CIPHER)
256  * we cannot assume that they will accept remote_cipher */
257  if (remote_cipher == NULL ||
258  (peer_info && strstr(peer_info, "IV_CIPHERS=")))
259  {
260  remote_cipher = "";
261  }
262 
263  char *tmp_ciphers = string_alloc(server_list, &gc_tmp);
264 
265  const char *token;
266  while ((token = strsep(&tmp_ciphers, ":")))
267  {
268  if (tls_item_in_cipher_list(token, peer_ncp_list)
269  || streq(token, remote_cipher))
270  {
271  break;
272  }
273  }
274 
275  char *ret = NULL;
276  if (token != NULL)
277  {
278  ret = string_alloc(token, gc);
279  }
280 
281  gc_free(&gc_tmp);
282  return ret;
283 }

307 bool
308 check_pull_client_ncp(struct context *c, const int found)
309 {
310  if (found & OPT_P_NCP)
311  {
312  msg(D_PUSH, "OPTIONS IMPORT: data channel crypto options modified");
313  return true;
314  }
315 
316  if (!c->options.ncp_enabled)
317  {
318  return true;
319  }
320  /* If the server did not push a --cipher, we will switch to the
321  * remote cipher if it is in our ncp-ciphers list */
322  if(tls_poor_mans_ncp(&c->options, c->c2.tls_multi->remote_ciphername))
323  {
324  return true;
325  }
326 
327  /* We could not figure out the peer's cipher but we have fallback
328  * enabled */
329  if (!c->c2.tls_multi->remote_ciphername && c->options.enable_ncp_fallback)
330  {
331  return true;
332  }
333 
334  /* We failed negotiation, give appropiate error message */
335  if (c->c2.tls_multi->remote_ciphername)
336  {
337  msg(D_TLS_ERRORS, "OPTIONS ERROR: failed to negotiate "
338  "cipher with server. Add the server's "
339  "cipher ('%s') to --data-ciphers (currently '%s') if "
340  "you want to connect to this server.",
341  c->c2.tls_multi->remote_ciphername,
342  c->options.ncp_ciphers);
343  return false;
344 
345  }
346  else
347  {
348  msg(D_TLS_ERRORS, "OPTIONS ERROR: failed to negotiate "
349  "cipher with server. Configure "
350  "--data-ciphers-fallback if you want to connect "
351  "to this server.");
352  return false;
353  }
