591 /*
592  * Compute earliest timeout expiry from the set of
593  * all instances. Output:
594  *
595  * m->earliest_wakeup : instance needing the earliest service.
596  * dest : earliest timeout as a delta in relation
597  * to current time.
598  */
599 static inline void
600 multi_get_timeout(struct multi_context *m, struct timeval *dest)
601 {
602  struct timeval tv, current;
603 
604  CLEAR(tv);
605  m->earliest_wakeup = (struct multi_instance *) schedule_get_earliest_wakeup(m->schedule, &tv);
606  if (m->earliest_wakeup)
607  {
608  ASSERT(!openvpn_gettimeofday(&current, NULL));
609  tv_delta(dest, &current, &tv);
610  if (dest->tv_sec >= REAP_MAX_WAKEUP)
611  {
612  m->earliest_wakeup = NULL;
613  dest->tv_sec = REAP_MAX_WAKEUP;
614  dest->tv_usec = 0;
615  }
616  }
617  else
618  {
619  dest->tv_sec = REAP_MAX_WAKEUP;
620  dest->tv_usec = 0;
621  }
622 }
