258 const tls_cipher_name_pair *
259 tls_get_cipher_name_pair(const char *cipher_name, size_t len)
260 {
261  const tls_cipher_name_pair *pair = tls_cipher_name_translation_table;
262 
263  while (pair->openssl_name != NULL)
264  {
265  if ((strlen(pair->openssl_name) == len && 0 == memcmp(cipher_name, pair->openssl_name, len))
266  || (strlen(pair->iana_name) == len && 0 == memcmp(cipher_name, pair->iana_name, len)))
267  {
268  return pair;
269  }
270  pair++;
271  }
272 
273  /* No entry found, return NULL */
274  return NULL;
275 }

314 static void
315 tls_init_control_channel_frame_parameters(const struct frame *data_channel_frame,
316  struct frame *frame)
317 {
318  /*
319  * frame->extra_frame is already initialized with tls_auth buffer requirements,
320  * if --tls-auth is enabled.
321  */
322 
323  /* inherit link MTU and extra_link from data channel */
324  frame->link_mtu = data_channel_frame->link_mtu;
325  frame->extra_link = data_channel_frame->extra_link;
326 
327  /* set extra_frame */
328  tls_adjust_frame_parameters(frame);

331

333  ASSERT(TUN_LINK_DELTA(frame) < min_int(frame->link_mtu, 1250));
334  frame->link_mtu_dynamic = min_int(frame->link_mtu, 1250) - TUN_LINK_DELTA(frame);

511  * If version string is not recognized and extra == "or-highest",

518  if (!strcmp(vstr, "1.0") && TLS_VER_1_0 <= max_version)
519  {

521  }

523  {

555 static void
556 tls_ctx_reload_crl(struct tls_root_ctx *ssl_ctx, const char *crl_file,
557  bool crl_file_inline)
558 {
559  /* if something goes wrong with stat(), we'll store 0 as mtime */
560  platform_stat_t crl_stat = {0};
561 
562  /*
563  * an inline CRL can't change at runtime, therefore there is no need to
564  * reload it. It will be reloaded upon config change + SIGHUP.
565  * Use always '1' as dummy timestamp in this case: it will trigger the
566  * first load, but will prevent any future reload.
567  */
568  if (crl_file_inline)
569  {
570  crl_stat.st_mtime = 1;
571  }
572  else if (platform_stat(crl_file, &crl_stat) < 0)
573  {
574  /* If crl_last_mtime is zero, the CRL file has not been read before. */
575  if (ssl_ctx->crl_last_mtime == 0)
576  {
577  msg(M_FATAL, "ERROR: Failed to stat CRL file during initialization, exiting.");
578  }
579  else
580  {
581  msg(M_WARN, "WARNING: Failed to stat CRL file, not reloading CRL.");
582  }
583  return;
584  }
585 
586  /*
587  * Store the CRL if this is the first time or if the file was changed since
588  * the last load.
589  * Note: Windows does not support tv_nsec.
590  */
591  if ((ssl_ctx->crl_last_size == crl_stat.st_size)
592  && (ssl_ctx->crl_last_mtime == crl_stat.st_mtime))
593  {
594  return;
595  }
596 
597  ssl_ctx->crl_last_mtime = crl_stat.st_mtime;
598  ssl_ctx->crl_last_size = crl_stat.st_size;
599  backend_tls_ctx_reload_crl(ssl_ctx, crl_file, crl_file_inline);
600 }

602 /*
603  * Initialize SSL context.
604  * All files are in PEM format.
605  */
606 void
607 init_ssl(const struct options *options, struct tls_root_ctx *new_ctx, bool in_chroot)
608 {
609  ASSERT(NULL != new_ctx);
610 
611  tls_clear_error();
612 
613  if (options->tls_server)
614  {
615  tls_ctx_server_new(new_ctx);
616 
617  if (options->dh_file)
618  {
619  tls_ctx_load_dh_params(new_ctx, options->dh_file,
620  options->dh_file_inline);
621  }
622  }
623  else /* if client */
624  {
625  tls_ctx_client_new(new_ctx);
626  }
627 
628  /* Restrict allowed certificate crypto algorithms */
629  tls_ctx_set_cert_profile(new_ctx, options->tls_cert_profile);
630 
631  /* Allowable ciphers */
632  /* Since @SECLEVEL also influences loading of certificates, set the
633  * cipher restrictions before loading certificates */
634  tls_ctx_restrict_ciphers(new_ctx, options->cipher_list);
635  tls_ctx_restrict_ciphers_tls13(new_ctx, options->cipher_list_tls13);
636 
637  /* Set the allow groups/curves for TLS if we want to override them */
638  if (options->tls_groups)
639  {
640  tls_ctx_set_tls_groups(new_ctx, options->tls_groups);
641  }
642 
643  if (!tls_ctx_set_options(new_ctx, options->ssl_flags))
644  {
645  goto err;
646  }
647 
648  if (options->pkcs12_file)
649  {
650  if (0 != tls_ctx_load_pkcs12(new_ctx, options->pkcs12_file,
651  options->pkcs12_file_inline, !options->ca_file))
652  {
653  goto err;
654  }
655  }
656 #ifdef ENABLE_PKCS11
657  else if (options->pkcs11_providers[0])
658  {
659  if (!tls_ctx_use_pkcs11(new_ctx, options->pkcs11_id_management, options->pkcs11_id))
660  {
661  msg(M_WARN, "Cannot load certificate \"%s\" using PKCS#11 interface",
662  options->pkcs11_id);
663  goto err;
664  }
665  }
666 #endif
667 #ifdef ENABLE_CRYPTOAPI
668  else if (options->cryptoapi_cert)
669  {
670  tls_ctx_load_cryptoapi(new_ctx, options->cryptoapi_cert);
671  }
672 #endif
673 #ifdef ENABLE_MANAGEMENT
674  else if (options->management_flags & MF_EXTERNAL_CERT)
675  {
676  char *cert = management_query_cert(management,
677  options->management_certificate);
678  tls_ctx_load_cert_file(new_ctx, cert, true);
679  free(cert);
680  }
681 #endif
682  else if (options->cert_file)
683  {
684  tls_ctx_load_cert_file(new_ctx, options->cert_file, options->cert_file_inline);
685  }
686 
687  if (options->priv_key_file)
688  {
689  if (0 != tls_ctx_load_priv_file(new_ctx, options->priv_key_file,
690  options->priv_key_file_inline))
691  {
692  goto err;
693  }
694  }
695 #ifdef ENABLE_MANAGEMENT
696  else if (options->management_flags & MF_EXTERNAL_KEY)
697  {
698  if (tls_ctx_use_management_external_key(new_ctx))
699  {
700  msg(M_WARN, "Cannot initialize mamagement-external-key");
701  goto err;
702  }
703  }
704 #endif
705 
706  if (options->ca_file || options->ca_path)
707  {
708  tls_ctx_load_ca(new_ctx, options->ca_file, options->ca_file_inline,
709  options->ca_path, options->tls_server);
710  }
711 
712  /* Load extra certificates that are part of our own certificate
713  * chain but shouldn't be included in the verify chain */
714  if (options->extra_certs_file)
715  {
716  tls_ctx_load_extra_certs(new_ctx, options->extra_certs_file, options->extra_certs_file_inline);
717  }
718 
719  /* Check certificate notBefore and notAfter */
720  tls_ctx_check_cert_time(new_ctx);
721 
722  /* Read CRL */
723  if (options->crl_file && !(options->ssl_flags & SSLF_CRL_VERIFY_DIR))
724  {
725  /* If we're running with the chroot option, we may run init_ssl() before
726  * and after chroot-ing. We can use the crl_file path as-is if we're
727  * not going to chroot, or if we already are inside the chroot.
728  *
729  * If we're going to chroot later, we need to prefix the path of the
730  * chroot directory to crl_file.
731  */
732  if (!options->chroot_dir || in_chroot || options->crl_file_inline)
733  {
734  tls_ctx_reload_crl(new_ctx, options->crl_file, options->crl_file_inline);
735  }
736  else
737  {
738  struct gc_arena gc = gc_new();
739  struct buffer crl_file_buf = prepend_dir(options->chroot_dir, options->crl_file, &gc);
740  tls_ctx_reload_crl(new_ctx, BSTR(&crl_file_buf), options->crl_file_inline);
741  gc_free(&gc);
742  }
743  }
744 
745  /* Once keys and cert are loaded, load ECDH parameters */
746  if (options->tls_server)
747  {
748  tls_ctx_load_ecdh_params(new_ctx, options->ecdh_curve);
749  }
750 
751 #ifdef ENABLE_CRYPTO_MBEDTLS
752  /* Personalise the random by mixing in the certificate */
753  tls_ctx_personalise_random(new_ctx);
754 #endif
755 
756  tls_clear_error();
757  return;
758 
759 err:
760  tls_clear_error();
761  tls_ctx_free(new_ctx);
762  return;
763 }

1387  * Move a packet authentication HMAC + related fields to or from the front
1388  * of the buffer so it can be processed by encrypt/decrypt.
1389  */
1390 
1391 /*
1392  * Dependent on hmac size, opcode size, and session_id size.
1393  * Will assert if too small.
1394  */
1395 #define SWAP_BUF_SIZE 256
1396 
1397 static bool
1398 swap_hmac(struct buffer *buf, const struct crypto_options *co, bool incoming)
1399 {
1400  ASSERT(co);
1401 
1402  const struct key_ctx *ctx = (incoming ? &co->key_ctx_bi.decrypt :
1403  &co->key_ctx_bi.encrypt);
1404  ASSERT(ctx->hmac);
1405 
1406  {
1407  /* hmac + packet_id (8 bytes) */
1408  const int hmac_size = hmac_ctx_size(ctx->hmac) + packet_id_size(true);
1409 
1410  /* opcode + session_id */
1411  const int osid_size = 1 + SID_SIZE;
1412 
1413  int e1, e2;
1414  uint8_t *b = BPTR(buf);
1415  uint8_t buf1[SWAP_BUF_SIZE];
1416  uint8_t buf2[SWAP_BUF_SIZE];
1417 
1418  if (incoming)
1419  {
1420  e1 = osid_size;
1421  e2 = hmac_size;
1422  }
1423  else
1424  {
1425  e1 = hmac_size;
1426  e2 = osid_size;
1427  }
1428 
1429  ASSERT(e1 <= SWAP_BUF_SIZE && e2 <= SWAP_BUF_SIZE);
1430 
1431  if (buf->len >= e1 + e2)
1432  {
1433  memcpy(buf1, b, e1);
1434  memcpy(buf2, b + e1, e2);
1435  memcpy(b, buf2, e2);
1436  memcpy(b + e2, buf1, e1);
1437  return true;
1438  }
1439  else
1440  {
1441  return false;
1442  }
1443  }
1444 }

1448 /*
1449  * Write a control channel authentication record.
1450  */
1451 static void
1452 write_control_auth(struct tls_session *session,
1453  struct key_state *ks,
1454  struct buffer *buf,
1455  struct link_socket_actual **to_link_addr,
1456  int opcode,
1457  int max_ack,
1458  bool prepend_ack)

1484  ASSERT(buf_write(&session->tls_wrap.work, &header, sizeof(header)));
1485  ASSERT(session_id_write(&session->session_id, &session->tls_wrap.work));

1490  }

1495  session->tls_wrap.tls_crypt_v2_wkc))
1496  {
1497  msg(D_TLS_ERRORS, "Could not append tls-crypt-v2 client key");
1498  buf->len = 0;
1499  return;
1500  }
1501  }
1502 
1503  /* Don't change the original data in buf, it's used by the reliability
1504  * layer to resend on failure. */
1505  *buf = session->tls_wrap.work;
1506  }

1508 }

1510 /*
1511  * Read a control channel authentication record.
1512  */
1513 static bool
1514 read_control_auth(struct buffer *buf,
1515  struct tls_wrap_ctx *ctx,
1516  const struct link_socket_actual *from,
1517  const struct tls_options *opt)
1518 {
1519  struct gc_arena gc = gc_new();
1520  bool ret = false;
1521 
1522  const uint8_t opcode = *(BPTR(buf)) >> P_OPCODE_SHIFT;
1523  if (opcode == P_CONTROL_HARD_RESET_CLIENT_V3
1524  && !tls_crypt_v2_extract_client_key(buf, ctx, opt))
1525  {
1526  msg(D_TLS_ERRORS,
1527  "TLS Error: can not extract tls-crypt-v2 client key from %s",
1528  print_link_socket_actual(from, &gc));
1529  goto cleanup;
1530  }
1531 
1532  if (ctx->mode == TLS_WRAP_AUTH)
1533  {
1534  struct buffer null = clear_buf();
1535 
1536  /* move the hmac record to the front of the packet */
1537  if (!swap_hmac(buf, &ctx->opt, true))
1538  {
1539  msg(D_TLS_ERRORS,
1540  "TLS Error: cannot locate HMAC in incoming packet from %s",
1541  print_link_socket_actual(from, &gc));
1542  gc_free(&gc);
1543  return false;
1544  }
1545 
1546  /* authenticate only (no decrypt) and remove the hmac record
1547  * from the head of the buffer */
1548  openvpn_decrypt(buf, null, &ctx->opt, NULL, BPTR(buf));
1549  if (!buf->len)
1550  {
1551  msg(D_TLS_ERRORS,
1552  "TLS Error: incoming packet authentication failed from %s",
1553  print_link_socket_actual(from, &gc));
1554  goto cleanup;
1555  }
1556 
1557  }
1558  else if (ctx->mode == TLS_WRAP_CRYPT)
1559  {
1560  struct buffer tmp = alloc_buf_gc(buf_forward_capacity_total(buf), &gc);
1561  if (!tls_crypt_unwrap(buf, &tmp, &ctx->opt))
1562  {
1563  msg(D_TLS_ERRORS, "TLS Error: tls-crypt unwrapping failed from %s",
1564  print_link_socket_actual(from, &gc));
1565  goto cleanup;
1566  }
1567  ASSERT(buf_init(buf, buf->offset));
1568  ASSERT(buf_copy(buf, &tmp));
1569  buf_clear(&tmp);
1570  }
1571  else if (ctx->tls_crypt_v2_server_key.cipher)
1572  {
1573  /* If tls-crypt-v2 is enabled, require *some* wrapping */
1574  msg(D_TLS_ERRORS, "TLS Error: could not determine wrapping from %s",
1575  print_link_socket_actual(from, &gc));
1576  /* TODO Do we want to support using tls-crypt-v2 and no control channel
1577  * wrapping at all simultaneously? That would allow server admins to
1578  * upgrade clients one-by-one without running a second instance, but we
1579  * should not enable it by default because it breaks DoS-protection.
1580  * So, add something like --tls-crypt-v2-allow-insecure-fallback ? */
1581  goto cleanup;
1582  }
1583 
1584  if (ctx->mode == TLS_WRAP_NONE || ctx->mode == TLS_WRAP_AUTH)
1585  {
1586  /* advance buffer pointer past opcode & session_id since our caller
1587  * already read it */
1588  buf_advance(buf, SID_SIZE + 1);
1589  }
1590 
1591  ret = true;
1592 cleanup:
1593  gc_free(&gc);
1594  return ret;
1595 }

1634 /*
1635  * Generate the hash required by for the \c tls1_PRF function.
1636  *
1637  * @param md_kt Message digest to use
1638  * @param sec Secret to base the hash on
1639  * @param sec_len Length of the secret
1640  * @param seed Seed to hash
1641  * @param seed_len Length of the seed
1642  * @param out Output buffer
1643  * @param olen Length of the output buffer
1644  */
1645 static void
1646 tls1_P_hash(const md_kt_t *md_kt,
1647  const uint8_t *sec,
1648  int sec_len,
1649  const uint8_t *seed,
1650  int seed_len,
1651  uint8_t *out,
1652  int olen)
1653 {
1654  struct gc_arena gc = gc_new();
1655  uint8_t A1[MAX_HMAC_KEY_LENGTH];
1656 
1657 #ifdef ENABLE_DEBUG
1658  const int olen_orig = olen;
1659  const uint8_t *out_orig = out;
1660 #endif
1661 
1662  hmac_ctx_t *ctx = hmac_ctx_new();
1663  hmac_ctx_t *ctx_tmp = hmac_ctx_new();
1664 
1665  dmsg(D_SHOW_KEY_SOURCE, "tls1_P_hash sec: %s", format_hex(sec, sec_len, 0, &gc));
1666  dmsg(D_SHOW_KEY_SOURCE, "tls1_P_hash seed: %s", format_hex(seed, seed_len, 0, &gc));
1667 
1668  int chunk = md_kt_size(md_kt);
1669  unsigned int A1_len = md_kt_size(md_kt);
1670 
1671  hmac_ctx_init(ctx, sec, sec_len, md_kt);
1672  hmac_ctx_init(ctx_tmp, sec, sec_len, md_kt);
1673 
1674  hmac_ctx_update(ctx,seed,seed_len);
1675  hmac_ctx_final(ctx, A1);
1676 
1677  for (;; )
1678  {
1679  hmac_ctx_reset(ctx);
1680  hmac_ctx_reset(ctx_tmp);
1681  hmac_ctx_update(ctx,A1,A1_len);
1682  hmac_ctx_update(ctx_tmp,A1,A1_len);
1683  hmac_ctx_update(ctx,seed,seed_len);
1684 
1685  if (olen > chunk)
1686  {
1687  hmac_ctx_final(ctx, out);
1688  out += chunk;
1689  olen -= chunk;
1690  hmac_ctx_final(ctx_tmp, A1); /* calc the next A1 value */
1691  }
1692  else /* last one */
1693  {
1694  hmac_ctx_final(ctx, A1);
1695  memcpy(out,A1,olen);
1696  break;
1697  }
1698  }
1699  hmac_ctx_cleanup(ctx);
1700  hmac_ctx_free(ctx);
1701  hmac_ctx_cleanup(ctx_tmp);
1702  hmac_ctx_free(ctx_tmp);
1703  secure_memzero(A1, sizeof(A1));
1704 
1705  dmsg(D_SHOW_KEY_SOURCE, "tls1_P_hash out: %s", format_hex(out_orig, olen_orig, 0, &gc));
1706  gc_free(&gc);
1707 }

1709 /*
1710  * Use the TLS PRF function for generating data channel keys.
1711  * This code is based on the OpenSSL library.
1712  *
1713  * TLS generates keys as such:
1714  *
1715  * master_secret[48] = PRF(pre_master_secret[48], "master secret",
1716  * ClientHello.random[32] + ServerHello.random[32])
1717  *
1718  * key_block[] = PRF(SecurityParameters.master_secret[48],
1719  * "key expansion",
1720  * SecurityParameters.server_random[32] +
1721  * SecurityParameters.client_random[32]);
1722  *
1723  * Notes:
1724  *
1725  * (1) key_block contains a full set of 4 keys.
1726  * (2) The pre-master secret is generated by the client.
1727  */
1728 static void
1729 tls1_PRF(const uint8_t *label,
1730  int label_len,
1731  const uint8_t *sec,
1732  int slen,
1733  uint8_t *out1,
1734  int olen)
1735 {
1736  struct gc_arena gc = gc_new();
1737  const md_kt_t *md5 = md_kt_get("MD5");
1738  const md_kt_t *sha1 = md_kt_get("SHA1");
1739 
1740  uint8_t *out2 = (uint8_t *) gc_malloc(olen, false, &gc);
1741 
1742  int len = slen/2;
1743  const uint8_t *S1 = sec;
1744  const uint8_t *S2 = &(sec[len]);
1745  len += (slen&1); /* add for odd, make longer */
1746 
1747  tls1_P_hash(md5,S1,len,label,label_len,out1,olen);
1748  tls1_P_hash(sha1,S2,len,label,label_len,out2,olen);
1749 
1750  for (int i = 0; i<olen; i++)
1751  {
1752  out1[i] ^= out2[i];
1753  }
1754 
1755  secure_memzero(out2, olen);
1756 
1757  dmsg(D_SHOW_KEY_SOURCE, "tls1_PRF out[%d]: %s", olen, format_hex(out1, olen, 0, &gc));
1758 
1759  gc_free(&gc);
1760 }
