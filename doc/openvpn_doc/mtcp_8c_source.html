123 static struct multi_instance *
124 multi_create_instance_tcp(struct multi_context *m)
125 {
126  struct gc_arena gc = gc_new();
127  struct multi_instance *mi = NULL;
128  struct hash *hash = m->hash;
129 
130  mi = multi_create_instance(m, NULL);
131  if (mi)
132  {
133  struct hash_element *he;
134  const uint32_t hv = hash_value(hash, &mi->real);
135  struct hash_bucket *bucket = hash_bucket(hash, hv);
136 
137  he = hash_lookup_fast(hash, bucket, &mi->real, hv);
138 
139  if (he)
140  {
141  struct multi_instance *oldmi = (struct multi_instance *) he->value;
142  msg(D_MULTI_LOW, "MULTI TCP: new incoming client address matches existing client address -- new client takes precedence");
143  oldmi->did_real_hash = false;
144  multi_close_instance(m, oldmi, false);
145  he->key = &mi->real;
146  he->value = mi;
147  }
148  else
149  {
150  hash_add_fast(hash, bucket, &mi->real, hv, mi);
151  }
152 
153  mi->did_real_hash = true;
154  }
155 
156 #ifdef ENABLE_DEBUG
157  if (mi)
158  {
159  dmsg(D_MULTI_DEBUG, "MULTI TCP: instance added: %s", mroute_addr_print(&mi->real, &gc));
160  }
161  else
162  {
163  dmsg(D_MULTI_DEBUG, "MULTI TCP: new client instance failed");
164  }
165 #endif
166 
167  gc_free(&gc);
168  ASSERT(!(mi && mi->halt));
169  return mi;
170 }

348 static bool
349 multi_tcp_process_outgoing_link(struct multi_context *m, bool defer, const unsigned int mpp_flags)
350 {
351  struct multi_instance *mi = multi_process_outgoing_link_pre(m);
352  bool ret = true;
353 
354  if (mi)
355  {
356  if (defer || mbuf_defined(mi->tcp_link_out_deferred))
357  {
358  /* save to queue */
359  struct buffer *buf = &mi->context.c2.to_link;
360  if (BLEN(buf) > 0)
361  {
362  struct mbuf_buffer *mb = mbuf_alloc_buf(buf);
363  struct mbuf_item item;
364 
365  set_prefix(mi);
366  dmsg(D_MULTI_TCP, "MULTI TCP: queuing deferred packet");
367  item.buffer = mb;
368  item.instance = mi;
369  mbuf_add_item(mi->tcp_link_out_deferred, &item);
370  mbuf_free_buf(mb);
371  buf_reset(buf);
372  ret = multi_process_post(m, mi, mpp_flags);
373  if (!ret)
374  {
375  mi = NULL;
376  }
377  clear_prefix();
378  }
379  }
380  else
381  {
382  ret = multi_process_outgoing_link_dowork(m, mi, mpp_flags);
383  if (!ret)
384  {
385  mi = NULL;
386  }
387  }
388  }
389  return ret;
390 }

392 static int
393 multi_tcp_wait_lite(struct multi_context *m, struct multi_instance *mi, const int action, bool *tun_input_pending)
394 {
395  struct context *c = multi_tcp_context(m, mi);
396  unsigned int looking_for = 0;
397 
398  dmsg(D_MULTI_DEBUG, "MULTI TCP: multi_tcp_wait_lite a=%s mi=" ptr_format,
399  pract(action),
400  (ptr_type)mi);
401 
402  tv_clear(&c->c2.timeval); /* ZERO-TIMEOUT */
403 
404  switch (action)
405  {
406  case TA_TUN_READ:
407  looking_for = TUN_READ;
408  tun_input_pending = NULL;
409  io_wait(c, IOW_READ_TUN);
410  break;
411 
412  case TA_SOCKET_READ:
413  looking_for = SOCKET_READ;
414  tun_input_pending = NULL;
415  io_wait(c, IOW_READ_LINK);
416  break;
417 
418  case TA_TUN_WRITE:
419  looking_for = TUN_WRITE;
420  tun_input_pending = NULL;
421  c->c2.timeval.tv_sec = 1; /* For some reason, the Linux 2.2 TUN/TAP driver hits this timeout */
422  perf_push(PERF_PROC_OUT_TUN_MTCP);
423  io_wait(c, IOW_TO_TUN);
424  perf_pop();
425  break;
426 
427  case TA_SOCKET_WRITE:
428  looking_for = SOCKET_WRITE;
429  io_wait(c, IOW_TO_LINK|IOW_READ_TUN_FORCE);
430  break;
431 
432  default:
433  msg(M_FATAL, "MULTI TCP: multi_tcp_wait_lite, unhandled action=%d", action);
434  }
435 
436  if (tun_input_pending && (c->c2.event_set_status & TUN_READ))
437  {
438  *tun_input_pending = true;
439  }
440 
441  if (c->c2.event_set_status & looking_for)
442  {
443  return action;
444  }
445  else
446  {
447  switch (action)
448  {
449  /* TCP socket output buffer is full */
450  case TA_SOCKET_WRITE:
451  return TA_SOCKET_WRITE_DEFERRED;
452 
453  /* TUN device timed out on accepting write */
454  case TA_TUN_WRITE:
455  return TA_TUN_WRITE_TIMEOUT;
456  }
457 
458  return TA_UNDEF;
459  }
460 }

462 static struct multi_instance *
463 multi_tcp_dispatch(struct multi_context *m, struct multi_instance *mi, const int action)
464 {
465  const unsigned int mpp_flags = MPP_PRE_SELECT|MPP_RECORD_TOUCH;
466  struct multi_instance *touched = mi;
467  m->mpp_touched = &touched;
468 
469  dmsg(D_MULTI_DEBUG, "MULTI TCP: multi_tcp_dispatch a=%s mi=" ptr_format,
470  pract(action),
471  (ptr_type)mi);
472 
473  switch (action)
474  {
475  case TA_TUN_READ:
476  read_incoming_tun(&m->top);
477  if (!IS_SIG(&m->top))
478  {
479  multi_process_incoming_tun(m, mpp_flags);
480  }
481  break;
482 
483  case TA_SOCKET_READ:
484  case TA_SOCKET_READ_RESIDUAL:
485  ASSERT(mi);
486  ASSERT(mi->context.c2.link_socket);
487  set_prefix(mi);
488  read_incoming_link(&mi->context);
489  clear_prefix();
490  if (!IS_SIG(&mi->context))
491  {
492  multi_process_incoming_link(m, mi, mpp_flags);
493  if (!IS_SIG(&mi->context))
494  {
495  stream_buf_read_setup(mi->context.c2.link_socket);
496  }
497  }
498  break;
499 
500  case TA_TIMEOUT:
501  multi_process_timeout(m, mpp_flags);
502  break;
503 
504  case TA_TUN_WRITE:
505  multi_process_outgoing_tun(m, mpp_flags);
506  break;
507 
508  case TA_TUN_WRITE_TIMEOUT:
509  multi_process_drop_outgoing_tun(m, mpp_flags);
510  break;
511 
512  case TA_SOCKET_WRITE_READY:
513  ASSERT(mi);
514  multi_tcp_process_outgoing_link_ready(m, mi, mpp_flags);
515  break;
516 
517  case TA_SOCKET_WRITE:
518  multi_tcp_process_outgoing_link(m, false, mpp_flags);
519  break;
520 
521  case TA_SOCKET_WRITE_DEFERRED:
522  multi_tcp_process_outgoing_link(m, true, mpp_flags);
523  break;
524 
525  case TA_INITIAL:
526  ASSERT(mi);
527  multi_tcp_set_global_rw_flags(m, mi);
528  multi_process_post(m, mi, mpp_flags);
529  break;
530 
531  default:
532  msg(M_FATAL, "MULTI TCP: multi_tcp_dispatch, unhandled action=%d", action);
533  }
534 
535  m->mpp_touched = NULL;
536  return touched;
537 }

539 static int
540 multi_tcp_post(struct multi_context *m, struct multi_instance *mi, const int action)
541 {
542  struct context *c = multi_tcp_context(m, mi);
543  int newaction = TA_UNDEF;
544 
545 #define MTP_NONE 0
546 #define MTP_TUN_OUT (1<<0)
547 #define MTP_LINK_OUT (1<<1)
548  unsigned int flags = MTP_NONE;
549 
550  if (TUN_OUT(c))
551  {
552  flags |= MTP_TUN_OUT;
553  }
554  if (LINK_OUT(c))
555  {
556  flags |= MTP_LINK_OUT;
557  }
558 
559  switch (flags)
560  {
561  case MTP_TUN_OUT|MTP_LINK_OUT:
562  case MTP_TUN_OUT:
563  newaction = TA_TUN_WRITE;
564  break;
565 
566  case MTP_LINK_OUT:
567  newaction = TA_SOCKET_WRITE;
568  break;
569 
570  case MTP_NONE:
571  if (mi && socket_read_residual(c->c2.link_socket))
572  {
573  newaction = TA_SOCKET_READ_RESIDUAL;
574  }
575  else
576  {
577  multi_tcp_set_global_rw_flags(m, mi);
578  }
579  break;
580 
581  default:
582  {
583  struct gc_arena gc = gc_new();
584  msg(M_FATAL, "MULTI TCP: multi_tcp_post bad state, mi=%s flags=%d",
585  multi_instance_string(mi, false, &gc),
586  flags);
587  gc_free(&gc);
588  break;
589  }
590  }
591 
592  dmsg(D_MULTI_DEBUG, "MULTI TCP: multi_tcp_post %s -> %s",
593  pract(action),
594  pract(newaction));
595 
596  return newaction;
597 }

599 static void
600 multi_tcp_action(struct multi_context *m, struct multi_instance *mi, int action, bool poll)
601 {
602  bool tun_input_pending = false;
603 
604  do
605  {
606  dmsg(D_MULTI_DEBUG, "MULTI TCP: multi_tcp_action a=%s p=%d",
607  pract(action),
608  poll);
609 
610  /*
611  * If TA_SOCKET_READ_RESIDUAL, it means we still have pending
612  * input packets which were read by a prior TCP recv.
613  *
614  * Otherwise do a "lite" wait, which means we wait with 0 timeout
615  * on I/O events only related to the current instance, not
616  * the big list of events.
617  *
618  * On our first pass, poll will be false because we already know
619  * that input is available, and to call io_wait would be redundant.
620  */
621  if (poll && action != TA_SOCKET_READ_RESIDUAL)
622  {
623  const int orig_action = action;
624  action = multi_tcp_wait_lite(m, mi, action, &tun_input_pending);
625  if (action == TA_UNDEF)
626  {
627  msg(M_FATAL, "MULTI TCP: I/O wait required blocking in multi_tcp_action, action=%d", orig_action);
628  }
629  }
630 
631  /*
632  * Dispatch the action
633  */
634  {
635  struct multi_instance *touched = multi_tcp_dispatch(m, mi, action);
636 
637  /*
638  * Signal received or TCP connection
639  * reset by peer?
640  */
641  if (touched && IS_SIG(&touched->context))
642  {
643  if (mi == touched)
644  {
645  mi = NULL;
646  }
647  multi_close_instance_on_signal(m, touched);
648  }
649  }
650 
651  /*
652  * If dispatch produced any pending output
653  * for a particular instance, point to
654  * that instance.
655  */
656  if (m->pending)
657  {
658  mi = m->pending;
659  }
660 
661  /*
662  * Based on the effects of the action,
663  * such as generating pending output,
664  * possibly transition to a new action state.
665  */
666  action = multi_tcp_post(m, mi, action);
667 
668  /*
669  * If we are finished processing the original action,
670  * check if we have any TUN input. If so, transition
671  * our action state to processing this input.
672  */
673  if (tun_input_pending && action == TA_UNDEF)
674  {
675  action = TA_TUN_READ;
676  mi = NULL;
677  tun_input_pending = false;
678  poll = false;
679  }
680  else
681  {
682  poll = true;
683  }
684 
685  } while (action != TA_UNDEF);
686 }

688 static void
689 multi_tcp_process_io(struct multi_context *m)
690 {
691  struct multi_tcp *mtcp = m->mtcp;
692  int i;
693 
694  for (i = 0; i < mtcp->n_esr; ++i)
695  {
696  struct event_set_return *e = &mtcp->esr[i];
697 
698  /* incoming data for instance? */
699  if (e->arg >= MTCP_N)
700  {
701  struct multi_instance *mi = (struct multi_instance *) e->arg;
702  if (mi)
703  {
704  if (e->rwflags & EVENT_WRITE)
705  {
706  multi_tcp_action(m, mi, TA_SOCKET_WRITE_READY, false);
707  }
708  else if (e->rwflags & EVENT_READ)
709  {
710  multi_tcp_action(m, mi, TA_SOCKET_READ, false);
711  }
712  }
713  }
714  else
715  {
716 #ifdef ENABLE_MANAGEMENT
717  if (e->arg == MTCP_MANAGEMENT)
718  {
719  ASSERT(management);
720  management_io(management);
721  }
722  else
723 #endif
724  /* incoming data on TUN? */
725  if (e->arg == MTCP_TUN)
726  {
727  if (e->rwflags & EVENT_WRITE)
728  {
729  multi_tcp_action(m, NULL, TA_TUN_WRITE, false);
730  }
731  else if (e->rwflags & EVENT_READ)
732  {
733  multi_tcp_action(m, NULL, TA_TUN_READ, false);
734  }
735  }
736  /* new incoming TCP client attempting to connect? */
737  else if (e->arg == MTCP_SOCKET)
738  {
739  struct multi_instance *mi;
740  ASSERT(m->top.c2.link_socket);
741  socket_reset_listen_persistent(m->top.c2.link_socket);
742  mi = multi_create_instance_tcp(m);
743  if (mi)
744  {
745  multi_tcp_action(m, mi, TA_INITIAL, false);
746  }

787  int status;
788 
789  top->mode = CM_TOP;
790  context_clear_2(top);
791 
792  /* initialize top-tunnel instance */
793  init_instance_handle_signals(top, top->es, CC_HARD_USR1_TO_HUP);
794  if (IS_SIG(top))
795  {
796  return;
797  }
798 
799  /* initialize global multi_context object */
800  multi_init(&multi, top, true, MC_SINGLE_THREADED);
801 
802  /* initialize our cloned top object */
803  multi_top_init(&multi, top);
804 
805  /* initialize management interface */
806  init_management_callback_multi(&multi);
807 
808  /* finished with initialization */
809  initialization_sequence_completed(top, ISC_SERVER); /* --mode server --proto tcp-server */
810 
811 #ifdef ENABLE_ASYNC_PUSH
812  multi.top.c2.inotify_fd = inotify_init();
813  if (multi.top.c2.inotify_fd < 0)
814  {
815  msg(D_MULTI_ERRORS | M_ERRNO, "MULTI: inotify_init error");
816  }
817 #endif
818

820  while (true)
821  {
822  perf_push(PERF_EVENT_LOOP);
823 
824  /* wait on tun/socket list */
825  multi_get_timeout(&multi, &multi.top.c2.timeval);
826  status = multi_tcp_wait(&multi.top, multi.mtcp);
827  MULTI_CHECK_SIG(&multi);
828 
829  /* check on status of coarse timers */
830  multi_process_per_second_timers(&multi);
831 
832  /* timeout? */
833  if (status > 0)
834  {
835  /* process the I/O which triggered select */
836  multi_tcp_process_io(&multi);
837  MULTI_CHECK_SIG(&multi);
838  }
839  else if (status == 0)
840  {
841  multi_tcp_action(&multi, NULL, TA_TIMEOUT, false);
842  }
843 
844  perf_pop();
845  }
846 
847 #ifdef ENABLE_ASYNC_PUSH
848  close(top->c2.inotify_fd);
849 #endif
850

852  uninit_management_callback();
853

855  multi_ifconfig_pool_persist(&multi, true);
856

858  multi_uninit(&multi);
859  multi_top_free(&multi);
