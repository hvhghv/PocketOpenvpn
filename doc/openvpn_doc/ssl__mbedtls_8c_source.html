574 static inline int
575 external_pkcs1_sign( void *ctx_voidptr,
576  int (*f_rng)(void *, unsigned char *, size_t), void *p_rng, int mode,
577  mbedtls_md_type_t md_alg, unsigned int hashlen, const unsigned char *hash,
578  unsigned char *sig )
579 {
580  struct external_context *const ctx = ctx_voidptr;
581  int rv;
582  uint8_t *to_sign = NULL;
583  size_t asn_len = 0, oid_size = 0;
584  const char *oid = NULL;
585 
586  if (NULL == ctx)
587  {
588  return MBEDTLS_ERR_RSA_BAD_INPUT_DATA;
589  }
590 
591  if (MBEDTLS_RSA_PRIVATE != mode)
592  {
593  return MBEDTLS_ERR_RSA_BAD_INPUT_DATA;
594  }
595 
596  /*
597  * Support a wide range of hashes. TLSv1.1 and before only need SIG_RSA_RAW,
598  * but TLSv1.2 needs the full suite of hashes.
599  *
600  * This code has been taken from mbed TLS pkcs11_sign(), under the GPLv2.0+.
601  */
602  if (md_alg != MBEDTLS_MD_NONE)
603  {
604  const mbedtls_md_info_t *md_info = mbedtls_md_info_from_type( md_alg );
605  if (md_info == NULL)
606  {
607  return( MBEDTLS_ERR_RSA_BAD_INPUT_DATA );
608  }
609 
610  if (!mbed_ok(mbedtls_oid_get_oid_by_md( md_alg, &oid, &oid_size )))
611  {
612  return( MBEDTLS_ERR_RSA_BAD_INPUT_DATA );
613  }
614 
615  hashlen = mbedtls_md_get_size( md_info );
616  asn_len = 10 + oid_size;
617  }
618 
619  if ((SIZE_MAX - hashlen) < asn_len
620  || ctx->signature_length < (asn_len + hashlen))
621  {
622  return MBEDTLS_ERR_RSA_BAD_INPUT_DATA;
623  }
624 
625  ALLOC_ARRAY_CLEAR(to_sign, uint8_t, asn_len + hashlen);
626  uint8_t *p = to_sign;
627  if (md_alg != MBEDTLS_MD_NONE)
628  {
629  /*
630  * DigestInfo ::= SEQUENCE {
631  * digestAlgorithm DigestAlgorithmIdentifier,
632  * digest Digest }
633  *
634  * DigestAlgorithmIdentifier ::= AlgorithmIdentifier
635  *
636  * Digest ::= OCTET STRING
637  */
638  *p++ = MBEDTLS_ASN1_SEQUENCE | MBEDTLS_ASN1_CONSTRUCTED;
639  *p++ = (unsigned char) ( 0x08 + oid_size + hashlen );
640  *p++ = MBEDTLS_ASN1_SEQUENCE | MBEDTLS_ASN1_CONSTRUCTED;
641  *p++ = (unsigned char) ( 0x04 + oid_size );
642  *p++ = MBEDTLS_ASN1_OID;
643  *p++ = oid_size & 0xFF;
644  memcpy( p, oid, oid_size );
645  p += oid_size;
646  *p++ = MBEDTLS_ASN1_NULL;
647  *p++ = 0x00;
648  *p++ = MBEDTLS_ASN1_OCTET_STRING;
649  *p++ = hashlen;
650 
651  /* Double-check ASN length */
652  ASSERT(asn_len == p - to_sign);
653  }
654 
655  /* Copy the hash to be signed */
656  memcpy(p, hash, hashlen);
657 
658  /* Call external signature function */
659  if (!ctx->sign(ctx->sign_ctx, to_sign, asn_len + hashlen, sig,
660  ctx->signature_length))
661  {
662  rv = MBEDTLS_ERR_RSA_PRIVATE_FAILED;
663  goto done;
664  }
665 
666  rv = 0;
667 
668 done:
669  free(to_sign);
670  return rv;
671 }

849 static int
850 endless_buf_read( endless_buffer *in, unsigned char *out, size_t out_len )
851 {
852  size_t read_len = 0;
853 
854  if (in->first_block == NULL)
855  {
856  return MBEDTLS_ERR_SSL_WANT_READ;
857  }
858 
859  while (in->first_block != NULL && read_len < out_len)
860  {
861  int block_len = in->first_block->length - in->data_start;
862  if (block_len <= out_len - read_len)
863  {
864  buffer_entry *cur_entry = in->first_block;
865  memcpy(out + read_len, cur_entry->data + in->data_start,
866  block_len);
867 
868  read_len += block_len;
869 
870  in->first_block = cur_entry->next_block;
871  in->data_start = 0;
872 
873  if (in->first_block == NULL)
874  {
875  in->last_block = NULL;
876  }
877 
878  buf_free_entry(cur_entry);
879  }
880  else
881  {
882  memcpy(out + read_len, in->first_block->data + in->data_start,
883  out_len - read_len);
884  in->data_start += out_len - read_len;
885  read_len = out_len;
886  }
887  }
888 
889  return read_len;
890 }

892 static int
893 endless_buf_write( endless_buffer *out, const unsigned char *in, size_t len )
894 {
895  buffer_entry *new_block = malloc(sizeof(buffer_entry));
896  if (NULL == new_block)
897  {
898  return MBEDTLS_ERR_NET_SEND_FAILED;

901  new_block->data = malloc(len);
902  if (NULL == new_block->data)
903  {
904  free(new_block);
905  return MBEDTLS_ERR_NET_SEND_FAILED;
906  }
907 
908  new_block->length = len;
909  new_block->next_block = NULL;
910 
911  memcpy(new_block->data, in, len);
912 
913  if (NULL == out->first_block)
914  {
915  out->first_block = new_block;
916  }
917 
918  if (NULL != out->last_block)
919  {
920  out->last_block->next_block = new_block;
921  }
922 
923  out->last_block = new_block;
