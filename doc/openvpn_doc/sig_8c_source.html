129 void
130 print_signal(const struct signal_info *si, const char *title, int msglevel)
131 {
132  if (si)
133  {
134  const char *type = (si->signal_text ? si->signal_text : "");
135  const char *t = (title ? title : "process");
136  const char *hs = NULL;
137  switch (si->source)
138  {
139  case SIG_SOURCE_SOFT:
140  hs = "soft";
141  break;
142 
143  case SIG_SOURCE_HARD:
144  hs = "hard";
145  break;
146 
147  case SIG_SOURCE_CONNECTION_FAILED:
148  hs = "connection failed(soft)";
149  break;
150 
151  default:
152  ASSERT(0);
153  }
154 
155  switch (si->signal_received)
156  {
157  case SIGINT:
158  case SIGTERM:
159  msg(msglevel, "%s[%s,%s] received, %s exiting",
160  signal_name(si->signal_received, true), hs, type, t);
161  break;
162 
163  case SIGHUP:
164  case SIGUSR1:
165  msg(msglevel, "%s[%s,%s] received, %s restarting",
166  signal_name(si->signal_received, true), hs, type, t);
167  break;
168 
169  default:
170  msg(msglevel, "Unknown signal %d [%s,%s] received by %s", si->signal_received, hs, type, t);
171  break;
172  }
173  }
174  else
175  {
176  msg(msglevel, "Unknown signal received");
177  }
178 }

410 static bool
411 ignore_restart_signals(struct context *c)
412 {
413  bool ret = false;
414  if ( (c->sig->signal_received == SIGUSR1 || c->sig->signal_received == SIGHUP)
415  && event_timeout_defined(&c->c2.explicit_exit_notification_interval) )
416  {
417  if (c->sig->source == SIG_SOURCE_HARD)
418  {
419  msg(M_INFO, "Ignoring %s received during exit notification",
420  signal_name(c->sig->signal_received, true));
421  signal_reset(c->sig);
422  ret = true;
423  }
424  else
425  {
426  msg(M_INFO, "Converting soft %s received during exit notification to SIGTERM",
427  signal_name(c->sig->signal_received, true));
428  register_signal(c, SIGTERM, "exit-with-notification");
429  ret = false;
430  }
431  }
432  return ret;
433 }

435 bool
436 process_signal(struct context *c)
437 {
438  bool ret = true;
439 
440  if (ignore_restart_signals(c))
441  {
442  ret = false;
443  }
444  else if (c->sig->signal_received == SIGTERM || c->sig->signal_received == SIGINT)
445  {
446  ret = process_sigterm(c);
447  }
448  else if (c->sig->signal_received == SIGUSR2)
449  {
450  process_sigusr2(c);
451  ret = false;
452  }
453  return ret;
454 }
