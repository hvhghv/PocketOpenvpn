236 x_msg_va(const unsigned int flags, const char *format, va_list arglist)
237 {

239 #if SYSLOG_CAPABILITY
240  int level;
241 #endif

243  char *m2;

249  void usage_small(void);
250 
251 #ifndef HAVE_VARARG_MACROS
252  /* the macro has checked this otherwise */
253  if (!msg_test(flags))
254  {
255  return;
256  }
257 #endif
258 
259  bool crt_error = false;
260  e = openvpn_errno_maybe_crt(&crt_error);
261 
262  /*
263  * Apply muting filter.
264  */
265 #ifndef HAVE_VARARG_MACROS
266  /* the macro has checked this otherwise */
267  if (!dont_mute(flags))
268  {
269  return;
270  }
271 #endif
272 
273  gc_init(&gc);
274 
275  m1 = (char *) gc_malloc(ERR_BUF_SIZE, false, &gc);
276  m2 = (char *) gc_malloc(ERR_BUF_SIZE, false, &gc);
277 
278  vsnprintf(m1, ERR_BUF_SIZE, format, arglist);
279  m1[ERR_BUF_SIZE - 1] = 0; /* windows vsnprintf needs this */
280 
281  if ((flags & M_ERRNO) && e)
282  {
283  openvpn_snprintf(m2, ERR_BUF_SIZE, "%s: %s (errno=%d)",
284  m1, openvpn_strerror(e, crt_error, &gc), e);
285  SWAP;
286  }
287 
288  if (flags & M_OPTERR)
289  {
290  openvpn_snprintf(m2, ERR_BUF_SIZE, "Options error: %s", m1);
291  SWAP;
292  }
293 
294 #if SYSLOG_CAPABILITY
295  if (flags & (M_FATAL|M_NONFATAL|M_USAGE_SMALL))
296  {
297  level = LOG_ERR;
298  }
299  else if (flags & M_WARN)
300  {
301  level = LOG_WARNING;
302  }
303  else
304  {
305  level = LOG_NOTICE;
306  }
307 #endif
308 
309  /* set up client prefix */
310  if (flags & M_NOIPREFIX)
311  {
312  prefix = NULL;
313  }
314  else
315  {
316  prefix = msg_get_prefix();
317  }
318  prefix_sep = " ";
319  if (!prefix)
320  {
321  prefix_sep = prefix = "";
322  }
323 
324  /* virtual output capability used to copy output to management subsystem */
325  if (!forked)
326  {
327  const struct virtual_output *vo = msg_get_virtual_output();
328  if (vo)
329  {
330  openvpn_snprintf(m2, ERR_BUF_SIZE, "%s%s%s",
331  prefix,
332  prefix_sep,
333  m1);
334  virtual_output_print(vo, flags, m2);
335  }
336  }
337 
338  if (!(flags & M_MSG_VIRT_OUT))
339  {
340  if (use_syslog && !std_redir && !forked)
341  {
342 #if SYSLOG_CAPABILITY
343  syslog(level, "%s%s%s",
344  prefix,
345  prefix_sep,
346  m1);
347 #endif
348  }
349  else
350  {
351  FILE *fp = msg_fp(flags);
352  const bool show_usec = check_debug_level(DEBUG_LEVEL_USEC_TIME);
353 
354  if (machine_readable_output)
355  {
356  struct timeval tv;
357  gettimeofday(&tv, NULL);
358 
359  fprintf(fp, "%" PRIi64 ".%06ld %x %s%s%s%s",
360  (int64_t)tv.tv_sec,
361  (long)tv.tv_usec,
362  flags,
363  prefix,
364  prefix_sep,
365  m1,
366  "\n");
367 
368  }
369  else if ((flags & M_NOPREFIX) || suppress_timestamps)
370  {
371  fprintf(fp, "%s%s%s%s",
372  prefix,
373  prefix_sep,
374  m1,
375  (flags&M_NOLF) ? "" : "\n");
376  }
377  else
378  {
379  fprintf(fp, "%s %s%s%s%s",
380  time_string(0, 0, show_usec, &gc),
381  prefix,
382  prefix_sep,
383  m1,
384  (flags&M_NOLF) ? "" : "\n");
385  }
386  fflush(fp);
387  ++x_msg_line_num;
388  }
389  }
390 
391  if (flags & M_FATAL)
392  {
393  msg(M_INFO, "Exiting due to fatal error");
394  }
395 
396  if (flags & M_FATAL)
397  {
398  openvpn_exit(OPENVPN_EXIT_STATUS_ERROR); /* exit point */
399 
400  }
401  if (flags & M_USAGE_SMALL)
402  {
403  usage_small();
404  }
405 
406  gc_free(&gc);
407 }

409 /*
410  * Apply muting filter.
411  */
412 bool
413 dont_mute(unsigned int flags)
414 {
415  bool ret = true;
416  if (mute_cutoff > 0 && !(flags & M_NOMUTE))
417  {
418  const int mute_level = DECODE_MUTE_LEVEL(flags);
419  if (mute_level > 0 && mute_level == mute_category)
420  {
421  if (mute_count == mute_cutoff)
422  {
423  msg(M_INFO | M_NOMUTE, "NOTE: --mute triggered...");
424  }
425  if (++mute_count > mute_cutoff)
426  {
427  ret = false;
428  }
429  }
430  else
431  {
432  const int suppressed = mute_count - mute_cutoff;
433  if (suppressed > 0)
434  {
435  msg(M_INFO | M_NOMUTE,
436  "%d variation(s) on previous %d message(s) suppressed by --mute",
437  suppressed,
438  mute_cutoff);
439  }
440  mute_count = 1;
441  mute_category = mute_level;
442  }
443  }
444  return ret;
445 }

522 void
523 redirect_stdout_stderr(const char *file, bool append)
524 {
525 #if defined(_WIN32)
526  if (!std_redir)
527  {
528  struct gc_arena gc = gc_new();
529  HANDLE log_handle;
530  int log_fd;
531 
532  SECURITY_ATTRIBUTES saAttr;
533  saAttr.nLength = sizeof(SECURITY_ATTRIBUTES);
534  saAttr.bInheritHandle = TRUE;
535  saAttr.lpSecurityDescriptor = NULL;
536 
537  log_handle = CreateFileW(wide_string(file, &gc),
538  GENERIC_WRITE,
539  FILE_SHARE_READ,
540  &saAttr,
541  append ? OPEN_ALWAYS : CREATE_ALWAYS,
542  FILE_ATTRIBUTE_NORMAL,
543  NULL);
544 
545  gc_free(&gc);
546 
547  if (log_handle == INVALID_HANDLE_VALUE)
548  {
549  msg(M_WARN|M_ERRNO, "Warning: cannot open --log file: %s", file);
550  return;
551  }
552 
553  /* append to logfile? */
554  if (append)
555  {
556  if (SetFilePointer(log_handle, 0, NULL, FILE_END) == INVALID_SET_FILE_POINTER)
557  {
558  msg(M_ERR, "Error: cannot seek to end of --log file: %s", file);
559  }
560  }
561 
562  /* save original stderr for password prompts */
563  orig_stderr = _dup(_fileno(stderr));
564  if (orig_stderr == -1)
565  {
566  msg(M_WARN | M_ERRNO, "Warning: cannot duplicate stderr, password prompts will appear in log file instead of console.");
567  orig_stderr = _fileno(stderr);
568  }
569 
570  /* direct stdout/stderr to point to log_handle */
571  log_fd = _open_osfhandle((intptr_t)log_handle, _O_TEXT);
572  if (log_fd == -1)
573  {
574  msg(M_ERR, "Error: --log redirect failed due to _open_osfhandle failure");
575  }
576 
577  /* open log_handle as FILE stream */
578  ASSERT(msgfp == NULL);
579  msgfp = _fdopen(log_fd, "wt");
580  if (msgfp == NULL)
581  {
582  msg(M_ERR, "Error: --log redirect failed due to _fdopen");
583  }
584 
585  /* redirect C-library stdout/stderr to log file */
586  if (_dup2(log_fd, 1) == -1 || _dup2(log_fd, 2) == -1)
587  {
588  msg(M_WARN, "Error: --log redirect of stdout/stderr failed");
589  }
590 
591  std_redir = true;
592  }
593 #elif defined(HAVE_DUP2)
594  if (!std_redir)
595  {
596  int out = open(file,
597  O_CREAT | O_WRONLY | (append ? O_APPEND : O_TRUNC),
598  S_IRUSR | S_IWUSR);
599 
600  if (out < 0)
601  {
602  msg(M_WARN|M_ERRNO, "Warning: Error redirecting stdout/stderr to --log file: %s", file);
603  return;
604  }
605 
606  if (dup2(out, 1) == -1)
607  {
608  msg(M_ERR, "--log file redirection error on stdout");
609  }
610  if (dup2(out, 2) == -1)
611  {
612  msg(M_ERR, "--log file redirection error on stderr");
613  }
614 
615  if (out > 2)
616  {
617  close(out);
618  }
619 
620  std_redir = true;
621  }
622 
623 #else /* if defined(_WIN32) */
624  msg(M_WARN, "WARNING: The --log option is not supported on this OS because it lacks the dup2 function");
625 #endif /* if defined(_WIN32) */
626 }

651 /*
652  * Called after most socket or tun/tap operations, via the inline
653  * function check_status().
654  *
655  * Decide if we should print an error message, and see if we can
656  * extract any useful info from the error, such as a Path MTU hint
657  * from the OS.
658  */
659 void
660 x_check_status(int status,
661  const char *description,
662  struct link_socket *sock,
663  struct tuntap *tt)
664 {
665  const char *extended_msg = NULL;
666 
667  msg(x_cs_verbose_level, "%s %s returned %d",
668  sock ? proto2ascii(sock->info.proto, sock->info.af, true) : "",
669  description,
670  status);
671 
672  if (status < 0)
673  {
674  struct gc_arena gc = gc_new();
675 #if EXTENDED_SOCKET_ERROR_CAPABILITY
676  /* get extended socket error message and possible PMTU hint from OS */
677  if (sock)
678  {
679  int mtu;
680  extended_msg = format_extended_socket_error(sock->sd, &mtu, &gc);
681  if (mtu > 0 && sock->mtu != mtu)
682  {
683  sock->mtu = mtu;
684  sock->info.mtu_changed = true;
685  }
686  }
687 #endif /* EXTENDED_SOCKET_ERROR_CAPABILITY */
688 
689 #ifdef _WIN32
690  /* get possible driver error from TAP-Windows driver */
691  if (tuntap_defined(tt))
692  {
693  extended_msg = tap_win_getinfo(tt, &gc);
694  }
695 #endif
696 
697  bool crt_error = false;
698  int my_errno = openvpn_errno_maybe_crt(&crt_error);
699 
700  if (!ignore_sys_error(my_errno, crt_error))
701  {
702  if (extended_msg)
703  {
704  msg(x_cs_info_level, "%s %s [%s]: %s (code=%d)", description,
705  sock ? proto2ascii(sock->info.proto, sock->info.af, true) : "",
706  extended_msg, openvpn_strerror(my_errno, crt_error, &gc), my_errno);
707  }
708  else
709  {
710  msg(x_cs_info_level, "%s %s: %s (code=%d)", description,
711  sock ? proto2ascii(sock->info.proto, sock->info.af, true) : "",
712  openvpn_strerror(my_errno, crt_error, &gc), my_errno);
713  }
714 
715  if (x_cs_err_delay_ms)
716  {
717  platform_sleep_milliseconds(x_cs_err_delay_ms);
718  }
719  }
720  gc_free(&gc);
721  }
722 }
