155 bool
156 get_user_pass_cr(struct user_pass *up,
157  const char *auth_file,
158  const char *prefix,
159  const unsigned int flags,
160  const char *auth_challenge)
161 {
162  struct gc_arena gc = gc_new();
163 
164  if (!up->defined)
165  {
166  bool from_authfile = (auth_file && !streq(auth_file, "stdin"));
167  bool username_from_stdin = false;
168  bool password_from_stdin = false;
169  bool response_from_stdin = true;
170 
171  if (flags & GET_USER_PASS_PREVIOUS_CREDS_FAILED)
172  {
173  msg(M_WARN, "Note: previous '%s' credentials failed", prefix);
174  }
175 
176 #ifdef ENABLE_MANAGEMENT
177  /*
178  * Get username/password from management interface?
179  */
180  if (management
181  && (!from_authfile && (flags & GET_USER_PASS_MANAGEMENT))
182  && management_query_user_pass_enabled(management))
183  {
184  response_from_stdin = false;
185  if (!auth_user_pass_mgmt(up, prefix, flags, auth_challenge))
186  {
187  return false;
188  }
189  }
190  else
191 #endif /* ifdef ENABLE_MANAGEMENT */
192  /*
193  * Get NEED_OK confirmation from the console
194  */
195  if (flags & GET_USER_PASS_NEED_OK)
196  {
197  struct buffer user_prompt = alloc_buf_gc(128, &gc);
198 
199  buf_printf(&user_prompt, "NEED-OK|%s|%s:", prefix, up->username);
200  if (!query_user_SINGLE(BSTR(&user_prompt), BLEN(&user_prompt),
201  up->password, USER_PASS_LEN, false))
202  {
203  msg(M_FATAL, "ERROR: could not read %s ok-confirmation from stdin", prefix);
204  }
205 
206  if (!strlen(up->password))
207  {
208  strcpy(up->password, "ok");
209  }
210  }
211  else if (flags & GET_USER_PASS_INLINE_CREDS)
212  {
213  struct buffer buf;
214  buf_set_read(&buf, (uint8_t *) auth_file, strlen(auth_file) + 1);
215  if (!(flags & GET_USER_PASS_PASSWORD_ONLY))
216  {
217  buf_parse(&buf, '\n', up->username, USER_PASS_LEN);
218  }
219  buf_parse(&buf, '\n', up->password, USER_PASS_LEN);
220  }
221  /*
222  * Read from auth file unless this is a dynamic challenge request.
223  */
224  else if (from_authfile && !(flags & GET_USER_PASS_DYNAMIC_CHALLENGE))
225  {
226  /*
227  * Try to get username/password from a file.
228  */
229  FILE *fp;
230  char password_buf[USER_PASS_LEN] = { '\0' };
231 
232  fp = platform_fopen(auth_file, "r");
233  if (!fp)
234  {
235  msg(M_ERR, "Error opening '%s' auth file: %s", prefix, auth_file);
236  }
237 
238  if ((flags & GET_USER_PASS_PASSWORD_ONLY) == 0)
239  {
240  /* Read username first */
241  if (fgets(up->username, USER_PASS_LEN, fp) == NULL)
242  {
243  msg(M_FATAL, "Error reading username from %s authfile: %s",
244  prefix,
245  auth_file);
246  }
247  }
248  chomp(up->username);
249 
250  if (fgets(password_buf, USER_PASS_LEN, fp) != NULL)
251  {
252  chomp(password_buf);
253  }
254 
255  if (flags & GET_USER_PASS_PASSWORD_ONLY && !password_buf[0])
256  {
257  msg(M_FATAL, "Error reading password from %s authfile: %s", prefix, auth_file);
258  }
259 
260  if (password_buf[0])
261  {
262  strncpy(up->password, password_buf, USER_PASS_LEN);
263  }
264  /* The auth-file does not have the password: get both username
265  * and password from the management interface if possible.
266  * Otherwise set to read password from console.
267  */
268 #if defined(ENABLE_MANAGEMENT)
269  else if (management
270  && (flags & GET_USER_PASS_MANAGEMENT)
271  && management_query_user_pass_enabled(management))
272  {
273  msg(D_LOW, "No password found in %s authfile '%s'. Querying the management interface", prefix, auth_file);
274  if (!auth_user_pass_mgmt(up, prefix, flags, auth_challenge))
275  {
276  fclose(fp);
277  return false;
278  }
279  }
280 #endif
281  else
282  {
283  password_from_stdin = 1;
284  }
285 
286  fclose(fp);
287 
288  if (!(flags & GET_USER_PASS_PASSWORD_ONLY) && strlen(up->username) == 0)
289  {
290  msg(M_FATAL, "ERROR: username from %s authfile '%s' is empty", prefix, auth_file);
291  }
292  }
293  else
294  {
295  username_from_stdin = true;
296  password_from_stdin = true;
297  }
298 
299  /*
300  * Get username/password from standard input?
301  */
302  if (username_from_stdin || password_from_stdin || response_from_stdin)
303  {
304 #ifdef ENABLE_MANAGEMENT
305  if (auth_challenge && (flags & GET_USER_PASS_DYNAMIC_CHALLENGE) && response_from_stdin)
306  {
307  struct auth_challenge_info *ac = get_auth_challenge(auth_challenge, &gc);
308  if (ac)
309  {
310  char *response = (char *) gc_malloc(USER_PASS_LEN, false, &gc);
311  struct buffer packed_resp, challenge;
312 
313  challenge = alloc_buf_gc(14+strlen(ac->challenge_text), &gc);
314  buf_printf(&challenge, "CHALLENGE: %s", ac->challenge_text);
315  buf_set_write(&packed_resp, (uint8_t *)up->password, USER_PASS_LEN);
316 
317  if (!query_user_SINGLE(BSTR(&challenge), BLEN(&challenge),
318  response, USER_PASS_LEN, BOOL_CAST(ac->flags&CR_ECHO)))
319  {
320  msg(M_FATAL, "ERROR: could not read challenge response from stdin");
321  }
322  strncpynt(up->username, ac->user, USER_PASS_LEN);
323  buf_printf(&packed_resp, "CRV1::%s::%s", ac->state_id, response);
324  }
325  else
326  {
327  msg(M_FATAL, "ERROR: received malformed challenge request from server");
328  }
329  }
330  else
331 #endif /* ifdef ENABLE_MANAGEMENT */
332  {
333  struct buffer user_prompt = alloc_buf_gc(128, &gc);
334  struct buffer pass_prompt = alloc_buf_gc(128, &gc);
335 
336  query_user_clear();
337  buf_printf(&user_prompt, "Enter %s Username:", prefix);
338  buf_printf(&pass_prompt, "Enter %s Password:", prefix);
339 
340  if (username_from_stdin && !(flags & GET_USER_PASS_PASSWORD_ONLY))
341  {
342  query_user_add(BSTR(&user_prompt), BLEN(&user_prompt),
343  up->username, USER_PASS_LEN, true);
344  }
345 
346  if (password_from_stdin)
347  {
348  query_user_add(BSTR(&pass_prompt), BLEN(&pass_prompt),
349  up->password, USER_PASS_LEN, false);
350  }
351 
352  if (!query_user_exec() )
353  {
354  msg(M_FATAL, "ERROR: Failed retrieving username or password");
355  }
356 
357  if (!(flags & GET_USER_PASS_PASSWORD_ONLY))
358  {
359  if (strlen(up->username) == 0)
360  {
361  msg(M_FATAL, "ERROR: %s username is empty", prefix);
362  }
363  }
364 
365 #ifdef ENABLE_MANAGEMENT
366  if (auth_challenge && (flags & GET_USER_PASS_STATIC_CHALLENGE) && response_from_stdin)
367  {
368  char *response = (char *) gc_malloc(USER_PASS_LEN, false, &gc);
369  struct buffer packed_resp, challenge;
370  char *pw64 = NULL, *resp64 = NULL;
371 
372  challenge = alloc_buf_gc(14+strlen(auth_challenge), &gc);
373  buf_printf(&challenge, "CHALLENGE: %s", auth_challenge);
374 
375  if (!query_user_SINGLE(BSTR(&challenge), BLEN(&challenge),
376  response, USER_PASS_LEN,
377  BOOL_CAST(flags & GET_USER_PASS_STATIC_CHALLENGE_ECHO)))
378  {
379  msg(M_FATAL, "ERROR: could not retrieve static challenge response");
380  }
381  if (openvpn_base64_encode(up->password, strlen(up->password), &pw64) == -1
382  || openvpn_base64_encode(response, strlen(response), &resp64) == -1)
383  {
384  msg(M_FATAL, "ERROR: could not base64-encode password/static_response");
385  }
386  buf_set_write(&packed_resp, (uint8_t *)up->password, USER_PASS_LEN);
387  buf_printf(&packed_resp, "SCRV1:%s:%s", pw64, resp64);
388  string_clear(pw64);
389  free(pw64);
390  string_clear(resp64);
391  free(resp64);
392  }
393 #endif /* ifdef ENABLE_MANAGEMENT */
394  }
395  }
396 
397  string_mod(up->username, CC_PRINT, CC_CRLF, 0);
398  string_mod(up->password, CC_PRINT, CC_CRLF, 0);
399 
400  up->defined = true;
401  }
402 
403 #if 0
404  msg(M_INFO, "GET_USER_PASS %s u='%s' p='%s'", prefix, up->username, up->password);
405 #endif
406 
407  gc_free(&gc);
408 
409  return true;
410 }

414 /*
415  * See management/management-notes.txt for more info on the
416  * the dynamic challenge/response protocol implemented here.
417  */
418 struct auth_challenge_info *
419 get_auth_challenge(const char *auth_challenge, struct gc_arena *gc)
420 {
421  if (auth_challenge)
422  {
423  struct auth_challenge_info *ac;
424  const int len = strlen(auth_challenge);
425  char *work = (char *) gc_malloc(len+1, false, gc);
426  char *cp;
427 
428  struct buffer b;
429  buf_set_read(&b, (const uint8_t *)auth_challenge, len);
430 
431  ALLOC_OBJ_CLEAR_GC(ac, struct auth_challenge_info, gc);
432 
433  /* parse prefix */
434  if (!buf_parse(&b, ':', work, len))
435  {
436  return NULL;
437  }
438  if (strcmp(work, "CRV1"))
439  {
440  return NULL;
441  }
442 
443  /* parse flags */
444  if (!buf_parse(&b, ':', work, len))
445  {
446  return NULL;
447  }
448  for (cp = work; *cp != '\0'; ++cp)
449  {
450  const char c = *cp;
451  if (c == 'E')
452  {
453  ac->flags |= CR_ECHO;
454  }
455  else if (c == 'R')
456  {
457  ac->flags |= CR_RESPONSE;
458  }
459  }
460 
461  /* parse state ID */
462  if (!buf_parse(&b, ':', work, len))
463  {
464  return NULL;
465  }
466  ac->state_id = string_alloc(work, gc);
467 
468  /* parse user name */
469  if (!buf_parse(&b, ':', work, len))
470  {
471  return NULL;
472  }
473  ac->user = (char *) gc_malloc(strlen(work)+1, true, gc);
474  openvpn_base64_decode(work, (void *)ac->user, -1);
475 
476  /* parse challenge text */
477  ac->challenge_text = string_alloc(BSTR(&b), gc);
478 
479  return ac;
480  }
481  else
482  {
483  return NULL;
484  }
485 }

590 static const char **
591 make_inline_array(const char *str, struct gc_arena *gc)
592 {
593  char line[OPTION_LINE_SIZE];
594  struct buffer buf;
595  int len = 0;
596  char **ret = NULL;
597  int i = 0;
598 
599  buf_set_read(&buf, (const uint8_t *) str, strlen(str));
600  while (buf_parse(&buf, '\n', line, sizeof(line)))
601  {
602  ++len;
603  }
604 
605  /* alloc return array */
606  ALLOC_ARRAY_CLEAR_GC(ret, char *, len + 1, gc);
607 
608  buf_set_read(&buf, (const uint8_t *) str, strlen(str));
609  while (buf_parse(&buf, '\n', line, sizeof(line)))
610  {
611  chomp(line);
612  ASSERT(i < len);
613  ret[i] = string_alloc(skip_leading_whitespace(line), gc);
614  ++i;
615  }
616  ASSERT(i <= len);
617  ret[i] = NULL;
618  return (const char **)ret;
619 }

666 /*
667  * Remove security-sensitive strings from control message
668  * so that they will not be output to log file.
669  */
670 const char *
671 sanitize_control_message(const char *src, struct gc_arena *gc)
672 {
673  char *ret = gc_malloc(strlen(src)+1, false, gc);
674  char *dest = ret;
675  bool redact = false;
676  int skip = 0;
677 
678  for (;; )
679  {
680  const char c = *src;
681  if (c == '\0')
682  {
683  break;
684  }
685  if (c == 'S' && !strncmp(src, "SESS_ID_", 8))
686  {
687  skip = 7;
688  redact = true;
689  }
690  else if (c == 'e' && !strncmp(src, "echo ", 5))
691  {
692  skip = 4;
693  redact = true;
694  }
695  else if (!check_debug_level(D_SHOW_KEYS)
696  && (c == 'a' && !strncmp(src, "auth-token ", 11)))
697  {
698  /* Unless --verb is 7 or higher (D_SHOW_KEYS), hide
699  * the auth-token value coming in the src string
700  */
701  skip = 10;
702  redact = true;
703  }
704 
705  if (c == ',') /* end of redacted item? */
706  {
707  skip = 0;
708  redact = false;
709  }
710 
711  if (redact)
712  {
713  if (skip > 0)
714  {
715  --skip;
716  *dest++ = c;
717  }
718  }
719  else
720  {
721  *dest++ = c;
722  }
723 
724  ++src;
725  }
726  *dest = '\0';
727  return ret;

730 /* helper to parse peer_info received from multi client, validate
731  * (this is untrusted data) and put into environment
732  */
733 bool
734 validate_peer_info_line(char *line)
735 {
736  uint8_t c;
737  int state = 0;
738  while (*line)
739  {
740  c = *line;
741  switch (state)
742  {
743  case 0:
744  case 1:
745  if (c == '=' && state == 1)
746  {
747  state = 2;
748  }
749  else if (isalnum(c) || c == '_')
750  {
751  state = 1;
752  }
753  else
754  {
755  return false;
756  }
757 
758  case 2:
759  /* after the '=', replace non-printable or shell meta with '_' */
760  if (!isprint(c) || isspace(c)
761  || c == '$' || c == '(' || c == '`')
762  {
763  *line = '_';
764  }
765  }
766  line++;
767  }
768  return (state == 2);
769 }
