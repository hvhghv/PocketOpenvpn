108 static int
109 sitnl_addattr(struct nlmsghdr *n, int maxlen, int type, const void *data,
110  int alen)
111 {
112  int len = RTA_LENGTH(alen);
113  struct rtattr *rta;
114 
115  if ((int)(NLMSG_ALIGN(n->nlmsg_len) + RTA_ALIGN(len)) > maxlen)
116  {
117  msg(M_WARN, "%s: rtnl: message exceeded bound of %d", __func__,
118  maxlen);
119  return -EMSGSIZE;
120  }
121 
122  rta = NLMSG_TAIL(n);
123  rta->rta_type = type;
124  rta->rta_len = len;
125 
126  if (!data)
127  {
128  memset(RTA_DATA(rta), 0, alen);
129  }
130  else
131  {
132  memcpy(RTA_DATA(rta), data, alen);
133  }
134 
135  n->nlmsg_len = NLMSG_ALIGN(n->nlmsg_len) + RTA_ALIGN(len);
136 
137  return 0;
138 }

143 static int
144 sitnl_socket(void)
145 {
146  int sndbuf = SNDBUF_SIZE;
147  int rcvbuf = RCVBUF_SIZE;
148  int fd;
149 
150  fd = socket(AF_NETLINK, SOCK_RAW, NETLINK_ROUTE);
151  if (fd < 0)
152  {
153  msg(M_WARN, "%s: cannot open netlink socket", __func__);
154  return fd;
155  }
156 
157  if (setsockopt(fd, SOL_SOCKET, SO_SNDBUF, &sndbuf, sizeof(sndbuf)) < 0)
158  {
159  msg(M_WARN | M_ERRNO, "%s: SO_SNDBUF", __func__);
160  close(fd);
161  return -1;
162  }
163 
164  if (setsockopt(fd, SOL_SOCKET, SO_RCVBUF, &rcvbuf, sizeof(rcvbuf)) < 0)
165  {
166  msg(M_WARN | M_ERRNO, "%s: SO_RCVBUF", __func__);
167  close(fd);
168  return -1;
169  }
170 
171  return fd;
172 }

177 static int
178 sitnl_bind(int fd, uint32_t groups)
179 {
180  socklen_t addr_len;
181  struct sockaddr_nl local;
182 
183  CLEAR(local);
184 
185  local.nl_family = AF_NETLINK;
186  local.nl_groups = groups;
187 
188  if (bind(fd, (struct sockaddr *)&local, sizeof(local)) < 0)
189  {
190  msg(M_WARN | M_ERRNO, "%s: cannot bind netlink socket", __func__);
191  return -errno;
192  }
193 
194  addr_len = sizeof(local);
195  if (getsockname(fd, (struct sockaddr *)&local, &addr_len) < 0)
196  {
197  msg(M_WARN | M_ERRNO, "%s: cannot getsockname", __func__);
198  return -errno;
199  }
200 
201  if (addr_len != sizeof(local))
202  {
203  msg(M_WARN, "%s: wrong address length %d", __func__, addr_len);
204  return -EINVAL;
205  }
206 
207  if (local.nl_family != AF_NETLINK)
208  {
209  msg(M_WARN, "%s: wrong address family %d", __func__, local.nl_family);
210  return -EINVAL;
211  }
212 
213  return 0;
214 }

219 static int
220 sitnl_send(struct nlmsghdr *payload, pid_t peer, unsigned int groups,
221  sitnl_parse_reply_cb cb, void *arg_cb)
222 {
223  int len, rem_len, fd, ret, rcv_len;
224  struct sockaddr_nl nladdr;
225  struct nlmsgerr *err;
226  struct nlmsghdr *h;
227  unsigned int seq;
228  char buf[1024 * 16];
229  struct iovec iov =
230  {
231  .iov_base = payload,
232  .iov_len = payload->nlmsg_len,
233  };
234  struct msghdr nlmsg =
235  {
236  .msg_name = &nladdr,
237  .msg_namelen = sizeof(nladdr),
238  .msg_iov = &iov,
239  .msg_iovlen = 1,
240  };
241 
242  CLEAR(nladdr);
243 
244  nladdr.nl_family = AF_NETLINK;
245  nladdr.nl_pid = peer;
246  nladdr.nl_groups = groups;
247 
248  payload->nlmsg_seq = seq = time(NULL);
249 
250  /* no need to send reply */
251  if (!cb)
252  {
253  payload->nlmsg_flags |= NLM_F_ACK;
254  }
255 
256  fd = sitnl_socket();
257  if (fd < 0)
258  {
259  msg(M_WARN | M_ERRNO, "%s: can't open rtnl socket", __func__);
260  return -errno;
261  }
262 
263  ret = sitnl_bind(fd, 0);
264  if (ret < 0)
265  {
266  msg(M_WARN | M_ERRNO, "%s: can't bind rtnl socket", __func__);
267  ret = -errno;
268  goto out;
269  }
270 
271  ret = sendmsg(fd, &nlmsg, 0);
272  if (ret < 0)
273  {
274  msg(M_WARN | M_ERRNO, "%s: rtnl: error on sendmsg()", __func__);
275  ret = -errno;
276  goto out;
277  }
278 
279  /* prepare buffer to store RTNL replies */
280  memset(buf, 0, sizeof(buf));
281  iov.iov_base = buf;
282 
283  while (1)
284  {
285  /*
286  * iov_len is modified by recvmsg(), therefore has to be initialized before
287  * using it again
288  */
289  msg(D_RTNL, "%s: checking for received messages", __func__);
290  iov.iov_len = sizeof(buf);
291  rcv_len = recvmsg(fd, &nlmsg, 0);
292  msg(D_RTNL, "%s: rtnl: received %d bytes", __func__, rcv_len);
293  if (rcv_len < 0)
294  {
295  if ((errno == EINTR) || (errno == EAGAIN))
296  {
297  msg(D_RTNL, "%s: interrupted call", __func__);
298  continue;
299  }
300  msg(M_WARN | M_ERRNO, "%s: rtnl: error on recvmsg()", __func__);
301  ret = -errno;
302  goto out;
303  }
304 
305  if (rcv_len == 0)
306  {
307  msg(M_WARN, "%s: rtnl: socket reached unexpected EOF", __func__);
308  ret = -EIO;
309  goto out;
310  }
311 
312  if (nlmsg.msg_namelen != sizeof(nladdr))
313  {
314  msg(M_WARN, "%s: sender address length: %u (expected %zu)",
315  __func__, nlmsg.msg_namelen, sizeof(nladdr));
316  ret = -EIO;
317  goto out;
318  }
319 
320  h = (struct nlmsghdr *)buf;
321  while (rcv_len >= (int)sizeof(*h))
322  {
323  len = h->nlmsg_len;
324  rem_len = len - sizeof(*h);
325 
326  if ((rem_len < 0) || (len > rcv_len))
327  {
328  if (nlmsg.msg_flags & MSG_TRUNC)
329  {
330  msg(M_WARN, "%s: truncated message", __func__);
331  ret = -EIO;
332  goto out;
333  }
334  msg(M_WARN, "%s: malformed message: len=%d", __func__, len);
335  ret = -EIO;
336  goto out;
337  }
338 
339 /* if (((int)nladdr.nl_pid != peer) || (h->nlmsg_pid != nladdr.nl_pid)
340  * || (h->nlmsg_seq != seq))
341  * {
342  * rcv_len -= NLMSG_ALIGN(len);
343  * h = (struct nlmsghdr *)((char *)h + NLMSG_ALIGN(len));
344  * msg(M_DEBUG, "%s: skipping unrelated message. nl_pid:%d (peer:%d) nl_msg_pid:%d nl_seq:%d seq:%d",
345  * __func__, (int)nladdr.nl_pid, peer, h->nlmsg_pid,
346  * h->nlmsg_seq, seq);
347  * continue;
348  * }
349  */
350 
351  if (h->nlmsg_type == NLMSG_DONE)
352  {
353  ret = 0;
354  goto out;
355  }
356 
357  if (h->nlmsg_type == NLMSG_ERROR)
358  {
359  err = (struct nlmsgerr *)NLMSG_DATA(h);
360  if (rem_len < (int)sizeof(struct nlmsgerr))
361  {
362  msg(M_WARN, "%s: ERROR truncated", __func__);
363  ret = -EIO;
364  }
365  else
366  {
367  if (!err->error)
368  {
369  ret = 0;
370  if (cb)
371  {
372  int r = cb(h, arg_cb);
373  if (r <= 0)
374  {
375  ret = r;
376  }
377  }
378  }
379  else
380  {
381  msg(M_WARN, "%s: rtnl: generic error (%d): %s",
382  __func__, err->error, strerror(-err->error));
383  ret = err->error;
384  }
385  }
386  goto out;
387  }
388 
389  if (cb)
390  {
391  int r = cb(h, arg_cb);
392  if (r <= 0)
393  {
394  ret = r;
395  goto out;
396  }
397  }
398  else
399  {
400  msg(M_WARN, "%s: RTNL: unexpected reply", __func__);
401  }
402 
403  rcv_len -= NLMSG_ALIGN(len);
404  h = (struct nlmsghdr *)((char *)h + NLMSG_ALIGN(len));
405  }
406 
407  if (nlmsg.msg_flags & MSG_TRUNC)
408  {
409  msg(M_WARN, "%s: message truncated", __func__);
410  continue;
411  }
412 
413  if (rcv_len)
414  {
415  msg(M_WARN, "%s: rtnl: %d not parsed bytes", __func__, rcv_len);
416  ret = -1;
417  goto out;
418  }
419  }
420 out:
421  close(fd);
422 
423  return ret;
424 }

434 static int
435 sitnl_route_save(struct nlmsghdr *n, void *arg)
436 {
437  route_res_t *res = arg;
438  struct rtmsg *r = NLMSG_DATA(n);
439  struct rtattr *rta = RTM_RTA(r);
440  int len = n->nlmsg_len - NLMSG_LENGTH(sizeof(*r));
441  unsigned int table, ifindex = 0;
442  void *gw = NULL;
443 
444  /* filter-out non-zero dst prefixes */
445  if (res->default_only && r->rtm_dst_len != 0)
446  {
447  return 1;
448  }
449 
450  /* route table, ignored with RTA_TABLE */
451  table = r->rtm_table;
452 
453  while (RTA_OK(rta, len))
454  {
455  switch (rta->rta_type)
456  {
457  /* route interface */
458  case RTA_OIF:
459  ifindex = *(unsigned int *)RTA_DATA(rta);
460  break;
461 
462  /* route prefix */
463  case RTA_DST:
464  break;
465 
466  /* GW for the route */
467  case RTA_GATEWAY:
468  gw = RTA_DATA(rta);
469  break;
470 
471  /* route table */
472  case RTA_TABLE:
473  table = *(unsigned int *)RTA_DATA(rta);
474  break;
475  }
476 
477  rta = RTA_NEXT(rta, len);
478  }
479 
480  /* filter out any route not coming from the selected table */
481  if (res->table && res->table != table)
482  {
483  return 1;
484  }
485 
486  if (!if_indextoname(ifindex, res->iface))
487  {
488  msg(M_WARN | M_ERRNO, "%s: rtnl: can't get ifname for index %d",
489  __func__, ifindex);
490  return -1;
491  }
492 
493  if (gw)
494  {
495  memcpy(&res->gw, gw, res->addr_size);
496  }
497 
498  return 0;
499 }

501 static int
502 sitnl_route_best_gw(sa_family_t af_family, const inet_address_t *dst,
503  void *best_gw, char *best_iface)
504 {
505  struct sitnl_route_req req;
506  route_res_t res;
507  int ret = -EINVAL;
508 
509  ASSERT(best_gw);
510  ASSERT(best_iface);
511 
512  CLEAR(req);
513  CLEAR(res);
514 
515  req.n.nlmsg_len = NLMSG_LENGTH(sizeof(req.r));
516  req.n.nlmsg_type = RTM_GETROUTE;
517  req.n.nlmsg_flags = NLM_F_REQUEST;
518 
519  req.r.rtm_family = af_family;
520 
521  switch (af_family)
522  {
523  case AF_INET:
524  res.addr_size = sizeof(in_addr_t);
525  /*
526  * kernel can't return 0.0.0.0/8 host route, dump all
527  * the routes and filter for 0.0.0.0/0 in cb()
528  */
529  if (!dst || !dst->ipv4)
530  {
531  req.n.nlmsg_flags |= NLM_F_DUMP;
532  res.default_only = true;
533  res.table = RT_TABLE_MAIN;
534  }
535  else
536  {
537  req.r.rtm_dst_len = 32;
538  }
539  break;
540 
541  case AF_INET6:
542  res.addr_size = sizeof(struct in6_addr);
543  /* kernel can return ::/128 host route */
544  req.r.rtm_dst_len = 128;
545  break;
546 
547  default:
548  /* unsupported */
549  return -EINVAL;
550  }
551 
552  SITNL_ADDATTR(&req.n, sizeof(req), RTA_DST, dst, res.addr_size);
553 
554  ret = sitnl_send(&req.n, 0, 0, sitnl_route_save, &res);
555  if (ret < 0)
556  {
557  goto err;
558  }
559 
560  /* save result in output variables */
561  memcpy(best_gw, &res.gw, res.addr_size);
562  strncpy(best_iface, res.iface, IFNAMSIZ);
563 err:
564  return ret;
565 
566 }

654 {

664  }

670  strerror(errno));
671  return -ENOENT;
672  }
673 
674  req.n.nlmsg_len = NLMSG_LENGTH(sizeof(req.i));
675  req.n.nlmsg_flags = NLM_F_REQUEST;
676  req.n.nlmsg_type = RTM_NEWLINK;
677 
678  req.i.ifi_family = AF_PACKET;
679  req.i.ifi_index = ifindex;
680  req.i.ifi_change |= IFF_UP;
681  if (up)
682  {
683  req.i.ifi_flags |= IFF_UP;
684  }
685  else
686  {
687  req.i.ifi_flags &= ~IFF_UP;
688  }
689 
690  msg(M_INFO, "%s: set %s %s", __func__, iface, up ? "up" : "down");
691 
692  return sitnl_send(&req.n, 0, 0, NULL, NULL);
693 }
