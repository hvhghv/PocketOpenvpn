111 /*
112  * Mute levels are designed to avoid large numbers of
113  * mostly similar messages clogging the log file.
114  *
115  * A mute level of 0 is always printed.
116  */
117 #define MUTE_LEVEL_SHIFT 24
118 #define MUTE_LEVEL_MASK 0xFF
119 
120 #define ENCODE_MUTE_LEVEL(mute_level) (((mute_level) & MUTE_LEVEL_MASK) << MUTE_LEVEL_SHIFT)
121 #define DECODE_MUTE_LEVEL(flags) (((flags) >> MUTE_LEVEL_SHIFT) & MUTE_LEVEL_MASK)
122 
123 /*
124  * log_level: verbosity level n (--verb n) must be >= log_level to print.
125  * mute_level: don't print more than n (--mute n) consecutive messages at
126  * a given mute level, or if 0 disable muting and print everything.
127  *
128  * Mask map:
129  * Bits 0-3: log level
130  * Bits 4-23: M_x flags
131  * Bits 24-31: mute level
132  */
133 #define LOGLEV(log_level, mute_level, other) ((log_level) | ENCODE_MUTE_LEVEL(mute_level) | other)
