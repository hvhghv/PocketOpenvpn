224 static struct buffer
225 extract_command_buffer(struct buffer *buf, struct gc_arena *gc)
226 {
227  /* commands on the control channel are seperated by 0x00 bytes.
228  * cmdlen does not include the 0 byte of the string */
229  int cmdlen = (int)strnlen(BSTR(buf), BLEN(buf));
230 
231  if (cmdlen >= BLEN(buf))
232  {
233  buf_advance(buf, cmdlen);
234  /* Return empty buffer */
235  struct buffer empty = { 0 };
236  return empty;
237  }
238 
239  /* include the NUL byte and ensure NUL termination */
240  cmdlen += 1;
241 
242  /* Construct a buffer that only holds the current command and
243  * its closing NUL byte */
244  struct buffer cmdbuf = alloc_buf_gc(cmdlen, gc);
245  buf_write(&cmdbuf, BPTR(buf), cmdlen);
246 
247  /* Remove \r and \n at the end of the buffer to avoid
248  * problems with scripts and other that add extra \r and \n */
249  buf_chomp(&cmdbuf);
250 
251  /* check we have only printable characters or null byte in the
252  * command string and no newlines */
253  if (!string_check_buf(&cmdbuf, CC_PRINT | CC_NULL, CC_CRLF))
254  {
255  msg(D_PUSH_ERRORS, "WARNING: Received control with invalid characters: %s",
256  format_hex(BPTR(&cmdbuf), BLEN(&cmdbuf), 256, gc));
257  cmdbuf.len = 0;
258  }
259 
260  buf_advance(buf, cmdlen);
261  return cmdbuf;
262 }

563 /*
564  * Compress, fragment, encrypt and HMAC-sign an outgoing packet.
565  * Input: c->c2.buf
566  * Output: c->c2.to_link
567  */
568 void
569 encrypt_sign(struct context *c, bool comp_frag)
570 {
571  struct context_buffers *b = c->c2.buffers;
572  const uint8_t *orig_buf = c->c2.buf.data;
573  struct crypto_options *co = NULL;
574 
575  /*
576  * Drop non-TLS outgoing packet if client-connect script/plugin
577  * has not yet succeeded. In non-TLS mode tls_multi is not defined
578  * and we always pass packets.
579  */
580  if (c->c2.tls_multi && c->c2.tls_multi->multi_state != CAS_SUCCEEDED)
581  {
582  c->c2.buf.len = 0;
583  }
584 
585  if (comp_frag)
586  {
587 #ifdef USE_COMP
588  /* Compress the packet. */
589  if (c->c2.comp_context)
590  {
591  (*c->c2.comp_context->alg.compress)(&c->c2.buf, b->compress_buf, c->c2.comp_context, &c->c2.frame);
592  }
593 #endif
594 #ifdef ENABLE_FRAGMENT
595  if (c->c2.fragment)
596  {
597  fragment_outgoing(c->c2.fragment, &c->c2.buf, &c->c2.frame_fragment);
598  }
599 #endif
600  }
601 
602  /* initialize work buffer with FRAME_HEADROOM bytes of prepend capacity */
603  ASSERT(buf_init(&b->encrypt_buf, FRAME_HEADROOM(&c->c2.frame)));
604 
605  if (c->c2.tls_multi)
606  {
607  /* Get the key we will use to encrypt the packet. */
608  tls_pre_encrypt(c->c2.tls_multi, &c->c2.buf, &co);
609  /* If using P_DATA_V2, prepend the 1-byte opcode and 3-byte peer-id to the
610  * packet before openvpn_encrypt(), so we can authenticate the opcode too.
611  */
612  if (c->c2.buf.len > 0 && c->c2.tls_multi->use_peer_id)
613  {
614  tls_prepend_opcode_v2(c->c2.tls_multi, &b->encrypt_buf);
615  }
616  }
617  else
618  {
619  co = &c->c2.crypto_options;
620  }
621 
622  /* Encrypt and authenticate the packet */
623  openvpn_encrypt(&c->c2.buf, b->encrypt_buf, co);
624 
625  /* Do packet administration */
626  if (c->c2.tls_multi)
627  {
628  if (c->c2.buf.len > 0 && !c->c2.tls_multi->use_peer_id)
629  {
630  tls_prepend_opcode_v1(c->c2.tls_multi, &c->c2.buf);
631  }
632  tls_post_encrypt(c->c2.tls_multi, &c->c2.buf);
633  }
634 
635  /*
636  * Get the address we will be sending the packet to.
637  */
638  link_socket_get_outgoing_addr(&c->c2.buf, get_link_socket_info(c),
639  &c->c2.to_link_addr);
640 
641  /* if null encryption, copy result to read_tun_buf */
642  buffer_turnover(orig_buf, &c->c2.to_link, &c->c2.buf, &b->read_tun_buf);
643 }

1004 #ifdef ENABLE_FRAGMENT
1005  c->c2.frame_fragment = c->c2.frame_fragment_initial;
1006 #endif
1007  }
1008

1010

1012  if (c->options.ping_rec_timeout)
1013  {
1014  event_timeout_reset(&c->c2.ping_rec_interval);
1015  }
1016  }
1017  }
1018  else
1019  {
1020  co = &c->c2.crypto_options;
1021  }
1022 
1023  /*
1024  * Drop non-TLS packet if client-connect script/plugin and cipher selection
1025  * has not yet succeeded. In non-TLS mode tls_multi is not defined
1026  * and we always pass packets.
1027  */
1028  if (c->c2.tls_multi && c->c2.tls_multi->multi_state != CAS_SUCCEEDED)
1029  {
1030  c->c2.buf.len = 0;
1031  }
1032

1035  co, &c->c2.frame, ad_start);

1042  }
1043  }
1044  else
1045  {
1046  buf_reset(&c->c2.to_tun);
1047  }
1048  gc_free(&gc);
1049 
1050  return decrypt_status;
1051 }

1230 static void
1231 drop_if_recursive_routing(struct context *c, struct buffer *buf)
1232 {
1233  bool drop = false;
1234  struct openvpn_sockaddr tun_sa;
1235  int ip_hdr_offset = 0;
1236 
1237  if (c->c2.to_link_addr == NULL) /* no remote addr known */
1238  {
1239  return;
1240  }
1241 
1242  tun_sa = c->c2.to_link_addr->dest;
1243 
1244  int proto_ver = get_tun_ip_ver(TUNNEL_TYPE(c->c1.tuntap), &c->c2.buf, &ip_hdr_offset);
1245 
1246  if (proto_ver == 4)
1247  {
1248  const struct openvpn_iphdr *pip;
1249 
1250  /* make sure we got whole IP header */
1251  if (BLEN(buf) < ((int) sizeof(struct openvpn_iphdr) + ip_hdr_offset))
1252  {
1253  return;
1254  }
1255 
1256  /* skip ipv4 packets for ipv6 tun */
1257  if (tun_sa.addr.sa.sa_family != AF_INET)
1258  {
1259  return;
1260  }
1261 
1262  pip = (struct openvpn_iphdr *) (BPTR(buf) + ip_hdr_offset);
1263 
1264  /* drop packets with same dest addr as gateway */
1265  if (tun_sa.addr.in4.sin_addr.s_addr == pip->daddr)
1266  {
1267  drop = true;
1268  }
1269  }
1270  else if (proto_ver == 6)
1271  {
1272  const struct openvpn_ipv6hdr *pip6;
1273 
1274  /* make sure we got whole IPv6 header */
1275  if (BLEN(buf) < ((int) sizeof(struct openvpn_ipv6hdr) + ip_hdr_offset))
1276  {
1277  return;
1278  }
1279 
1280  /* skip ipv6 packets for ipv4 tun */
1281  if (tun_sa.addr.sa.sa_family != AF_INET6)
1282  {
1283  return;
1284  }
1285 
1286  /* drop packets with same dest addr as gateway */
1287  pip6 = (struct openvpn_ipv6hdr *) (BPTR(buf) + ip_hdr_offset);
1288  if (IN6_ARE_ADDR_EQUAL(&tun_sa.addr.in6.sin6_addr, &pip6->daddr))
1289  {
1290  drop = true;
1291  }
1292  }
1293 
1294  if (drop)
1295  {
1296  struct gc_arena gc = gc_new();
1297 
1298  c->c2.buf.len = 0;
1299 
1300  msg(D_LOW, "Recursive routing detected, drop tun packet to %s",
1301  print_link_socket_actual(c->c2.to_link_addr, &gc));
1302  gc_free(&gc);
1303  }
1304 }

1379 void
1380 ipv6_send_icmp_unreachable(struct context *c, struct buffer *buf, bool client)
1381 {
1382 #define MAX_ICMPV6LEN 1280
1383  struct openvpn_icmp6hdr icmp6out;
1384  CLEAR(icmp6out);
1385 
1386  /*
1387  * Get a buffer to the ip packet, is_ipv6 automatically forwards
1388  * the buffer to the ip packet
1389  */
1390  struct buffer inputipbuf = *buf;
1391 
1392  is_ipv6(TUNNEL_TYPE(c->c1.tuntap), &inputipbuf);
1393 
1394  if (BLEN(&inputipbuf) < (int)sizeof(struct openvpn_ipv6hdr))
1395  {
1396  return;
1397  }
1398 
1399  const struct openvpn_ipv6hdr *pip6 = (struct openvpn_ipv6hdr *)BPTR(&inputipbuf);
1400 
1401  /* Copy version, traffic class, flow label from input packet */
1402  struct openvpn_ipv6hdr pip6out = *pip6;
1403 
1404  pip6out.version_prio = pip6->version_prio;
1405  pip6out.daddr = pip6->saddr;
1406 
1407  /*
1408  * Use the IPv6 remote address if we have one, otherwise use a fake one
1409  * using the remote address is preferred since it makes debugging and
1410  * understanding where the ICMPv6 error originates easier
1411  */
1412  if (c->options.ifconfig_ipv6_remote)
1413  {
1414  inet_pton(AF_INET6, c->options.ifconfig_ipv6_remote, &pip6out.saddr);
1415  }
1416  else
1417  {
1418  inet_pton(AF_INET6, "fe80::7", &pip6out.saddr);
1419  }
1420 
1421  pip6out.nexthdr = OPENVPN_IPPROTO_ICMPV6;
1422 
1423  /*
1424  * The ICMPv6 unreachable code worked best in my (arne) tests with Windows,
1425  * Linux and Android. Windows did not like the administratively prohibited
1426  * return code (no fast fail)
1427  */
1428  icmp6out.icmp6_type = OPENVPN_ICMP6_DESTINATION_UNREACHABLE;
1429  icmp6out.icmp6_code = OPENVPN_ICMP6_DU_NOROUTE;
1430 
1431  int icmpheader_len = sizeof(struct openvpn_ipv6hdr)
1432  + sizeof(struct openvpn_icmp6hdr);
1433  int totalheader_len = icmpheader_len;
1434 
1435  if (TUNNEL_TYPE(c->c1.tuntap) == DEV_TYPE_TAP)
1436  {
1437  totalheader_len += sizeof(struct openvpn_ethhdr);
1438  }
1439 
1440  /*
1441  * Calculate size for payload, defined in the standard that the resulting
1442  * frame should be <= 1280 and have as much as possible of the original
1443  * packet
1444  */
1445  int max_payload_size = min_int(MAX_ICMPV6LEN,
1446  TUN_MTU_SIZE(&c->c2.frame) - icmpheader_len);
1447  int payload_len = min_int(max_payload_size, BLEN(&inputipbuf));
1448 
1449  pip6out.payload_len = htons(sizeof(struct openvpn_icmp6hdr) + payload_len);
1450 
1451  /* Construct the packet as outgoing packet back to the client */
1452  struct buffer *outbuf;
1453  if (client)
1454  {
1455  c->c2.to_tun = c->c2.buffers->aux_buf;
1456  outbuf = &(c->c2.to_tun);
1457  }
1458  else
1459  {
1460  c->c2.to_link = c->c2.buffers->aux_buf;
1461  outbuf = &(c->c2.to_link);
1462  }
1463  ASSERT(buf_init(outbuf, totalheader_len));
1464 
1465  /* Fill the end of the buffer with original packet */
1466  ASSERT(buf_safe(outbuf, payload_len));
1467  ASSERT(buf_copy_n(outbuf, &inputipbuf, payload_len));
1468 
1469  /* ICMP Header, copy into buffer to allow checksum calculation */
1470  ASSERT(buf_write_prepend(outbuf, &icmp6out, sizeof(struct openvpn_icmp6hdr)));
1471 
1472  /* Calculate checksum over the packet and write to header */
1473 
1474  uint16_t new_csum = ip_checksum(AF_INET6, BPTR(outbuf), BLEN(outbuf),
1475  (const uint8_t *)&pip6out.saddr,
1476  (uint8_t *)&pip6out.daddr, OPENVPN_IPPROTO_ICMPV6);
1477  ((struct openvpn_icmp6hdr *) BPTR(outbuf))->icmp6_cksum = htons(new_csum);
1478 
1479 
1480  /* IPv6 Header */
1481  ASSERT(buf_write_prepend(outbuf, &pip6out, sizeof(struct openvpn_ipv6hdr)));
1482 
1483  /*
1484  * Tap mode, we also need to create an Ethernet header.
1485  */
1486  if (TUNNEL_TYPE(c->c1.tuntap) == DEV_TYPE_TAP)
1487  {
1488  if (BLEN(buf) < (int)sizeof(struct openvpn_ethhdr))
1489  {
1490  return;
1491  }
1492 
1493  const struct openvpn_ethhdr *orig_ethhdr = (struct openvpn_ethhdr *) BPTR(buf);
1494 
1495  /* Copy frametype and reverse source/destination for the response */
1496  struct openvpn_ethhdr ethhdr;
1497  memcpy(ethhdr.source, orig_ethhdr->dest, OPENVPN_ETH_ALEN);
1498  memcpy(ethhdr.dest, orig_ethhdr->source, OPENVPN_ETH_ALEN);
1499  ethhdr.proto = htons(OPENVPN_ETH_P_IPV6);
1500  ASSERT(buf_write_prepend(outbuf, &ethhdr, sizeof(struct openvpn_ethhdr)));
1501  }
1502 #undef MAX_ICMPV6LEN
1503 }

1529  }

1532  {
1533  /*

1535  * us to examine the IPv4 header.

1537

1539 #if PASSTOS_CAPABILITY
1540  | PIPV4_PASSTOS
1541 #endif

1543  ))
