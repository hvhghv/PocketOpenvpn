176 generate_auth_token(const struct user_pass *up, struct tls_multi *multi)
177 {
178  struct gc_arena gc = gc_new();
179 
180  int64_t timestamp = htonll((uint64_t)now);
181  int64_t initial_timestamp = timestamp;
182 
183  hmac_ctx_t *ctx = multi->opt.auth_token_key.hmac;
184  ASSERT(hmac_ctx_size(ctx) == 256/8);
185 
186  uint8_t sessid[AUTH_TOKEN_SESSION_ID_LEN];
187 
188  if (multi->auth_token)
189  {
190  /* Just enough space to fit 8 bytes+ 1 extra to decode a non padded
191  * base64 string (multiple of 3 bytes). 9 bytes => 12 bytes base64
192  * bytes
193  */
194  char old_tstamp_decode[9];
195 
196  /*
197  * reuse the same session id and timestamp and null terminate it at
198  * for base64 decode it only decodes the session id part of it
199  */
200  char *old_sessid = multi->auth_token + strlen(SESSION_ID_PREFIX);
201  char *old_tsamp_initial = old_sessid + AUTH_TOKEN_SESSION_ID_LEN*8/6;
202 
203  old_tsamp_initial[12] = '\0';
204  ASSERT(openvpn_base64_decode(old_tsamp_initial, old_tstamp_decode, 9) == 9);
205 
206  /*
207  * Avoid old gcc (4.8.x) complaining about strict aliasing
208  * by using a temporary variable instead of doing it in one
209  * line
210  */
211  uint64_t *tstamp_ptr = (uint64_t *) old_tstamp_decode;
212  initial_timestamp = *tstamp_ptr;
213 
214  old_tsamp_initial[0] = '\0';
215  ASSERT(openvpn_base64_decode(old_sessid, sessid, AUTH_TOKEN_SESSION_ID_LEN)==AUTH_TOKEN_SESSION_ID_LEN);
216 
217 
218  /* free the auth-token, we will replace it with a new one */
219  free(multi->auth_token);
220  }
221  else if (!rand_bytes(sessid, AUTH_TOKEN_SESSION_ID_LEN))
222  {
223  msg( M_FATAL, "Failed to get enough randomness for "
224  "authentication token");
225  }
226 
227  /* Calculate the HMAC */
228  /* We enforce up->username to be \0 terminated in ssl.c.. Allowing username
229  * with \0 in them is asking for troubles in so many ways anyway that we
230  * ignore that corner case here
231  */
232  uint8_t hmac_output[256/8];
233 
234  hmac_ctx_reset(ctx);
235 
236  /*
237  * If the token was only valid for the empty user, also generate
238  * a new token with the empty username since we do not want to loose
239  * the information that the username cannot be trusted
240  */
241  struct key_state *ks = &multi->session[TM_ACTIVE].key[KS_PRIMARY];
242  if (ks->auth_token_state_flags & AUTH_TOKEN_VALID_EMPTYUSER)
243  {
244  hmac_ctx_update(ctx, (const uint8_t *) "", 0);
245  }
246  else
247  {
248  hmac_ctx_update(ctx, (uint8_t *) up->username, (int) strlen(up->username));
249  }
250  hmac_ctx_update(ctx, sessid, AUTH_TOKEN_SESSION_ID_LEN);
251  hmac_ctx_update(ctx, (uint8_t *) &initial_timestamp, sizeof(initial_timestamp));
252  hmac_ctx_update(ctx, (uint8_t *) &timestamp, sizeof(timestamp));
253  hmac_ctx_final(ctx, hmac_output);
254 
255  /* Construct the unencoded session token */
256  struct buffer token = alloc_buf_gc(
257  2*sizeof(uint64_t) + AUTH_TOKEN_SESSION_ID_LEN + 256/8, &gc);
258 
259  ASSERT(buf_write(&token, sessid, sizeof(sessid)));
260  ASSERT(buf_write(&token, &initial_timestamp, sizeof(initial_timestamp)));
261  ASSERT(buf_write(&token, &timestamp, sizeof(timestamp)));
262  ASSERT(buf_write(&token, hmac_output, sizeof(hmac_output)));
263 
264  char *b64output = NULL;
265  openvpn_base64_encode(BPTR(&token), BLEN(&token), &b64output);
266 
267  struct buffer session_token = alloc_buf_gc(
268  strlen(SESSION_ID_PREFIX) + strlen(b64output) + 1, &gc);
269 
270  ASSERT(buf_write(&session_token, SESSION_ID_PREFIX, strlen(SESSION_ID_PREFIX)));
271  ASSERT(buf_write(&session_token, b64output, (int)strlen(b64output)));
272  ASSERT(buf_write_u8(&session_token, 0));
273 
274  free(b64output);
275 
276  multi->auth_token = strdup((char *)BPTR(&session_token));
277 
278  dmsg(D_SHOW_KEYS, "Generated token for client: %s (%s)",
279  multi->auth_token, up->username);
280 
281  gc_free(&gc);
282 }

285 static bool
286 check_hmac_token(hmac_ctx_t *ctx, const uint8_t *b64decoded, const char *username)
287 {
288  ASSERT(hmac_ctx_size(ctx) == 256/8);
289 
290  uint8_t hmac_output[256/8];
291 
292  hmac_ctx_reset(ctx);
293  hmac_ctx_update(ctx, (uint8_t *) username, (int)strlen(username));

301 unsigned int
302 verify_auth_token(struct user_pass *up, struct tls_multi *multi,
303  struct tls_session *session)
304 {
305  /*
306  * Base64 is <= input and input is < USER_PASS_LEN, so using USER_PASS_LEN
307  * is safe here but a bit overkill
308  */
309  uint8_t b64decoded[USER_PASS_LEN];
310  int decoded_len = openvpn_base64_decode(up->password + strlen(SESSION_ID_PREFIX),
311  b64decoded, USER_PASS_LEN);
312 
313  /*
314  * Ensure that the decoded data is the size of the
315  * timestamp + hmac + session id
316  */
317  if (decoded_len != TOKEN_DATA_LEN)
318  {
319  msg(M_WARN, "ERROR: --auth-token wrong size (%d!=%d)",
320  decoded_len, (int) TOKEN_DATA_LEN);
321  return 0;
322  }
323 
324  unsigned int ret = 0;
325 
326  const uint8_t *sessid = b64decoded;
327  const uint8_t *tstamp_initial = sessid + AUTH_TOKEN_SESSION_ID_LEN;
328  const uint8_t *tstamp = tstamp_initial + sizeof(int64_t);
329 
330  uint64_t timestamp = ntohll(*((uint64_t *) (tstamp)));
331  uint64_t timestamp_initial = ntohll(*((uint64_t *) (tstamp_initial)));
332 
333  hmac_ctx_t *ctx = multi->opt.auth_token_key.hmac;
334  if (check_hmac_token(ctx, b64decoded, up->username))
335  {
336  ret |= AUTH_TOKEN_HMAC_OK;
337  }
338  else if (check_hmac_token(ctx, b64decoded, ""))
339  {
340  ret |= AUTH_TOKEN_HMAC_OK;
341  ret |= AUTH_TOKEN_VALID_EMPTYUSER;
342  /* overwrite the username of the client with the empty one */
343  strcpy(up->username, "");
344  }
345  else
346  {
347  msg(M_WARN, "--auth-token-gen: HMAC on token from client failed (%s)",
348  up->username);
349  return 0;
350  }
351 
352  /* Accept session tokens only if their timestamp is in the acceptable range
353  * for renegotiations */
354  bool in_renog_time = now >= timestamp
355  && now < timestamp + 2 * session->opt->renegotiate_seconds;
356 
357  /* We could still have a client that does not update
358  * its auth-token, so also allow the initial auth-token */
359  bool initialtoken = multi->auth_token_initial
360  && memcmp_constant_time(up->password, multi->auth_token_initial,
361  strlen(multi->auth_token_initial)) == 0;
362 
363  if (!in_renog_time && !initialtoken)
364  {
365  msg(M_WARN, "Timestamp (%" PRIu64 ") of auth-token is out of the renegotiation window",
366  timestamp);
367  ret |= AUTH_TOKEN_EXPIRED;
368  }
369 
370  /* Sanity check the initial timestamp */
371  if (timestamp < timestamp_initial)
372  {
373  msg(M_WARN, "Initial timestamp (%" PRIu64 ") in token from client earlier than "
374  "current timestamp %" PRIu64 ". Broken/unsynchronised clock?",
375  timestamp_initial, timestamp);
376  ret |= AUTH_TOKEN_EXPIRED;
377  }
378 
379  if (multi->opt.auth_token_lifetime
380  && now > timestamp_initial + multi->opt.auth_token_lifetime)
381  {
382  ret |= AUTH_TOKEN_EXPIRED;
383  }
384 
385  if (ret & AUTH_TOKEN_EXPIRED)
386  {
387  /* Tell client that the session token is expired */
388  auth_set_client_reason(multi, "SESSION: token expired");
389  msg(M_INFO, "--auth-token-gen: auth-token from client expired");
390  }
391  return ret;
392 }
