75 struct mroute_addr {

77  uint8_t unused;
78  uint8_t type; /* MR_ADDR/MR_WITH flags */
79  uint8_t netbits; /* number of bits in network part of address,
80  * valid if MR_WITH_NETBITS is set */
81  union {
82  uint8_t raw_addr[MR_MAX_ADDR_LEN]; /* actual address */
83  struct {
84  uint8_t addr[OPENVPN_ETH_ALEN];
85  uint16_t vid;
86  } ether;
87  struct {
88  in_addr_t addr; /* _network order_ IPv4 address */
89  in_port_t port; /* _network order_ TCP/UDP port */
90  } v4;
91  struct {
92  struct in6_addr addr;
93  in_port_t port; /* _network order_ TCP/UDP port */
94  } v6;
95  struct {
96  uint8_t prefix[12];
97  in_addr_t addr; /* _network order_ IPv4 address */
98  } v4mappedv6;
99  }

185 /*
186  * Given a raw packet in buf, return the src and dest
187  * addresses of the packet.
188  */
189 static inline unsigned int
190 mroute_extract_addr_from_packet(struct mroute_addr *src,
191  struct mroute_addr *dest,
192  struct mroute_addr *esrc,
193  struct mroute_addr *edest,
194  uint16_t vid,
195  const struct buffer *buf,
196  int tunnel_type)
197 {
198  unsigned int ret = 0;
199  verify_align_4(buf);
200  if (tunnel_type == DEV_TYPE_TUN)
201  {
202  ret = mroute_extract_addr_ip(src, dest, buf);
203  }
204  else if (tunnel_type == DEV_TYPE_TAP)
205  {
206  ret = mroute_extract_addr_ether(src, dest, esrc, edest, vid, buf);
207  }
208  return ret;
209 }

211 static inline bool
212 mroute_addr_equal(const struct mroute_addr *a1, const struct mroute_addr *a2)
213 {
214  if (a1->type != a2->type)
215  {
216  return false;
217  }
218  if (a1->netbits != a2->netbits)
219  {
220  return false;
221  }
222  if (a1->len != a2->len)
223  {
224  return false;
225  }
226  return memcmp(a1->raw_addr, a2->raw_addr, a1->len) == 0;
227 }
