248 int
249 tls_version_max(void)
250 {
251 #if defined(TLS1_3_VERSION)
252  /* If this is defined we can safely assume TLS 1.3 support */
253  return TLS_VER_1_3;
254 #elif OPENSSL_VERSION_NUMBER >= 0x10100000L
255  /*
256  * If TLS_VER_1_3 is not defined, we were compiled against a version that
257  * did not support TLS 1.3.
258  *
259  * However, the library we are *linked* against might be OpenSSL 1.1.1
260  * and therefore supports TLS 1.3. This needs to be checked at runtime
261  * since we can be compiled against 1.1.0 and then the library can be
262  * upgraded to 1.1.1.
263  * We only need to check this for OpenSSL versions that can be
264  * upgraded to 1.1.1 without recompile (>= 1.1.0)
265  */
266  if (OpenSSL_version_num() >= 0x1010100fL)
267  {
268  return TLS_VER_1_3;
269  }
270  else
271  {
272  return TLS_VER_1_2;
273  }
274 #elif defined(TLS1_2_VERSION) || defined(SSL_OP_NO_TLSv1_2)
275  return TLS_VER_1_2;
276 #elif defined(TLS1_1_VERSION) || defined(SSL_OP_NO_TLSv1_1)
277  return TLS_VER_1_1;
278 #else /* if defined(TLS1_3_VERSION) */
279  return TLS_VER_1_0;
280 #endif

399 void
400 convert_tls_list_to_openssl(char *openssl_ciphers, size_t len,const char *ciphers)
401 {
402  /* Parse supplied cipher list and pass on to OpenSSL */
403  size_t begin_of_cipher, end_of_cipher;
404 
405  const char *current_cipher;
406  size_t current_cipher_len;
407 
408  const tls_cipher_name_pair *cipher_pair;
409 
410  size_t openssl_ciphers_len = 0;
411  openssl_ciphers[0] = '\0';
412 
413  /* Translate IANA cipher suite names to OpenSSL names */
414  begin_of_cipher = end_of_cipher = 0;
415  for (; begin_of_cipher < strlen(ciphers); begin_of_cipher = end_of_cipher)
416  {
417  end_of_cipher += strcspn(&ciphers[begin_of_cipher], ":");
418  cipher_pair = tls_get_cipher_name_pair(&ciphers[begin_of_cipher], end_of_cipher - begin_of_cipher);
419 
420  if (NULL == cipher_pair)
421  {
422  /* No translation found, use original */
423  current_cipher = &ciphers[begin_of_cipher];
424  current_cipher_len = end_of_cipher - begin_of_cipher;
425 
426  /* Issue warning on missing translation */
427  /* %.*s format specifier expects length of type int, so guarantee */
428  /* that length is small enough and cast to int. */
429  msg(D_LOW, "No valid translation found for TLS cipher '%.*s'",
430  constrain_int(current_cipher_len, 0, 256), current_cipher);
431  }
432  else
433  {
434  /* Use OpenSSL name */
435  current_cipher = cipher_pair->openssl_name;
436  current_cipher_len = strlen(current_cipher);
437 
438  if (end_of_cipher - begin_of_cipher == current_cipher_len
439  && 0 != memcmp(&ciphers[begin_of_cipher], cipher_pair->iana_name,
440  end_of_cipher - begin_of_cipher))
441  {
442  /* Non-IANA name used, show warning */
443  msg(M_WARN, "Deprecated TLS cipher name '%s', please use IANA name '%s'", cipher_pair->openssl_name, cipher_pair->iana_name);
444  }
445  }
446 
447  /* Make sure new cipher name fits in cipher string */
448  if ((SIZE_MAX - openssl_ciphers_len) < current_cipher_len
449  || (len - 1) < (openssl_ciphers_len + current_cipher_len))
450  {
451  msg(M_FATAL,
452  "Failed to set restricted TLS cipher list, too long (>%d).",
453  (int)(len - 1));
454  }
455 
456  /* Concatenate cipher name to OpenSSL cipher string */
457  memcpy(&openssl_ciphers[openssl_ciphers_len], current_cipher, current_cipher_len);
458  openssl_ciphers_len += current_cipher_len;
459  openssl_ciphers[openssl_ciphers_len] = ':';
460  openssl_ciphers_len++;
461 
462  end_of_cipher++;
463  }
464 
465  if (openssl_ciphers_len > 0)
466  {
467  openssl_ciphers[openssl_ciphers_len-1] = '\0';
468  }
469 }

605  /* This method could be as easy as
606  * SSL_CTX_set1_groups_list(ctx->ctx, groups)

823  BIO_free(bio);
824  }
825  else

827  /* Load the PKCS #12 file */
828  if (!(fp = platform_fopen(pkcs12_file, "rb")))
829  {
830  crypto_msg(M_FATAL, "Error opening file %s", pkcs12_file);
831  }
832  p12 = d2i_PKCS12_fp(fp, NULL);
833  fclose(fp);
834  if (!p12)
835  {
836  crypto_msg(M_FATAL, "Error reading PKCS#12 file %s", pkcs12_file);
837  }

839 
840  /* Parse the PKCS #12 file */
841  if (!PKCS12_parse(p12, "", &pkey, &cert, &ca))
842  {
843  pem_password_callback(password, sizeof(password) - 1, 0, NULL);
844  /* Reparse the PKCS #12 file with password */
845  ca = NULL;

855 #endif

857  return 1;
858  }
859  }

863  if (!SSL_CTX_use_certificate(ctx->ctx, cert))
864  {

884  * They're also used to fill in the chain of intermediate certs as
885  * necessary.

894  crypto_msg(M_FATAL,"Cannot add certificate to certificate chain (X509_STORE_add_cert)");
895  }

897  {
898  crypto_msg(M_FATAL,"Cannot add certificate to client CA list (SSL_CTX_add_client_CA)");
899  }
900  }
901  }
902  }
903  else
904  {

906  * ones in PKCS12, do load PKCS12-provided certs to the client extra
907  * certs chain just in case they include intermediate CAs needed to
908  * prove my identity to the other end. This does not make them trusted.
909  */
910  if (ca && sk_X509_num(ca))
911  {
912  for (i = 0; i < sk_X509_num(ca); i++)
913  {
914  if (!SSL_CTX_add_extra_chain_cert(ctx->ctx,sk_X509_value(ca, i)))
915  {
916  crypto_msg(M_FATAL, "Cannot add extra certificate to chain (SSL_CTX_add_extra_chain_cert)");
917  }
918  }
919  }
920  }
921  return 0;
