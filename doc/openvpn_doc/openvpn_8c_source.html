58 static void
59 tunnel_point_to_point(struct context *c)
60 {
61  context_clear_2(c);
62 
63  /* set point-to-point mode */
64  c->mode = CM_P2P;
65 
66  /* initialize tunnel instance */
67  init_instance_handle_signals(c, c->es, CC_HARD_USR1_TO_HUP);
68  if (IS_SIG(c))
69  {
70  return;
71  }
72 
73  /* main event loop */
74  while (true)
75  {
76  perf_push(PERF_EVENT_LOOP);
77 
78  /* process timers, TLS, etc. */
79  pre_select(c);
80  P2P_CHECK_SIG();
81 
82  /* set up and do the I/O wait */
83  io_wait(c, p2p_iow_flags(c));
84  P2P_CHECK_SIG();
85 
86  /* timeout? */
87  if (c->c2.event_set_status == ES_TIMEOUT)
88  {
89  perf_pop();
90  continue;
91  }
92 
93  /* process the I/O which triggered select */
94  process_io(c);
95  P2P_CHECK_SIG();
96 
97  perf_pop();
98  }
99 
100  uninit_management_callback();
101 
102  /* tear down tunnel instance (unless --persist-tun) */
103  close_instance(c);
104 }

108 void init_early(struct context *c)
109 {
110  net_ctx_init(c, &c->net_ctx);
111 
112  /* init verbosity and mute levels */
113  init_verb_mute(c, IVM_LEVEL_1);
114 
115  /* Initialise OpenSSL provider, this needs to be initialised this
116  * early since option post-processing and also openssl info
117  * printing depends on it */
118  for (int j=1; j < MAX_PARMS && c->options.providers.names[j]; j++)
119  {
120  c->options.providers.providers[j] =
121  crypto_load_provider(c->options.providers.names[j]);
122  }
123 }

125 static void uninit_early(struct context *c)
126 {

128  for (int j=1; j < MAX_PARMS && c->options.providers.providers[j]; j++)

131  c->options.providers.providers[j]);
132  }
133  net_ctx_free(&c->net_ctx);
134 }

157 int
158 openvpn_main(int argc, char *argv[])
159 {
160  struct context c;
161 
162 #if PEDANTIC
163  fprintf(stderr, "Sorry, I was built with --enable-pedantic and I am incapable of doing any real work!\n");
164  return 1;
165 #endif
166 
167 #ifdef _WIN32
168  SetConsoleOutputCP(CP_UTF8);
169 #endif
170 
171  CLEAR(c);
172 
173  /* signify first time for components which can
174  * only be initialized once per program instantiation. */
175  c.first_time = true;
176 
177  /* initialize program-wide statics */
178  if (init_static())
179  {
180  /*
181  * This loop is initially executed on startup and then
182  * once per SIGHUP.
183  */
184  do
185  {
186  /* enter pre-initialization mode with regard to signal handling */
187  pre_init_signal_catch();
188 
189  /* zero context struct but leave first_time member alone */
190  context_clear_all_except_first_time(&c);
191 
192  /* static signal info object */
193  CLEAR(siginfo_static);
194  c.sig = &siginfo_static;
195 
196  /* initialize garbage collector scoped to context object */
197  gc_init(&c.gc);
198 
199  /* initialize environmental variable store */
200  c.es = env_set_create(NULL);
201 #ifdef _WIN32
202  set_win_sys_path_via_env(c.es);
203 #endif
204 
205 #ifdef ENABLE_MANAGEMENT
206  /* initialize management subsystem */
207  init_management();
208 #endif
209 
210  /* initialize options to default state */
211  init_options(&c.options, true);
212 
213  /* parse command line options, and read configuration file */
214  parse_argv(&c.options, argc, argv, M_USAGE, OPT_P_DEFAULT, NULL, c.es);
215 
216 #ifdef ENABLE_PLUGIN
217  /* plugins may contribute options configuration */
218  init_verb_mute(&c, IVM_LEVEL_1);
219  init_plugins(&c);
220  open_plugins(&c, true, OPENVPN_PLUGIN_INIT_PRE_CONFIG_PARSE);
221 #endif
222 
223  /* Early initialisation that need to happen before option
224  * post processing and other early startup but after parsing */
225  init_early(&c);
226 
227  /* set dev options */
228  init_options_dev(&c.options);
229 
230  /* openssl print info? */
231  if (print_openssl_info(&c.options))
232  {
233  break;
234  }
235 
236  /* --genkey mode? */
237  if (do_genkey(&c.options))
238  {
239  break;
240  }
241 
242  /* tun/tap persist command? */
243  if (do_persist_tuntap(&c.options, &c.net_ctx))
244  {
245  break;
246  }
247 
248  /* sanity check on options */
249  options_postprocess(&c.options);
250 
251  /* show all option settings */
252  show_settings(&c.options);
253 
254  /* print version number */
255  msg(M_INFO, "%s", title_string);
256 #ifdef _WIN32
257  show_windows_version(M_INFO);
258 #endif
259  show_library_versions(M_INFO);
260 
261  /* misc stuff */
262  pre_setup(&c.options);
263 
264  /* test crypto? */
265  if (do_test_crypto(&c.options))
266  {
267  break;
268  }
269 
270  /* Query passwords before becoming a daemon if we don't use the
271  * management interface to get them. */
272 #ifdef ENABLE_MANAGEMENT
273  if (!(c.options.management_flags & MF_QUERY_PASSWORDS))
274 #endif
275  init_query_passwords(&c);
276 
277  /* become a daemon if --daemon */
278  if (c.first_time)
279  {
280  c.did_we_daemonize = possibly_become_daemon(&c.options);
281  write_pid_file(c.options.writepid, c.options.chroot_dir);
282  }
283 
284 #ifdef ENABLE_MANAGEMENT
285  /* open management subsystem */
286  if (!open_management(&c))
287  {
288  break;
289  }
290  /* query for passwords through management interface, if needed */
291  if (c.options.management_flags & MF_QUERY_PASSWORDS)
292  {
293  init_query_passwords(&c);
294  }
295 #endif
296 
297  /* set certain options as environmental variables */
298  setenv_settings(c.es, &c.options);
299 
300  /* finish context init */
301  context_init_1(&c);
302 
303  do
304  {
305  /* run tunnel depending on mode */
306  switch (c.options.mode)
307  {
308  case MODE_POINT_TO_POINT:
309  tunnel_point_to_point(&c);
310  break;
311 
312  case MODE_SERVER:
313  tunnel_server(&c);
314  break;
315 
316  default:
317  ASSERT(0);
318  }
319 
320  /* indicates first iteration -- has program-wide scope */
321  c.first_time = false;
322 
323  /* any signals received? */
324  if (IS_SIG(&c))
325  {
326  print_signal(c.sig, NULL, M_INFO);
327  }
328 
329  /* pass restart status to management subsystem */
330  signal_restart_status(c.sig);
331  }
332  while (c.sig->signal_received == SIGUSR1);
333 
334  env_set_destroy(c.es);
335  uninit_options(&c.options);
336  gc_reset(&c.gc);
337  uninit_early(&c);
338  }
339  while (c.sig->signal_received == SIGHUP);
340  }
341 
342  context_gc_free(&c);
343 
344 #ifdef ENABLE_MANAGEMENT
345  /* close management interface */
346  close_management();
347 #endif
348 
349  /* uninitialize program-wide statics */
350  uninit_static();
351 
352  openvpn_exit(OPENVPN_EXIT_STATUS_GOOD); /* exit point */
353  return 0; /* NOTREACHED */
354 }
