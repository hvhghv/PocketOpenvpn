58 static bool
59 get_console_input_win32(const char *prompt, const bool echo, char *input, const int capacity)
60 {
61  ASSERT(prompt);
62  ASSERT(input);
63  ASSERT(capacity > 0);
64 
65  input[0] = '\0';
66 
67  HANDLE in = GetStdHandle(STD_INPUT_HANDLE);
68  int orig_stderr = get_orig_stderr(); // guaranteed to be always valid
69  if ((in == INVALID_HANDLE_VALUE)
70  || win32_service_interrupt(&win32_signal)
71  || (_write(orig_stderr, prompt, strlen(prompt)) == -1))
72  {
73  msg(M_WARN|M_ERRNO, "get_console_input_win32(): unexpected error");
74  return false;
75  }
76 
77  bool is_console = (GetFileType(in) == FILE_TYPE_CHAR);
78  DWORD flags_save = 0;
79  int status = 0;
80  WCHAR *winput;
81 
82  if (is_console)
83  {
84  if (GetConsoleMode(in, &flags_save))
85  {
86  DWORD flags = ENABLE_LINE_INPUT | ENABLE_PROCESSED_INPUT;
87  if (echo)
88  {
89  flags |= ENABLE_ECHO_INPUT;
90  }
91  SetConsoleMode(in, flags);
92  }
93  else
94  {
95  is_console = 0;
96  }
97  }
98 
99  DWORD len = 0;
100 
101  if (is_console)
102  {
103  winput = malloc(capacity * sizeof(WCHAR));
104  if (winput == NULL)
105  {
106  return false;
107  }
108 
109  status = ReadConsoleW(in, winput, capacity, &len, NULL);
110  WideCharToMultiByte(CP_UTF8, 0, winput, len, input, capacity, NULL, NULL);
111  free(winput);
112  }
113  else
114  {
115  status = ReadFile(in, input, capacity, &len, NULL);
116  }
117 
118  string_null_terminate(input, (int)len, capacity);
119  chomp(input);
120 
121  if (!echo)
122  {
123  _write(orig_stderr, "\r\n", 2);
124  }
125  if (is_console)
126  {
127  SetConsoleMode(in, flags_save);
128  }
129  if (status && !win32_service_interrupt(&win32_signal))
130  {
131  return true;
132  }
133 
134  return false;
135 }

192 static bool
193 get_console_input(const char *prompt, const bool echo, char *input, const int capacity)
194 {
195  bool ret = false;
196  ASSERT(prompt);
197  ASSERT(input);
198  ASSERT(capacity > 0);
199  input[0] = '\0';
200 
201 #if defined(_WIN32)
202  return get_console_input_win32(prompt, echo, input, capacity);
203 #elif defined(HAVE_GETPASS)
204 
205  /* did we --daemon'ize before asking for passwords?
206  * (in which case neither stdin or stderr are connected to a tty and
207  * /dev/tty can not be open()ed anymore)
208  */
209  if (!isatty(0) && !isatty(2) )
210  {
211  int fd = open( "/dev/tty", O_RDWR );
212  if (fd < 0)
213  {
214  msg(M_FATAL, "neither stdin nor stderr are a tty device and you have neither a "
215  "controlling tty nor systemd - can't ask for '%s'. If you used --daemon, "
216  "you need to use --askpass to make passphrase-protected keys work, and you "
217  "can not use --auth-nocache.", prompt );
218  }
219  close(fd);
220  }
221 
222  if (echo)
223  {
224  FILE *fp;
225 
226  fp = open_tty(true);
227  fprintf(fp, "%s", prompt);
228  fflush(fp);
229  close_tty(fp);
230 
231  fp = open_tty(false);
232  if (fgets(input, capacity, fp) != NULL)
233  {
234  chomp(input);
235  ret = true;
236  }
237  close_tty(fp);
238  }
239  else
240  {
241  char *gp = getpass(prompt);
242  if (gp)
243  {
244  strncpynt(input, gp, capacity);
245  secure_memzero(gp, strlen(gp));
246  ret = true;
247  }
248  }
249 #else /* if defined(_WIN32) */
250  msg(M_FATAL, "Sorry, but I can't get console input on this OS (%s)", prompt);
251 #endif /* if defined(_WIN32) */
252  return ret;
253 }
