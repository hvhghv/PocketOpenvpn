24 /*
25  * This file implements a simple OpenVPN plugin module which
26  * can do either an instant authentication or a deferred auth.
27  * The purpose of this plug-in is to test multiple auth plugins
28  * in the same configuration file
29  *
30  * Plugin arguments:
31  *
32  * multi-auth.so LOG_ID DEFER_TIME USERNAME PASSWORD
33  *
34  * LOG_ID is just an ID string used to separate auth results in the log
35  * DEFER_TIME is the time to defer the auth. Set to 0 to return immediately
36  * USERNAME is the username for a valid authentication
37  * PASSWORD is the password for a valid authentication
38  *
39  * The DEFER_TIME time unit is in ms.
40  *
41  * Sample usage:
42  *
43  * plugin multi-auth.so MA_1 0 foo bar # Instant reply user:foo pass:bar
44  * plugin multi-auth.so MA_2 5000 fux bax # Defer 5 sec, user:fux pass: bax
45  *
46  */

116 /*
117  * Given an environmental variable name, search
118  * the envp array for its value, returning it
119  * if found or NULL otherwise.
120  */
121 static const char *
122 get_env(const char *name, const char *envp[])
123 {
124  if (envp)
125  {
126  int i;
127  const int namelen = strlen(name);
128  for (i = 0; envp[i]; ++i)
129  {
130  if (!strncmp(envp[i], name, namelen))
131  {
132  const char *cp = envp[i] + namelen;
133  if (*cp == '=')
134  {
135  return cp + 1;
136  }
137  }
138  }
139  }
140  return NULL;
141 }

208  context->test_deferred_auth = atoi_null0(args->argv[2]);
209  context->test_valid_user = strdup(args->argv[3]);

215  goto error;
216  }
217

241 static bool
242 do_auth_user_pass(struct plugin_context *context,
243  const char *username, const char *password)
244 {
245  plog(context, PLOG_NOTE,
246  "expect_user=%s, received_user=%s, expect_passw=%s, received_passw=%s",
247  np(context->test_valid_user),
248  np(username),
249  np(context->test_valid_pass),
250  np(password));
251 
252  if (context->test_valid_user && context->test_valid_pass)
253  {
254  if ((strcmp(context->test_valid_user, username) != 0)
255  || (strcmp(context->test_valid_pass, password) != 0))
256  {
257  plog(context, PLOG_ERR,
258  "User/Password auth result: FAIL");
259  return false;
260  }
261  else
262  {
263  plog(context, PLOG_NOTE,
264  "User/Password auth result: PASS");
265  return true;
266  }
267  }
268  return false;
269 }

272 static int
273 auth_user_pass_verify(struct plugin_context *context,
274  struct plugin_per_client_context *pcc,
275  const char *argv[], const char *envp[])
276 {
277  /* get username/password from envp string array */
278  const char *username = get_env("username", envp);
279  const char *password = get_env("password", envp);
280 
281  if (!context->test_deferred_auth)
282  {
283  plog(context, PLOG_NOTE, "Direct authentication");
284  return do_auth_user_pass(context, username, password) ?
285  OPENVPN_PLUGIN_FUNC_SUCCESS : OPENVPN_PLUGIN_FUNC_ERROR;
286  }
287 
288  /* get auth_control_file filename from envp string array*/
289  const char *auth_control_file = get_env("auth_control_file", envp);
290  plog(context, PLOG_NOTE, "auth_control_file=%s", auth_control_file);
291 
292  /* Authenticate asynchronously in n seconds */
293  if (!auth_control_file)
294  {
295  return OPENVPN_PLUGIN_FUNC_ERROR;
296  }
297 
298  /* we do not want to complicate our lives with having to wait()
299  * for child processes (so they are not zombiefied) *and* we MUST NOT
300  * fiddle with signal handlers (= shared with openvpn main), so
301  * we use double-fork() trick.
302  */
303 
304  /* fork, sleep, succeed (no "real" auth done = always succeed) */
305  pid_t p1 = fork();
306  if (p1 < 0) /* Fork failed */
307  {
308  return OPENVPN_PLUGIN_FUNC_ERROR;
309  }
310  if (p1 > 0) /* parent process */
311  {
312  waitpid(p1, NULL, 0);
313  return OPENVPN_PLUGIN_FUNC_DEFERRED;
314  }
315 
316  /* first gen child process, fork() again and exit() right away */
317  pid_t p2 = fork();
318  if (p2 < 0)
319  {
320  plog(context, PLOG_ERR|PLOG_ERRNO, "BACKGROUND: fork(2) failed");
321  exit(1);
322  }
323 
324  if (p2 != 0) /* new parent: exit right away */
325  {
326  exit(0);
327  }
328 
329  /* (grand-)child process
330  * - never call "return" now (would mess up openvpn)
331  * - return status is communicated by file
332  * - then exit()
333  */
334 
335  /* do mighty complicated work that will really take time here... */
336  plog(context, PLOG_NOTE, "in async/deferred handler, usleep(%d)",
337  context->test_deferred_auth*1000);
338  usleep(context->test_deferred_auth*1000);
339 
340  /* now signal success state to openvpn */
341  int fd = open(auth_control_file, O_WRONLY);
342  if (fd < 0)
343  {
344  plog(context, PLOG_ERR|PLOG_ERRNO,
345  "open('%s') failed", auth_control_file);
346  exit(1);
347  }
348 
349  char result[2] = "0\0";
350  if (do_auth_user_pass(context, username, password))
351  {
352  result[0] = '1';
353  }
354 
355  if (write(fd, result, 1) != 1)
356  {
357  plog(context, PLOG_ERR|PLOG_ERRNO, "write to '%s' failed", auth_control_file );
358  }
359  close(fd);
360 
361  exit(0);
362 }
