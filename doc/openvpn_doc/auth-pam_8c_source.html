289 /*
290  * Split and decode up->password in the form SCRV1:base64_pass:base64_response
291  * into pass and response and save in up->password and up->response.
292  * If the password is not in the expected format, input is not changed.
293  */
294 static void
295 split_scrv1_password(struct user_pass *up)
296 {
297  const int skip = strlen("SCRV1:");
298  if (strncmp(up->password, "SCRV1:", skip) != 0)
299  {
300  return;
301  }
302 
303  char *tmp = strdup(up->password);
304  if (!tmp)
305  {
306  plugin_log(PLOG_ERR, MODULE, "out of memory parsing static challenge password");
307  goto out;
308  }
309 
310  char *pass = tmp + skip;
311  char *resp = strchr(pass, ':');
312  if (!resp) /* string not in SCRV1:xx:yy format */
313  {
314  goto out;
315  }
316  *resp++ = '\0';
317 
318  int n = plugin_base64_decode(pass, up->password, sizeof(up->password)-1);
319  if (n >= 0)
320  {
321  up->password[n] = '\0';
322  n = plugin_base64_decode(resp, up->response, sizeof(up->response)-1);
323  if (n >= 0)
324  {
325  up->response[n] = '\0';
326  if (DEBUG(up->verb))
327  {
328  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: parsed static challenge password");
329  }
330  goto out;
331  }
332  }
333 
334  /* decode error: reinstate original value of up->password and return */
335  plugin_secure_memzero(up->password, sizeof(up->password));
336  plugin_secure_memzero(up->response, sizeof(up->response));
337  strcpy(up->password, tmp); /* tmp is guaranteed to fit in up->password */
338 
339  plugin_log(PLOG_ERR, MODULE, "base64 decode error while parsing static challenge password");
340 
341 out:
342  if (tmp)
343  {
344  plugin_secure_memzero(tmp, strlen(tmp));
345  free(tmp);
346  }
347 }

349 OPENVPN_EXPORT int
350 openvpn_plugin_open_v3(const int v3structver,
351  struct openvpn_plugin_args_open_in const *args,
352  struct openvpn_plugin_args_open_return *ret)
353 {
354  pid_t pid;
355  int fd[2];
356 
357  struct auth_pam_context *context;
358  struct name_value_list name_value_list;
359 
360  const int base_parms = 2;
361 
362  const char **argv = args->argv;
363  const char **envp = args->envp;
364 
365  /* Check API compatibility -- struct version 5 or higher needed */
366  if (v3structver < 5)
367  {
368  fprintf(stderr, "AUTH-PAM: This plugin is incompatible with the running version of OpenVPN\n");
369  return OPENVPN_PLUGIN_FUNC_ERROR;
370  }
371 
372  /*
373  * Allocate our context
374  */
375  context = (struct auth_pam_context *) calloc(1, sizeof(struct auth_pam_context));
376  if (!context)
377  {
378  goto error;
379  }
380  context->foreground_fd = -1;
381 
382  /*
383  * Intercept the --auth-user-pass-verify callback.
384  */
385  ret->type_mask = OPENVPN_PLUGIN_MASK(OPENVPN_PLUGIN_AUTH_USER_PASS_VERIFY);
386 
387  /* Save global pointers to functions exported from openvpn */
388  plugin_log = args->callbacks->plugin_log;
389  plugin_secure_memzero = args->callbacks->plugin_secure_memzero;
390  plugin_base64_decode = args->callbacks->plugin_base64_decode;
391 
392  /*
393  * Make sure we have two string arguments: the first is the .so name,
394  * the second is the PAM service type.
395  */
396  if (string_array_len(argv) < base_parms)
397  {
398  plugin_log(PLOG_ERR, MODULE, "need PAM service parameter");
399  goto error;
400  }
401 
402  /*
403  * See if we have optional name/value pairs to match against
404  * PAM module queried fields in the conversation function.
405  */
406  name_value_list.len = 0;
407  if (string_array_len(argv) > base_parms)
408  {
409  const int nv_len = string_array_len(argv) - base_parms;
410  int i;
411 
412  if ((nv_len & 1) == 1 || (nv_len / 2) > N_NAME_VALUE)
413  {
414  plugin_log(PLOG_ERR, MODULE, "bad name/value list length");
415  goto error;
416  }
417 
418  name_value_list.len = nv_len / 2;
419  for (i = 0; i < name_value_list.len; ++i)
420  {
421  const int base = base_parms + i * 2;
422  name_value_list.data[i].name = argv[base];
423  name_value_list.data[i].value = argv[base+1];
424  }
425  }
426 
427  /*
428  * Get verbosity level from environment
429  */
430  {
431  const char *verb_string = get_env("verb", envp);
432  if (verb_string)
433  {
434  context->verb = atoi(verb_string);
435  }
436  }
437 
438  /*
439  * Make a socket for foreground and background processes
440  * to communicate.
441  */
442  if (socketpair(PF_UNIX, SOCK_DGRAM, 0, fd) == -1)
443  {
444  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "socketpair call failed");
445  goto error;
446  }
447 
448  /*
449  * Fork off the privileged process. It will remain privileged
450  * even after the foreground process drops its privileges.
451  */
452  pid = fork();
453 
454  if (pid)
455  {
456  int status;
457 
458  /*
459  * Foreground Process
460  */
461 
462  context->background_pid = pid;
463 
464  /* close our copy of child's socket */
465  close(fd[1]);
466 
467  /* don't let future subprocesses inherit child socket */
468  if (fcntl(fd[0], F_SETFD, FD_CLOEXEC) < 0)
469  {
470  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "Set FD_CLOEXEC flag on socket file descriptor failed");
471  }
472 
473  /* wait for background child process to initialize */
474  status = recv_control(fd[0]);
475  if (status == RESPONSE_INIT_SUCCEEDED)
476  {
477  context->foreground_fd = fd[0];
478  ret->handle = (openvpn_plugin_handle_t *) context;
479  plugin_log( PLOG_NOTE, MODULE, "initialization succeeded (fg)" );
480  return OPENVPN_PLUGIN_FUNC_SUCCESS;
481  }
482  }
483  else
484  {
485  /*
486  * Background Process
487  */
488 
489  /* close all parent fds except our socket back to parent */
490  close_fds_except(fd[1]);
491 
492  /* Ignore most signals (the parent will receive them) */
493  set_signals();
494 
495 #ifdef DO_DAEMONIZE
496  /* Daemonize if --daemon option is set. */
497  daemonize(envp);
498 #endif
499 
500  /* execute the event loop */
501  pam_server(fd[1], argv[1], context->verb, &name_value_list);
502 
503  close(fd[1]);
504 
505  exit(0);
506  return 0; /* NOTREACHED */
507  }
508 
509 error:
510  if (context)
511  {
512  free(context);
513  }
514  return OPENVPN_PLUGIN_FUNC_ERROR;
515 }

517 OPENVPN_EXPORT int
518 openvpn_plugin_func_v1(openvpn_plugin_handle_t handle, const int type, const char *argv[], const char *envp[])
519 {
520  struct auth_pam_context *context = (struct auth_pam_context *) handle;
521 
522  if (type == OPENVPN_PLUGIN_AUTH_USER_PASS_VERIFY && context->foreground_fd >= 0)
523  {
524  /* get username/password from envp string array */
525  const char *username = get_env("username", envp);
526  const char *password = get_env("password", envp);
527  const char *common_name = get_env("common_name", envp) ? get_env("common_name", envp) : "";
528 
529  /* should we do deferred auth?
530  * yes, if there is "auth_control_file" and "deferred_auth_pam" env
531  */
532  const char *auth_control_file = get_env("auth_control_file", envp);
533  const char *deferred_auth_pam = get_env("deferred_auth_pam", envp);
534  if (auth_control_file != NULL && deferred_auth_pam != NULL)
535  {
536  if (DEBUG(context->verb))
537  {
538  plugin_log(PLOG_NOTE, MODULE, "do deferred auth '%s'",
539  auth_control_file);
540  }
541  }
542  else
543  {
544  auth_control_file = "";
545  }
546 
547  if (username && strlen(username) > 0 && password)
548  {
549  if (send_control(context->foreground_fd, COMMAND_VERIFY) == -1
550  || send_string(context->foreground_fd, username) == -1
551  || send_string(context->foreground_fd, password) == -1
552  || send_string(context->foreground_fd, common_name) == -1
553  || send_string(context->foreground_fd, auth_control_file) == -1)
554  {
555  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "Error sending auth info to background process");
556  }
557  else
558  {
559  const int status = recv_control(context->foreground_fd);
560  if (status == RESPONSE_VERIFY_SUCCEEDED)
561  {
562  return OPENVPN_PLUGIN_FUNC_SUCCESS;
563  }
564  if (status == RESPONSE_DEFER)
565  {
566  if (DEBUG(context->verb))
567  {
568  plugin_log(PLOG_NOTE, MODULE, "deferred authentication");
569  }
570  return OPENVPN_PLUGIN_FUNC_DEFERRED;
571  }
572  if (status == -1)
573  {
574  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "Error receiving auth confirmation from background process");
575  }
576  }
577  }
578  }
579  return OPENVPN_PLUGIN_FUNC_ERROR;
580 }

627 /*
628  * PAM conversation function
629  */
630 static int
631 my_conv(int n, const struct pam_message **msg_array,
632  struct pam_response **response_array, void *appdata_ptr)
633 {
634  const struct user_pass *up = ( const struct user_pass *) appdata_ptr;
635  struct pam_response *aresp;
636  int i;
637  int ret = PAM_SUCCESS;
638 
639  *response_array = NULL;
640 
641  if (n <= 0 || n > PAM_MAX_NUM_MSG)
642  {
643  return (PAM_CONV_ERR);
644  }
645  if ((aresp = calloc(n, sizeof *aresp)) == NULL)
646  {
647  return (PAM_BUF_ERR);
648  }
649 
650  /* loop through each PAM-module query */
651  for (i = 0; i < n; ++i)
652  {
653  const struct pam_message *msg = msg_array[i];
654  aresp[i].resp_retcode = 0;
655  aresp[i].resp = NULL;
656 
657  if (DEBUG(up->verb))
658  {
659  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: my_conv[%d] query='%s' style=%d",
660  i,
661  msg->msg ? msg->msg : "NULL",
662  msg->msg_style);
663  }
664 
665  if (up->name_value_list && up->name_value_list->len > 0)
666  {
667  /* use name/value list match method */
668  const struct name_value_list *list = up->name_value_list;
669  int j;
670 
671  /* loop through name/value pairs */
672  for (j = 0; j < list->len; ++j)
673  {
674  const char *match_name = list->data[j].name;
675  const char *match_value = list->data[j].value;
676 
677  if (name_value_match(msg->msg, match_name))
678  {
679  /* found name/value match */
680  aresp[i].resp = NULL;
681 
682  if (DEBUG(up->verb))
683  {
684  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: name match found, query/match-string ['%s', '%s'] = '%s'",
685  msg->msg,
686  match_name,
687  match_value);
688  }
689 
690  if (strstr(match_value, "USERNAME"))
691  {
692  aresp[i].resp = searchandreplace(match_value, "USERNAME", up->username);
693  }
694  else if (strstr(match_value, "PASSWORD"))
695  {
696  aresp[i].resp = searchandreplace(match_value, "PASSWORD", up->password);
697  }
698  else if (strstr(match_value, "COMMONNAME"))
699  {
700  aresp[i].resp = searchandreplace(match_value, "COMMONNAME", up->common_name);
701  }
702  else if (strstr(match_value, "OTP"))
703  {
704  aresp[i].resp = searchandreplace(match_value, "OTP", up->response);
705  }
706  else
707  {
708  aresp[i].resp = strdup(match_value);
709  }
710 
711  if (aresp[i].resp == NULL)
712  {
713  ret = PAM_CONV_ERR;
714  }
715  break;
716  }
717  }
718 
719  if (j == list->len)
720  {
721  ret = PAM_CONV_ERR;
722  }
723  }
724  else
725  {
726  /* use PAM_PROMPT_ECHO_x hints */
727  switch (msg->msg_style)
728  {
729  case PAM_PROMPT_ECHO_OFF:
730  aresp[i].resp = strdup(up->password);
731  if (aresp[i].resp == NULL)
732  {
733  ret = PAM_CONV_ERR;
734  }
735  break;
736 
737  case PAM_PROMPT_ECHO_ON:
738  aresp[i].resp = strdup(up->username);
739  if (aresp[i].resp == NULL)
740  {
741  ret = PAM_CONV_ERR;
742  }
743  break;
744 
745  case PAM_ERROR_MSG:
746  case PAM_TEXT_INFO:
747  break;
748 
749  default:
750  ret = PAM_CONV_ERR;
751  break;
752  }
753  }
754  }
755 
756  if (ret == PAM_SUCCESS)
757  {
758  *response_array = aresp;
759  }
760  else
761  {
762  free(aresp);
763  }
764 
765  return ret;
766 }

768 /*
769  * Return 1 if authenticated and 0 if failed.
770  * Called once for every username/password
771  * to be authenticated.
772  */
773 static int
774 pam_auth(const char *service, const struct user_pass *up)
775 {
776  struct pam_conv conv;

814 /*
815  * deferred auth handler
816  * - fork() (twice, to avoid the need for async wait / SIGCHLD handling)
817  * - query PAM stack via pam_auth()
818  * - send response back to OpenVPN via "ac_file_name"
819  *
820  * parent process returns "0" for "fork() and wait() succeeded",
821  * "-1" for "something went wrong, abort program"
822  */
823 
824 static void
825 do_deferred_pam_auth(int fd, const char *ac_file_name,
826  const char *service, const struct user_pass *up)
827 {
828  if (send_control(fd, RESPONSE_DEFER) == -1)
829  {
830  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "BACKGROUND: write error on response socket [4]");
831  return;
832  }
833 
834  /* double forking so we do not need to wait() for async auth kids */
835  pid_t p1 = fork();
836 
837  if (p1 < 0)
838  {
839  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "BACKGROUND: fork(1) failed");
840  return;
841  }
842  if (p1 != 0) /* parent */
843  {
844  waitpid(p1, NULL, 0);
845  return; /* parent's job succeeded */
846  }
847 
848  /* child */
849  close(fd); /* socketpair no longer needed */
850 
851  pid_t p2 = fork();
852  if (p2 < 0)
853  {
854  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "BACKGROUND: fork(2) failed");
855  exit(1);
856  }
857 
858  if (p2 != 0) /* new parent: exit right away */
859  {
860  exit(0);
861  }
862 
863  /* grandchild */
864  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: deferred auth for '%s', pid=%d",
865  up->username, (int) getpid() );
866 
867  /* the rest is very simple: do PAM, write status byte to file, done */
868  int ac_fd = open( ac_file_name, O_WRONLY );
869  if (ac_fd < 0)
870  {
871  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "cannot open '%s' for writing",
872  ac_file_name );
873  exit(1);
874  }
875  int pam_success = pam_auth(service, up);
876 
877  if (write( ac_fd, pam_success ? "1" : "0", 1 ) != 1)
878  {
879  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "cannot write to '%s'",
880  ac_file_name );
881  }
882  close(ac_fd);
883  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: %s: deferred auth: PAM %s",
884  up->username, pam_success ? "succeeded" : "rejected" );
885  exit(0);
886 }

888 /*
889  * Background process -- runs with privilege.
890  */
891 static void
892 pam_server(int fd, const char *service, int verb, const struct name_value_list *name_value_list)
893 {
894  struct user_pass up;
895  char ac_file_name[PATH_MAX];
896  int command;
897 #ifdef USE_PAM_DLOPEN
898  static const char pam_so[] = "libpam.so";
899 #endif
900 
901  /*
902  * Do initialization
903  */
904  if (DEBUG(verb))
905  {
906  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: INIT service='%s'", service);
907  }
908 
909 #ifdef USE_PAM_DLOPEN
910  /*
911  * Load PAM shared object
912  */
913  if (!dlopen_pam(pam_so))
914  {
915  plugin_log(PLOG_ERR, MODULE, "BACKGROUND: could not load PAM lib %s: %s", pam_so, dlerror());
916  send_control(fd, RESPONSE_INIT_FAILED);
917  goto done;
918  }
919 #endif
920 
921  /*
922  * Tell foreground that we initialized successfully
923  */
924  if (send_control(fd, RESPONSE_INIT_SUCCEEDED) == -1)
925  {
926  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "BACKGROUND: write error on response socket [1]");
927  goto done;
928  }
929 
930  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: initialization succeeded");
931 
932  /*
933  * Event loop
934  */
935  while (1)
936  {
937  memset(&up, 0, sizeof(up));
938  up.verb = verb;
939  up.name_value_list = name_value_list;
940 
941  /* get a command from foreground process */
942  command = recv_control(fd);
943 
944  if (DEBUG(verb))
945  {
946  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: received command code: %d", command);
947  }
948 
949  switch (command)
950  {
951  case COMMAND_VERIFY:
952  if (recv_string(fd, up.username, sizeof(up.username)) == -1
953  || recv_string(fd, up.password, sizeof(up.password)) == -1
954  || recv_string(fd, up.common_name, sizeof(up.common_name)) == -1
955  || recv_string(fd, ac_file_name, sizeof(ac_file_name)) == -1)
956  {
957  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "BACKGROUND: read error on command channel: code=%d, exiting",
958  command);
959  goto done;
960  }
961 
962  if (DEBUG(verb))
963  {
964 #if 0
965  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: USER/PASS: %s/%s",
966  up.username, up.password);
967 #else
968  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: USER: %s", up.username);
969 #endif
970  }
971 
972  /* If password is of the form SCRV1:base64:base64 split it up */
973  split_scrv1_password(&up);
974 
975  /* client wants deferred auth
976  */
977  if (strlen(ac_file_name) > 0)
978  {
979  do_deferred_pam_auth(fd, ac_file_name, service, &up);
980  break;
981  }
982 
983 
984  /* non-deferred auth: wait for pam result and send
985  * result back via control socketpair
986  */
987  if (pam_auth(service, &up)) /* Succeeded */
988  {
989  if (send_control(fd, RESPONSE_VERIFY_SUCCEEDED) == -1)
990  {
991  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "BACKGROUND: write error on response socket [2]");
992  goto done;
993  }
994  }
995  else /* Failed */
996  {
997  if (send_control(fd, RESPONSE_VERIFY_FAILED) == -1)
998  {
999  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "BACKGROUND: write error on response socket [3]");
1000  goto done;
1001  }
1002  }
1003  plugin_secure_memzero(up.password, sizeof(up.password));
1004  break;
1005 
1006  case COMMAND_EXIT:
1007  goto done;
1008 
1009  case -1:
1010  plugin_log(PLOG_ERR|PLOG_ERRNO, MODULE, "BACKGROUND: read error on command channel");
1011  goto done;
1012 
1013  default:
1014  plugin_log(PLOG_ERR, MODULE, "BACKGROUND: unknown command code: code=%d, exiting",
1015  command);
1016  goto done;
1017  }
1018  plugin_secure_memzero(up.response, sizeof(up.response));
1019  }
1020 done:
1021  plugin_secure_memzero(up.password, sizeof(up.password));
1022  plugin_secure_memzero(up.response, sizeof(up.response));
1023 #ifdef USE_PAM_DLOPEN
1024  dlclose_pam();
1025 #endif
1026  if (DEBUG(verb))
1027  {
1028  plugin_log(PLOG_NOTE, MODULE, "BACKGROUND: EXIT");
1029  }
1030 
1031  return;
1032 }
