59 static void
60 create_des_keys(const unsigned char *hash, unsigned char *key)
61 {
62  key[0] = hash[0];
63  key[1] = ((hash[0] & 1) << 7) | (hash[1] >> 1);
64  key[2] = ((hash[1] & 3) << 6) | (hash[2] >> 2);
65  key[3] = ((hash[2] & 7) << 5) | (hash[3] >> 3);
66  key[4] = ((hash[3] & 15) << 4) | (hash[4] >> 4);
67  key[5] = ((hash[4] & 31) << 3) | (hash[5] >> 5);
68  key[6] = ((hash[5] & 63) << 2) | (hash[6] >> 6);
69  key[7] = ((hash[6] & 127) << 1);
70  key_des_fixup(key, 8, 1);
71 }

74 gen_md4_hash(const uint8_t *data, int data_len, uint8_t *result)
75 {
76  /* result is 16 byte md4 hash */
77  const md_kt_t *md4_kt = md_kt_get("MD4");
78  uint8_t md[MD4_DIGEST_LENGTH];
79 
80  md_full(md4_kt, data, data_len, md);
81  memcpy(result, md, MD4_DIGEST_LENGTH);

84 static void
85 gen_hmac_md5(const uint8_t *data, int data_len, const uint8_t *key, int key_len,
86  uint8_t *result)
87 {

90 
91  hmac_ctx_init(hmac_ctx, key, key_len, md5_kt);
92  hmac_ctx_update(hmac_ctx, data, data_len);
93  hmac_ctx_final(hmac_ctx, result);
94  hmac_ctx_cleanup(hmac_ctx);
95  hmac_ctx_free(hmac_ctx);

98 static void
99 gen_timestamp(uint8_t *timestamp)
100 {
101  /* Copies 8 bytes long timestamp into "timestamp" buffer.
102  * Timestamp is Little-endian, 64-bit signed value representing the
103  * number of tenths of a microsecond since January 1, 1601.
104  */
105 
106  UINTEGER64 timestamp_ull;
107 
108  timestamp_ull = openvpn_time(NULL);
109  timestamp_ull = (timestamp_ull + UINT64(11644473600)) * UINT64(10000000);
110 
111  /* store little endian value */
112  timestamp[0] = timestamp_ull & UINT64(0xFF);
113  timestamp[1] = (timestamp_ull >> 8) & UINT64(0xFF);
114  timestamp[2] = (timestamp_ull >> 16) & UINT64(0xFF);
115  timestamp[3] = (timestamp_ull >> 24) & UINT64(0xFF);
116  timestamp[4] = (timestamp_ull >> 32) & UINT64(0xFF);
117  timestamp[5] = (timestamp_ull >> 40) & UINT64(0xFF);
118  timestamp[6] = (timestamp_ull >> 48) & UINT64(0xFF);
119  timestamp[7] = (timestamp_ull >> 56) & UINT64(0xFF);
120 }

173 static void
174 add_security_buffer(int sb_offset, void *data, int length,
175  unsigned char *msg_buf, int *msg_bufpos)
176 {
177  /* Adds security buffer data to a message and sets security buffer's
178  * offset and length */
179  msg_buf[sb_offset] = (unsigned char)length;
180  msg_buf[sb_offset + 2] = msg_buf[sb_offset];
181  msg_buf[sb_offset + 4] = (unsigned char)(*msg_bufpos & 0xff);
182  msg_buf[sb_offset + 5] = (unsigned char)((*msg_bufpos >> 8) & 0xff);
183  memcpy(&msg_buf[*msg_bufpos], data, msg_buf[sb_offset]);
184  *msg_bufpos += length;
185 }

204 const char *
205 ntlm_phase_3(const struct http_proxy_info *p, const char *phase_2,
206  struct gc_arena *gc)
207 {
208  /* NTLM handshake
209  *
210  * http://davenport.sourceforge.net/ntlm.html
211  *
212  */
213 
214  char pwbuf[sizeof(p->up.password) * 2]; /* for unicode password */
215  uint8_t buf2[128]; /* decoded reply from proxy */
216  uint8_t phase3[464];
217 
218  uint8_t md4_hash[MD4_DIGEST_LENGTH + 5];
219  uint8_t challenge[8], ntlm_response[24];
220  int i, ret_val;
221 
222  uint8_t ntlmv2_response[144];
223  char userdomain_u[256]; /* for uppercase unicode username and domain */
224  char userdomain[128]; /* the same as previous but ascii */
225  uint8_t ntlmv2_hash[MD5_DIGEST_LENGTH];
226  uint8_t ntlmv2_hmacmd5[16];
227  uint8_t *ntlmv2_blob = ntlmv2_response + 16; /* inside ntlmv2_response, length: 128 */
228  int ntlmv2_blob_size = 0;
229  int phase3_bufpos = 0x40; /* offset to next security buffer data to be added */
230  size_t len;
231 
232  char domain[128];
233  char username[128];
234  char *separator;
235 
236  bool ntlmv2_enabled = (p->auth_method == HTTP_AUTH_NTLM2);
237 
238  CLEAR(buf2);
239 
240  ASSERT(strlen(p->up.username) > 0);
241  ASSERT(strlen(p->up.password) > 0);
242 
243  /* username parsing */
244  separator = strchr(p->up.username, '\\');
245  if (separator == NULL)
246  {
247  strncpy(username, p->up.username, sizeof(username)-1);
248  username[sizeof(username)-1] = 0;
249  domain[0] = 0;
250  }
251  else
252  {
253  strncpy(username, separator+1, sizeof(username)-1);
254  username[sizeof(username)-1] = 0;
255  len = separator - p->up.username;
256  if (len > sizeof(domain) - 1)
257  {
258  len = sizeof(domain) - 1;
259  }
260  strncpy(domain, p->up.username, len);
261  domain[len] = 0;
262  }
263 
264 
265  /* fill 1st 16 bytes with md4 hash, disregard terminating null */
266  int unicode_len = unicodize(pwbuf, p->up.password) - 2;
267  gen_md4_hash((uint8_t *)pwbuf, unicode_len, md4_hash);
268 
269  /* pad to 21 bytes */
270  memset(md4_hash + MD4_DIGEST_LENGTH, 0, 5);
271 
272  ret_val = openvpn_base64_decode(phase_2, buf2, -1);
273  if (ret_val < 0)
274  {
275  return NULL;
276  }
277 
278  /* we can be sure that phase_2 is less than 128
279  * therefore buf2 needs to be (3/4 * 128) */
280 
281  /* extract the challenge from bytes 24-31 */
282  for (i = 0; i<8; i++)
283  {
284  challenge[i] = buf2[i+24];
285  }
286 
287  if (ntlmv2_enabled) /* Generate NTLMv2 response */
288  {
289  int tib_len;
290 
291  /* NTLMv2 hash */
292  strcpy(userdomain, username);
293  my_strupr(userdomain);
294  if (strlen(username) + strlen(domain) < sizeof(userdomain))
295  {
296  strcat(userdomain, domain);
297  }
298  else
299  {
300  msg(M_INFO, "Warning: Username or domain too long");
301  }
302  unicodize(userdomain_u, userdomain);
303  gen_hmac_md5((uint8_t *)userdomain_u, 2 * strlen(userdomain), md4_hash,
304  MD5_DIGEST_LENGTH, ntlmv2_hash);
305 
306  /* NTLMv2 Blob */
307  memset(ntlmv2_blob, 0, 128); /* Clear blob buffer */
308  ntlmv2_blob[0x00] = 1; /* Signature */
309  ntlmv2_blob[0x01] = 1; /* Signature */
310  ntlmv2_blob[0x04] = 0; /* Reserved */
311  gen_timestamp(&ntlmv2_blob[0x08]); /* 64-bit Timestamp */
312  gen_nonce(&ntlmv2_blob[0x10]); /* 64-bit Client Nonce */
313  ntlmv2_blob[0x18] = 0; /* Unknown, zero should work */
314 
315  /* Add target information block to the blob */
316 
317  /* Check for Target Information block */
318  /* The NTLM spec instructs to interpret these 4 consecutive bytes as a
319  * 32bit long integer. However, no endianness is specified.
320  * The code here and that found in other NTLM implementations point
321  * towards the assumption that the byte order on the wire has to
322  * match the order on the sending and receiving hosts. Probably NTLM has
323  * been thought to be always running on x86_64/i386 machine thus
324  * implying Little-Endian everywhere.
325  *
326  * This said, in case of future changes, we should keep in mind that the
327  * byte order on the wire for the NTLM header is LE.
328  */
329  const size_t hoff = 0x14;
330  unsigned long flags = buf2[hoff] | (buf2[hoff + 1] << 8)
331  |(buf2[hoff + 2] << 16) | (buf2[hoff + 3] << 24);
332  if ((flags & 0x00800000) == 0x00800000)
333  {
334  tib_len = buf2[0x28]; /* Get Target Information block size */
335  if (tib_len > 96)
336  {
337  tib_len = 96;
338  }
339 
340  {
341  uint8_t *tib_ptr;
342  uint8_t tib_pos = buf2[0x2c];
343  if (tib_pos + tib_len > sizeof(buf2))
344  {
345  return NULL;
346  }
347  /* Get Target Information block pointer */
348  tib_ptr = buf2 + tib_pos;
349  /* Copy Target Information block into the blob */
350  memcpy(&ntlmv2_blob[0x1c], tib_ptr, tib_len);
351  }
352  }
353  else
354  {
355  tib_len = 0;
356  }
357 
358  /* Unknown, zero works */
359  ntlmv2_blob[0x1c + tib_len] = 0;
360 
361  /* Get blob length */
362  ntlmv2_blob_size = 0x20 + tib_len;
363 
364  /* Add challenge from message 2 */
365  memcpy(&ntlmv2_response[8], challenge, 8);
366 
367  /* hmac-md5 */
368  gen_hmac_md5(&ntlmv2_response[8], ntlmv2_blob_size + 8, ntlmv2_hash,
369  MD5_DIGEST_LENGTH, ntlmv2_hmacmd5);
370 
371  /* Add hmac-md5 result to the blob.
372  * Note: This overwrites challenge previously written at
373  * ntlmv2_response[8..15] */
374  memcpy(ntlmv2_response, ntlmv2_hmacmd5, MD5_DIGEST_LENGTH);
375  }
376  else /* Generate NTLM response */
377  {
378  unsigned char key1[DES_KEY_LENGTH], key2[DES_KEY_LENGTH];
379  unsigned char key3[DES_KEY_LENGTH];
380 
381  create_des_keys(md4_hash, key1);
382  cipher_des_encrypt_ecb(key1, challenge, ntlm_response);
383 
384  create_des_keys(&md4_hash[DES_KEY_LENGTH - 1], key2);
385  cipher_des_encrypt_ecb(key2, challenge, &ntlm_response[DES_KEY_LENGTH]);
386 
387  create_des_keys(&md4_hash[2 * (DES_KEY_LENGTH - 1)], key3);
388  cipher_des_encrypt_ecb(key3, challenge,
389  &ntlm_response[DES_KEY_LENGTH * 2]);
390  }
391 
392 
393  memset(phase3, 0, sizeof(phase3)); /* clear reply */
394 
395  strcpy((char *)phase3, "NTLMSSP\0"); /* signature */
396  phase3[8] = 3; /* type 3 */
397 
398  if (ntlmv2_enabled) /* NTLMv2 response */
399  {
400  add_security_buffer(0x14, ntlmv2_response, ntlmv2_blob_size + 16,
401  phase3, &phase3_bufpos);
402  }
403  else /* NTLM response */
404  {
405  add_security_buffer(0x14, ntlm_response, 24, phase3, &phase3_bufpos);
406  }
407 
408  /* username in ascii */
409  add_security_buffer(0x24, username, strlen(username), phase3,
410  &phase3_bufpos);
411 
412  /* Set domain. If <domain> is empty, default domain will be used
413  * (i.e. proxy's domain) */
414  add_security_buffer(0x1c, domain, strlen(domain), phase3, &phase3_bufpos);
415 
416  /* other security buffers will be empty */
417  phase3[0x10] = phase3_bufpos; /* lm not used */
418  phase3[0x30] = phase3_bufpos; /* no workstation name supplied */
419  phase3[0x38] = phase3_bufpos; /* no session key */
420 
421  /* flags */
422  phase3[0x3c] = 0x02; /* negotiate oem */
423  phase3[0x3d] = 0x02; /* negotiate ntlm */
424 
425  return ((const char *)make_base64_string2((unsigned char *)phase3,
426  phase3_bufpos, gc));
427 }
