52 static struct key_type
53 tls_crypt_kt(void)
54 {
55  struct key_type kt;
56  kt.cipher = cipher_kt_get("AES-256-CTR");
57  kt.digest = md_kt_get("SHA256");
58 
59  if (!kt.cipher)
60  {
61  msg(M_WARN, "ERROR: --tls-crypt requires AES-256-CTR support.");
62  return (struct key_type) { 0 };
63  }
64  if (!kt.digest)
65  {
66  msg(M_WARN, "ERROR: --tls-crypt requires HMAC-SHA-256 support.");
67  return (struct key_type) { 0 };
68  }
69 
70  kt.cipher_length = cipher_kt_key_size(kt.cipher);
71  kt.hmac_length = md_kt_size(kt.digest);
72 
73  return kt;
74 }

107 bool
108 tls_crypt_wrap(const struct buffer *src, struct buffer *dst,
109  struct crypto_options *opt)
110 {
111  const struct key_ctx *ctx = &opt->key_ctx_bi.encrypt;
112  struct gc_arena gc;
113 
114  /* IV, packet-ID and implicit IV required for this mode. */
115  ASSERT(ctx->cipher);
116  ASSERT(ctx->hmac);
117  ASSERT(packet_id_initialized(&opt->packet_id));
118  ASSERT(hmac_ctx_size(ctx->hmac) == 256/8);
119 
120  gc_init(&gc);
121 
122  dmsg(D_PACKET_CONTENT, "TLS-CRYPT WRAP FROM: %s",
123  format_hex(BPTR(src), BLEN(src), 80, &gc));
124 
125  /* Get packet ID */
126  if (!packet_id_write(&opt->packet_id.send, dst, true, false))
127  {
128  msg(D_CRYPT_ERRORS, "TLS-CRYPT ERROR: packet ID roll over.");
129  goto err;
130  }
131 
132  dmsg(D_PACKET_CONTENT, "TLS-CRYPT WRAP AD: %s",
133  format_hex(BPTR(dst), BLEN(dst), 0, &gc));
134 
135  /* Buffer overflow check */
136  if (!buf_safe(dst, BLEN(src) + TLS_CRYPT_BLOCK_SIZE + TLS_CRYPT_TAG_SIZE))
137  {
138  msg(D_CRYPT_ERRORS, "TLS-CRYPT WRAP: buffer size error, "
139  "sc=%d so=%d sl=%d dc=%d do=%d dl=%d", src->capacity, src->offset,
140  src->len, dst->capacity, dst->offset, dst->len);
141  goto err;
142  }
143 
144  /* Calculate auth tag and synthetic IV */
145  {
146  uint8_t *tag = NULL;
147  hmac_ctx_reset(ctx->hmac);
148  hmac_ctx_update(ctx->hmac, BPTR(dst), BLEN(dst));
149  hmac_ctx_update(ctx->hmac, BPTR(src), BLEN(src));
150 
151  ASSERT(tag = buf_write_alloc(dst, TLS_CRYPT_TAG_SIZE));
152  hmac_ctx_final(ctx->hmac, tag);
153 
154  dmsg(D_PACKET_CONTENT, "TLS-CRYPT WRAP TAG: %s",
155  format_hex(tag, TLS_CRYPT_TAG_SIZE, 0, &gc));
156 
157  /* Use the 128 most significant bits of the tag as IV */
158  ASSERT(cipher_ctx_reset(ctx->cipher, tag));
159  }
160 
161  /* Encrypt src */
162  {
163  int outlen = 0;
164  ASSERT(cipher_ctx_update(ctx->cipher, BEND(dst), &outlen,
165  BPTR(src), BLEN(src)));
166  ASSERT(buf_inc_len(dst, outlen));
167  ASSERT(cipher_ctx_final(ctx->cipher, BPTR(dst), &outlen));
168  ASSERT(buf_inc_len(dst, outlen));
169  }
170 
171  dmsg(D_PACKET_CONTENT, "TLS-CRYPT WRAP TO: %s",
172  format_hex(BPTR(dst), BLEN(dst), 80, &gc));
173 
174  gc_free(&gc);
175  return true;
176 
177 err:
178  crypto_clear_error();
179  dst->len = 0;
180  gc_free(&gc);
181  return false;
182 }

184 bool
185 tls_crypt_unwrap(const struct buffer *src, struct buffer *dst,
186  struct crypto_options *opt)
187 {
188  static const char error_prefix[] = "tls-crypt unwrap error";
189  const struct key_ctx *ctx = &opt->key_ctx_bi.decrypt;
190  struct gc_arena gc;
191 
192  gc_init(&gc);
193 
194  ASSERT(opt);
195  ASSERT(src->len > 0);
196  ASSERT(ctx->cipher);
197  ASSERT(packet_id_initialized(&opt->packet_id)
198  || (opt->flags & CO_IGNORE_PACKET_ID));
199 
200  dmsg(D_PACKET_CONTENT, "TLS-CRYPT UNWRAP FROM: %s",
201  format_hex(BPTR(src), BLEN(src), 80, &gc));
202 
203  if (buf_len(src) < TLS_CRYPT_OFF_CT)
204  {
205  CRYPT_ERROR("packet too short");
206  }
207 
208  /* Decrypt cipher text */
209  {
210  int outlen = 0;
211 
212  /* Buffer overflow check (should never fail) */
213  if (!buf_safe(dst, BLEN(src) - TLS_CRYPT_OFF_CT + TLS_CRYPT_BLOCK_SIZE))
214  {
215  CRYPT_ERROR("potential buffer overflow");
216  }
217 
218  if (!cipher_ctx_reset(ctx->cipher, BPTR(src) + TLS_CRYPT_OFF_TAG))
219  {
220  CRYPT_ERROR("cipher reset failed");
221  }
222  if (!cipher_ctx_update(ctx->cipher, BPTR(dst), &outlen,
223  BPTR(src) + TLS_CRYPT_OFF_CT, BLEN(src) - TLS_CRYPT_OFF_CT))
224  {
225  CRYPT_ERROR("cipher update failed");
226  }
227  ASSERT(buf_inc_len(dst, outlen));
228  if (!cipher_ctx_final(ctx->cipher, BPTR(dst), &outlen))
229  {
230  CRYPT_ERROR("cipher final failed");
231  }
232  ASSERT(buf_inc_len(dst, outlen));
233  }
234 
235  /* Check authentication */
236  {
237  const uint8_t *tag = BPTR(src) + TLS_CRYPT_OFF_TAG;
238  uint8_t tag_check[TLS_CRYPT_TAG_SIZE] = { 0 };
239 
240  dmsg(D_PACKET_CONTENT, "TLS-CRYPT UNWRAP AD: %s",
241  format_hex(BPTR(src), TLS_CRYPT_OFF_TAG, 0, &gc));
242  dmsg(D_PACKET_CONTENT, "TLS-CRYPT UNWRAP TO: %s",
243  format_hex(BPTR(dst), BLEN(dst), 80, &gc));
244 
245  hmac_ctx_reset(ctx->hmac);
246  hmac_ctx_update(ctx->hmac, BPTR(src), TLS_CRYPT_OFF_TAG);
247  hmac_ctx_update(ctx->hmac, BPTR(dst), BLEN(dst));
248  hmac_ctx_final(ctx->hmac, tag_check);
249 
250  if (memcmp_constant_time(tag, tag_check, sizeof(tag_check)))
251  {
252  dmsg(D_CRYPTO_DEBUG, "tag : %s",
253  format_hex(tag, sizeof(tag_check), 0, &gc));
254  dmsg(D_CRYPTO_DEBUG, "tag_check: %s",
255  format_hex(tag_check, sizeof(tag_check), 0, &gc));
256  CRYPT_ERROR("packet authentication failed");
257  }
258  }
259 
260  /* Check replay */
261  if (!(opt->flags & CO_IGNORE_PACKET_ID))
262  {
263  struct packet_id_net pin;
264  struct buffer tmp = *src;
265  ASSERT(buf_advance(&tmp, TLS_CRYPT_OFF_PID));
266  ASSERT(packet_id_read(&pin, &tmp, true));
267  if (!crypto_check_replay(opt, &pin, error_prefix, &gc))
268  {
269  CRYPT_ERROR("packet replay");
270  }
271  }
272 
273  gc_free(&gc);
274 
275  return true;
276 
277 error_exit:
278  crypto_clear_error();
279  dst->len = 0;
280  gc_free(&gc);
281  return false;
282 }

347 static bool
348 tls_crypt_v2_wrap_client_key(struct buffer *wkc,
349  const struct key2 *src_key,
350  const struct buffer *src_metadata,
351  struct key_ctx *server_key, struct gc_arena *gc)
352 {
353  cipher_ctx_t *cipher_ctx = server_key->cipher;
354  struct buffer work = alloc_buf_gc(TLS_CRYPT_V2_MAX_WKC_LEN
355  + cipher_ctx_block_size(cipher_ctx), gc);
356 
357  /* Calculate auth tag and synthetic IV */
358  uint8_t *tag = buf_write_alloc(&work, TLS_CRYPT_TAG_SIZE);
359  if (!tag)
360  {
361  msg(M_WARN, "ERROR: could not write tag");
362  return false;
363  }
364  uint16_t net_len = htons(sizeof(src_key->keys) + BLEN(src_metadata)
365  + TLS_CRYPT_V2_TAG_SIZE + sizeof(uint16_t));
366  hmac_ctx_t *hmac_ctx = server_key->hmac;
367  hmac_ctx_reset(hmac_ctx);
368  hmac_ctx_update(hmac_ctx, (void *)&net_len, sizeof(net_len));
369  hmac_ctx_update(hmac_ctx, (void *)src_key->keys, sizeof(src_key->keys));
370  hmac_ctx_update(hmac_ctx, BPTR(src_metadata), BLEN(src_metadata));
371  hmac_ctx_final(hmac_ctx, tag);
372 
373  dmsg(D_CRYPTO_DEBUG, "TLS-CRYPT WRAP TAG: %s",
374  format_hex(tag, TLS_CRYPT_TAG_SIZE, 0, gc));
375 
376  /* Use the 128 most significant bits of the tag as IV */
377  ASSERT(cipher_ctx_reset(cipher_ctx, tag));
378 
379  /* Overflow check (OpenSSL requires an extra block in the dst buffer) */
380  if (buf_forward_capacity(&work) < (sizeof(src_key->keys)
381  + BLEN(src_metadata)
382  + sizeof(net_len)
383  + cipher_ctx_block_size(cipher_ctx)))
384  {
385  msg(M_WARN, "ERROR: could not crypt: insufficient space in dst");
386  return false;
387  }
388 
389  /* Encrypt */
390  int outlen = 0;
391  ASSERT(cipher_ctx_update(cipher_ctx, BEND(&work), &outlen,
392  (void *)src_key->keys, sizeof(src_key->keys)));
393  ASSERT(buf_inc_len(&work, outlen));
394  ASSERT(cipher_ctx_update(cipher_ctx, BEND(&work), &outlen,
395  BPTR(src_metadata), BLEN(src_metadata)));
396  ASSERT(buf_inc_len(&work, outlen));
397  ASSERT(cipher_ctx_final(cipher_ctx, BEND(&work), &outlen));
398  ASSERT(buf_inc_len(&work, outlen));
399  ASSERT(buf_write(&work, &net_len, sizeof(net_len)));
400 
401  return buf_copy(wkc, &work);
402 }

404 static bool
405 tls_crypt_v2_unwrap_client_key(struct key2 *client_key, struct buffer *metadata,
406  struct buffer wrapped_client_key,
407  struct key_ctx *server_key)
408 {
409  const char *error_prefix = __func__;
410  bool ret = false;
411  struct gc_arena gc = gc_new();
412  /* The crypto API requires one extra cipher block of buffer head room when
413  * decrypting, which nicely matches the tag size of WKc. So
414  * TLS_CRYPT_V2_MAX_WKC_LEN is always large enough for the plaintext. */
415  uint8_t plaintext_buf_data[TLS_CRYPT_V2_MAX_WKC_LEN] = { 0 };
416  struct buffer plaintext = { 0 };
417 
418  dmsg(D_TLS_DEBUG_MED, "%s: unwrapping client key (len=%d): %s", __func__,
419  BLEN(&wrapped_client_key), format_hex(BPTR(&wrapped_client_key),
420  BLEN(&wrapped_client_key),
421  0, &gc));
422 
423  if (TLS_CRYPT_V2_MAX_WKC_LEN < BLEN(&wrapped_client_key))
424  {
425  CRYPT_ERROR("wrapped client key too big");
426  }
427 
428  /* Decrypt client key and metadata */
429  uint16_t net_len = 0;
430  const uint8_t *tag = BPTR(&wrapped_client_key);
431 
432  if (BLEN(&wrapped_client_key) < sizeof(net_len))
433  {
434  CRYPT_ERROR("failed to read length");
435  }
436  memcpy(&net_len, BEND(&wrapped_client_key) - sizeof(net_len),
437  sizeof(net_len));
438 
439  if (ntohs(net_len) != BLEN(&wrapped_client_key))
440  {
441  dmsg(D_TLS_DEBUG_LOW, "%s: net_len=%u, BLEN=%i", __func__,
442  ntohs(net_len), BLEN(&wrapped_client_key));
443  CRYPT_ERROR("invalid length");
444  }
445 
446  buf_inc_len(&wrapped_client_key, -(int)sizeof(net_len));
447 
448  if (!buf_advance(&wrapped_client_key, TLS_CRYPT_TAG_SIZE))
449  {
450  CRYPT_ERROR("failed to read tag");
451  }
452 
453  if (!cipher_ctx_reset(server_key->cipher, tag))
454  {
455  CRYPT_ERROR("failed to initialize IV");
456  }
457  buf_set_write(&plaintext, plaintext_buf_data, sizeof(plaintext_buf_data));
458  int outlen = 0;
459  if (!cipher_ctx_update(server_key->cipher, BPTR(&plaintext), &outlen,
460  BPTR(&wrapped_client_key),
461  BLEN(&wrapped_client_key)))
462  {
463  CRYPT_ERROR("could not decrypt client key");
464  }
465  ASSERT(buf_inc_len(&plaintext, outlen));
466 
467  if (!cipher_ctx_final(server_key->cipher, BEND(&plaintext), &outlen))
468  {
469  CRYPT_ERROR("cipher final failed");
470  }
471  ASSERT(buf_inc_len(&plaintext, outlen));
472 
473  /* Check authentication */
474  uint8_t tag_check[TLS_CRYPT_TAG_SIZE] = { 0 };
475  hmac_ctx_reset(server_key->hmac);
476  hmac_ctx_update(server_key->hmac, (void *)&net_len, sizeof(net_len));
477  hmac_ctx_update(server_key->hmac, BPTR(&plaintext),
478  BLEN(&plaintext));
479  hmac_ctx_final(server_key->hmac, tag_check);
480 
481  if (memcmp_constant_time(tag, tag_check, sizeof(tag_check)))
482  {
483  dmsg(D_CRYPTO_DEBUG, "tag : %s",
484  format_hex(tag, sizeof(tag_check), 0, &gc));
485  dmsg(D_CRYPTO_DEBUG, "tag_check: %s",
486  format_hex(tag_check, sizeof(tag_check), 0, &gc));
487  CRYPT_ERROR("client key authentication error");
488  }
489 
490  if (buf_len(&plaintext) < sizeof(client_key->keys))
491  {
492  CRYPT_ERROR("failed to read client key");
493  }
494  memcpy(&client_key->keys, BPTR(&plaintext), sizeof(client_key->keys));
495  ASSERT(buf_advance(&plaintext, sizeof(client_key->keys)));
496 
497  if (!buf_copy(metadata, &plaintext))
498  {
499  CRYPT_ERROR("metadata too large for supplied buffer");
500  }
501 
502  ret = true;
503 error_exit:
504  if (!ret)
505  {
506  secure_memzero(client_key, sizeof(*client_key));
507  }
508  buf_clear(&plaintext);
509  gc_free(&gc);
510  return ret;
511 }

572 bool
573 tls_crypt_v2_extract_client_key(struct buffer *buf,
574  struct tls_wrap_ctx *ctx,
575  const struct tls_options *opt)
576 {
577  if (!ctx->tls_crypt_v2_server_key.cipher)
578  {
579  msg(D_TLS_ERRORS,
580  "Client wants tls-crypt-v2, but no server key present.");
581  return false;
582  }
583 
584  msg(D_HANDSHAKE, "Control Channel: using tls-crypt-v2 key");
585 
586  struct buffer wrapped_client_key = *buf;
587  uint16_t net_len = 0;
588 
589  if (BLEN(&wrapped_client_key) < sizeof(net_len))
590  {
591  msg(D_TLS_ERRORS, "Can not read tls-crypt-v2 client key length");
592  return false;
593  }
594  memcpy(&net_len, BEND(&wrapped_client_key) - sizeof(net_len),
595  sizeof(net_len));
596 
597  size_t wkc_len = ntohs(net_len);
598  if (!buf_advance(&wrapped_client_key, BLEN(&wrapped_client_key) - wkc_len))
599  {
600  msg(D_TLS_ERRORS, "Can not locate tls-crypt-v2 client key");
601  return false;
602  }
603 
604  struct key2 client_key = { 0 };
605  ctx->tls_crypt_v2_metadata = alloc_buf(TLS_CRYPT_V2_MAX_METADATA_LEN);
606  if (!tls_crypt_v2_unwrap_client_key(&client_key,
607  &ctx->tls_crypt_v2_metadata,
608  wrapped_client_key,
609  &ctx->tls_crypt_v2_server_key))
610  {
611  msg(D_TLS_ERRORS, "Can not unwrap tls-crypt-v2 client key");
612  secure_memzero(&client_key, sizeof(client_key));
613  return false;
614  }
615 
616  /* Load the decrypted key */
617  ctx->mode = TLS_WRAP_CRYPT;
618  ctx->cleanup_key_ctx = true;
619  ctx->opt.flags |= CO_PACKET_ID_LONG_FORM;

641 void
642 tls_crypt_v2_write_client_key_file(const char *filename,
643  const char *b64_metadata,
644  const char *server_key_file,
645  bool server_key_inline)
646 {
647  struct gc_arena gc = gc_new();
648  struct key_ctx server_key = { 0 };
649  struct buffer client_key_pem = { 0 };
650  struct buffer dst = alloc_buf_gc(TLS_CRYPT_V2_CLIENT_KEY_LEN
651  + TLS_CRYPT_V2_MAX_WKC_LEN, &gc);
652  struct key2 client_key = { 2 };
653 
654  if (!rand_bytes((void *)client_key.keys, sizeof(client_key.keys)))
655  {
656  msg(M_FATAL, "ERROR: could not generate random key");
657  goto cleanup;
658  }
659  ASSERT(buf_write(&dst, client_key.keys, sizeof(client_key.keys)));
660 
661  struct buffer metadata = alloc_buf_gc(TLS_CRYPT_V2_MAX_METADATA_LEN, &gc);
662  if (b64_metadata)
663  {
664  if (TLS_CRYPT_V2_MAX_B64_METADATA_LEN < strlen(b64_metadata))
665  {
666  msg(M_FATAL,
667  "ERROR: metadata too long (%d bytes, max %u bytes)",
668  (int)strlen(b64_metadata), TLS_CRYPT_V2_MAX_B64_METADATA_LEN);
669  }
670  ASSERT(buf_write(&metadata, &TLS_CRYPT_METADATA_TYPE_USER, 1));
671  int decoded_len = openvpn_base64_decode(b64_metadata, BEND(&metadata),
672  BCAP(&metadata));
673  if (decoded_len < 0)
674  {
675  msg(M_FATAL, "ERROR: failed to base64 decode provided metadata");
676  goto cleanup;
677  }
678  ASSERT(buf_inc_len(&metadata, decoded_len));
679  }
680  else
681  {
682  int64_t timestamp = htonll((uint64_t)now);
683  ASSERT(buf_write(&metadata, &TLS_CRYPT_METADATA_TYPE_TIMESTAMP, 1));
684  ASSERT(buf_write(&metadata, &timestamp, sizeof(timestamp)));
685  }
686 
687  tls_crypt_v2_init_server_key(&server_key, true, server_key_file,
688  server_key_inline);
689  if (!tls_crypt_v2_wrap_client_key(&dst, &client_key, &metadata, &server_key,
690  &gc))
691  {
692  msg(M_FATAL, "ERROR: could not wrap generated client key");
693  goto cleanup;
694  }
695 
696  /* PEM-encode Kc || WKc */
697  if (!crypto_pem_encode(tls_crypt_v2_cli_pem_name, &client_key_pem, &dst,
698  &gc))
699  {
700  msg(M_FATAL, "ERROR: could not PEM-encode client key");
701  goto cleanup;
702  }
703 
704  const char *client_file = filename;
705  bool client_inline = false;
706 
707  if (!filename || streq(filename, ""))
708  {
709  printf("%.*s\n", BLEN(&client_key_pem), BPTR(&client_key_pem));
710  client_file = (const char *)BPTR(&client_key_pem);
711  client_inline = true;
712  }
713  else if (!buffer_write_file(filename, &client_key_pem))
714  {
715  msg(M_FATAL, "ERROR: could not write client key file");
716  goto cleanup;
717  }
718 
719  /* Sanity check: load client key (as "client") */
720  struct key_ctx_bi test_client_key;
721  struct buffer test_wrapped_client_key;
722  msg(D_GENKEY, "Testing client-side key loading...");
723  tls_crypt_v2_init_client_key(&test_client_key, &test_wrapped_client_key,
724  client_file, client_inline);
725  free_key_ctx_bi(&test_client_key);
726 
727  /* Sanity check: unwrap and load client key (as "server") */
728  struct buffer test_metadata = alloc_buf_gc(TLS_CRYPT_V2_MAX_METADATA_LEN,
729  &gc);
730  struct key2 test_client_key2 = { 0 };
731  free_key_ctx(&server_key);
732  tls_crypt_v2_init_server_key(&server_key, false, server_key_file,
733  server_key_inline);
734  msg(D_GENKEY, "Testing server-side key loading...");
735  ASSERT(tls_crypt_v2_unwrap_client_key(&test_client_key2, &test_metadata,
736  test_wrapped_client_key, &server_key));
737  secure_memzero(&test_client_key2, sizeof(test_client_key2));
738  free_buf(&test_wrapped_client_key);
739 
740 cleanup:
741  secure_memzero(&client_key, sizeof(client_key));
742  free_key_ctx(&server_key);
743  buf_clear(&client_key_pem);
744  buf_clear(&dst);
745 
746  gc_free(&gc);
747 }
