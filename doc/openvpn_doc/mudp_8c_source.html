44  * the instance doesn't exist, create it while

48 struct multi_instance *
49 multi_get_create_instance_udp(struct multi_context *m, bool *floated)
50 {
51  struct gc_arena gc = gc_new();
52  struct mroute_addr real;
53  struct multi_instance *mi = NULL;
54  struct hash *hash = m->hash;
55 
56  if (mroute_extract_openvpn_sockaddr(&real, &m->top.c2.from.dest, true)
57  && m->top.c2.buf.len > 0)
58  {
59  struct hash_element *he;
60  const uint32_t hv = hash_value(hash, &real);
61  struct hash_bucket *bucket = hash_bucket(hash, hv);
62  uint8_t *ptr = BPTR(&m->top.c2.buf);
63  uint8_t op = ptr[0] >> P_OPCODE_SHIFT;
64  bool v2 = (op == P_DATA_V2) && (m->top.c2.buf.len >= (1 + 3));
65  bool peer_id_disabled = false;
66 
67  /* make sure buffer has enough length to read opcode (1 byte) and peer-id (3 bytes) */
68  if (v2)
69  {
70  uint32_t peer_id = ntohl(*(uint32_t *)ptr) & 0xFFFFFF;
71  peer_id_disabled = (peer_id == MAX_PEER_ID);
72 
73  if (!peer_id_disabled && (peer_id < m->max_clients) && (m->instances[peer_id]))
74  {
75  mi = m->instances[peer_id];
76 
77  *floated = !link_socket_actual_match(&mi->context.c2.from, &m->top.c2.from);
78 
79  if (*floated)
80  {
81  /* reset prefix, since here we are not sure peer is the one it claims to be */
82  ungenerate_prefix(mi);
83  msg(D_MULTI_MEDIUM, "Float requested for peer %" PRIu32 " to %s", peer_id,
84  mroute_addr_print(&real, &gc));
85  }
86  }
87  }
88  if (!v2 || peer_id_disabled)
89  {
90  he = hash_lookup_fast(hash, bucket, &real, hv);
91  if (he)
92  {
93  mi = (struct multi_instance *) he->value;
94  }
95  }
96  if (!mi)
97  {
98  if (!m->top.c2.tls_auth_standalone
99  || tls_pre_decrypt_lite(m->top.c2.tls_auth_standalone, &m->top.c2.from, &m->top.c2.buf))
100  {
101  if (frequency_limit_event_allowed(m->new_connection_limiter))
102  {
103  mi = multi_create_instance(m, &real);
104  if (mi)
105  {
106  int i;
107 
108  hash_add_fast(hash, bucket, &mi->real, hv, mi);
109  mi->did_real_hash = true;
110 
111  /* max_clients must be less then max peer-id value */
112  ASSERT(m->max_clients < MAX_PEER_ID);
113 
114  for (i = 0; i < m->max_clients; ++i)
115  {
116  if (!m->instances[i])
117  {
118  mi->context.c2.tls_multi->peer_id = i;
119  m->instances[i] = mi;
120  break;
121  }
122  }
123 
124  /* should not really end up here, since multi_create_instance returns null
125  * if amount of clients exceeds max_clients */
126  ASSERT(i < m->max_clients);
127  }
128  }
129  else
130  {
131  msg(D_MULTI_ERRORS,
132  "MULTI: Connection from %s would exceed new connection frequency limit as controlled by --connect-freq",
133  mroute_addr_print(&real, &gc));
134  }
135  }
136  }
137 
138 #ifdef ENABLE_DEBUG
139  if (check_debug_level(D_MULTI_DEBUG))
140  {
141  const char *status = mi ? "[ok]" : "[failed]";
142 
143  dmsg(D_MULTI_DEBUG, "GET INST BY REAL: %s %s",
144  mroute_addr_print(&real, &gc),
145  status);
146  }
147 #endif
148  }
149 
150  gc_free(&gc);
151  ASSERT(!(mi && mi->halt));
152  return mi;
153 }

168 /*
169  * Process an I/O event.
170  */
171 static void
172 multi_process_io_udp(struct multi_context *m)
173 {
174  const unsigned int status = m->top.c2.event_set_status;
175  const unsigned int mpp_flags = m->top.c2.fast_io
176  ? (MPP_CONDITIONAL_PRE_SELECT | MPP_CLOSE_ON_SIGNAL)
177  : (MPP_PRE_SELECT | MPP_CLOSE_ON_SIGNAL);
178 
179 #ifdef MULTI_DEBUG_EVENT_LOOP
180  char buf[16];
181  buf[0] = 0;
182  if (status & SOCKET_READ)
183  {
184  strcat(buf, "SR/");
185  }
186  else if (status & SOCKET_WRITE)
187  {
188  strcat(buf, "SW/");
189  }
190  else if (status & TUN_READ)
191  {
192  strcat(buf, "TR/");
193  }
194  else if (status & TUN_WRITE)
195  {
196  strcat(buf, "TW/");
197  }
198 #ifdef ENABLE_ASYNC_PUSH
199  else if (status & FILE_CLOSED)
200  {
201  strcat(buf, "FC/");
202  }
203 #endif
204  printf("IO %s\n", buf);
205 #endif /* ifdef MULTI_DEBUG_EVENT_LOOP */
206 
207 #ifdef ENABLE_MANAGEMENT
208  if (status & (MANAGEMENT_READ|MANAGEMENT_WRITE))
209  {
210  ASSERT(management);
211  management_io(management);
212  }
213 #endif
214 
215  /* UDP port ready to accept write */
216  if (status & SOCKET_WRITE)
217  {
218  multi_process_outgoing_link(m, mpp_flags);
219  }
220  /* TUN device ready to accept write */
221  else if (status & TUN_WRITE)
222  {
223  multi_process_outgoing_tun(m, mpp_flags);
224  }
225  /* Incoming data on UDP port */
226  else if (status & SOCKET_READ)
227  {
228  read_incoming_link(&m->top);
229  if (!IS_SIG(&m->top))
230  {
231  multi_process_incoming_link(m, NULL, mpp_flags);
232  }
233  }
234  /* Incoming data on TUN device */
235  else if (status & TUN_READ)
236  {
237  read_incoming_tun(&m->top);
238  if (!IS_SIG(&m->top))
239  {
240  multi_process_incoming_tun(m, mpp_flags);
241  }
242  }
243 #ifdef ENABLE_ASYNC_PUSH
244  /* INOTIFY callback */
245  else if (status & FILE_CLOSED)
246  {
247  multi_process_file_closed(m, mpp_flags);
248  }
249 #endif
250 }

252 /*
253  * Return the io_wait() flags appropriate for
254  * a point-to-multipoint tunnel.
255  */
256 static inline unsigned int
257 p2mp_iow_flags(const struct multi_context *m)
258 {
259  unsigned int flags = IOW_WAIT_SIGNAL;
260  if (m->pending)
261  {
262  if (TUN_OUT(&m->pending->context))
263  {
264  flags |= IOW_TO_TUN;
265  }
266  if (LINK_OUT(&m->pending->context))
267  {
268  flags |= IOW_TO_LINK;
269  }
270  }
271  else if (mbuf_defined(m->mbuf))
272  {
273  flags |= IOW_MBUF;
274  }
275  else
276  {
277  flags |= IOW_READ;
278  }
279 #ifdef _WIN32
280  if (tuntap_ring_empty(m->top.c1.tuntap))
281  {
282  flags &= ~IOW_READ_TUN;
283  }
284 #endif
285  return flags;
286 }

300 static void
301 tunnel_server_udp_single_threaded(struct context *top)
302 {
303  struct multi_context multi;
304 
305  top->mode = CM_TOP;
306  context_clear_2(top);
307 
308  /* initialize top-tunnel instance */
309  init_instance_handle_signals(top, top->es, CC_HARD_USR1_TO_HUP);
310  if (IS_SIG(top))
311  {
312  return;
313  }
314 
315  /* initialize global multi_context object */
316  multi_init(&multi, top, false, MC_SINGLE_THREADED);
317 
318  /* initialize our cloned top object */
319  multi_top_init(&multi, top);
320 
321  /* initialize management interface */
322  init_management_callback_multi(&multi);
323 
324  /* finished with initialization */
325  initialization_sequence_completed(top, ISC_SERVER); /* --mode server --proto udp */
326 
327 #ifdef ENABLE_ASYNC_PUSH
328  multi.top.c2.inotify_fd = inotify_init();
329  if (multi.top.c2.inotify_fd < 0)
330  {
331  msg(D_MULTI_ERRORS | M_ERRNO, "MULTI: inotify_init error");
332  }
333 #endif
334 
335  /* per-packet event loop */
336  while (true)
337  {
338  perf_push(PERF_EVENT_LOOP);
339 
340  /* set up and do the io_wait() */
341  multi_get_timeout(&multi, &multi.top.c2.timeval);
342  io_wait(&multi.top, p2mp_iow_flags(&multi));
343  MULTI_CHECK_SIG(&multi);
344 
345  /* check on status of coarse timers */
346  multi_process_per_second_timers(&multi);
347 
348  /* timeout? */
349  if (multi.top.c2.event_set_status == ES_TIMEOUT)
350  {
351  multi_process_timeout(&multi, MPP_PRE_SELECT|MPP_CLOSE_ON_SIGNAL);
352  }
353  else
354  {
355  /* process I/O */
356  multi_process_io_udp(&multi);
357  MULTI_CHECK_SIG(&multi);
358  }
359 
360  perf_pop();
361  }
362 
363 #ifdef ENABLE_ASYNC_PUSH
364  close(top->c2.inotify_fd);
365 #endif
366 
367  /* shut down management interface */
368  uninit_management_callback();
369 
370  /* save ifconfig-pool */
371  multi_ifconfig_pool_persist(&multi, true);
372 
373  /* tear down tunnel instance (unless --persist-tun) */
374  multi_uninit(&multi);
375  multi_top_free(&multi);
376  close_instance(top);
377 }
