344 static result_t
345 verify_peer_cert(const struct tls_options *opt, openvpn_x509_cert_t *peer_cert,
346  const char *subject, const char *common_name)
347 {
348  /* verify certificate nsCertType */
349  if (opt->ns_cert_type != NS_CERT_CHECK_NONE)
350  {
351  if (SUCCESS == x509_verify_ns_cert_type(peer_cert, opt->ns_cert_type))
352  {
353  msg(D_HANDSHAKE, "VERIFY OK: nsCertType=%s",
354  print_nsCertType(opt->ns_cert_type));
355  }
356  else
357  {
358  msg(D_HANDSHAKE, "VERIFY nsCertType ERROR: %s, require nsCertType=%s",
359  subject, print_nsCertType(opt->ns_cert_type));
360  return FAILURE; /* Reject connection */
361  }
362  }
363 
364  /* verify certificate ku */
365  if (opt->remote_cert_ku[0] != 0)
366  {
367  if (SUCCESS == x509_verify_cert_ku(peer_cert, opt->remote_cert_ku, MAX_PARMS))
368  {
369  msg(D_HANDSHAKE, "VERIFY KU OK");
370  }
371  else
372  {
373  msg(D_HANDSHAKE, "VERIFY KU ERROR");
374  return FAILURE; /* Reject connection */
375  }
376  }
377 
378  /* verify certificate eku */
379  if (opt->remote_cert_eku != NULL)
380  {
381  if (SUCCESS == x509_verify_cert_eku(peer_cert, opt->remote_cert_eku))
382  {
383  msg(D_HANDSHAKE, "VERIFY EKU OK");
384  }
385  else
386  {
387  msg(D_HANDSHAKE, "VERIFY EKU ERROR");
388  return FAILURE; /* Reject connection */
389  }
390  }
391 
392  /* verify X509 name or username against --verify-x509-[user]name */
393  if (opt->verify_x509_type != VERIFY_X509_NONE)
394  {
395  if ( (opt->verify_x509_type == VERIFY_X509_SUBJECT_DN
396  && strcmp(opt->verify_x509_name, subject) == 0)
397  || (opt->verify_x509_type == VERIFY_X509_SUBJECT_RDN
398  && strcmp(opt->verify_x509_name, common_name) == 0)
399  || (opt->verify_x509_type == VERIFY_X509_SUBJECT_RDN_PREFIX
400  && strncmp(opt->verify_x509_name, common_name,
401  strlen(opt->verify_x509_name)) == 0) )
402  {
403  msg(D_HANDSHAKE, "VERIFY X509NAME OK: %s", subject);
404  }
405  else
406  {
407  msg(D_HANDSHAKE, "VERIFY X509NAME ERROR: %s, must be %s",
408  subject, opt->verify_x509_name);
409  return FAILURE; /* Reject connection */
410  }
411  }
412 
413  return SUCCESS;
414 }

645 result_t
646 verify_cert(struct tls_session *session, openvpn_x509_cert_t *cert, int cert_depth)
647 {
648  result_t ret = FAILURE;
649  char *subject = NULL;
650  char common_name[TLS_USERNAME_LEN+1] = {0}; /* null-terminated */
651  const struct tls_options *opt;
652  struct gc_arena gc = gc_new();
653 
654  opt = session->opt;
655  ASSERT(opt);
656 
657  session->verified = false;
658 
659  /* get the X509 name */
660  subject = x509_get_subject(cert, &gc);
661  if (!subject)
662  {
663  msg(D_TLS_ERRORS, "VERIFY ERROR: depth=%d, could not extract X509 "
664  "subject string from certificate", cert_depth);
665  goto cleanup;
666  }
667 
668  /* enforce character class restrictions in X509 name */
669  string_mod_remap_name(subject);
670  string_replace_leading(subject, '-', '_');
671 
672  /* extract the username (default is CN) */
673  if (SUCCESS != backend_x509_get_username(common_name, sizeof(common_name),
674  opt->x509_username_field, cert))
675  {
676  if (!cert_depth)
677  {
678  msg(D_TLS_ERRORS, "VERIFY ERROR: could not extract %s from X509 "
679  "subject string ('%s') -- note that the username length is "
680  "limited to %d characters",
681  opt->x509_username_field,
682  subject,
683  TLS_USERNAME_LEN);
684  goto cleanup;
685  }
686  }
687 
688  /* enforce character class restrictions in common name */
689  string_mod_remap_name(common_name);
690 
691  /* warn if cert chain is too deep */
692  if (cert_depth >= MAX_CERT_DEPTH)
693  {
694  msg(D_TLS_ERRORS, "TLS Error: Convoluted certificate chain detected with depth [%d] greater than %d", cert_depth, MAX_CERT_DEPTH);
695  goto cleanup; /* Reject connection */
696  }
697 
698  /* verify level 1 cert, i.e. the CA that signed our leaf cert */
699  if (cert_depth == 1 && opt->verify_hash)
700  {
701  struct buffer ca_hash = {0};
702 
703  switch (opt->verify_hash_algo)
704  {
705  case MD_SHA1:
706  ca_hash = x509_get_sha1_fingerprint(cert, &gc);
707  break;
708 
709  case MD_SHA256:
710  ca_hash = x509_get_sha256_fingerprint(cert, &gc);
711  break;
712 
713  default:
714  /* This should normally not happen at all; the algorithm used
715  * is parsed by add_option() [options.c] and set to a predefined
716  * value in an enumerated type. So if this unlikely scenario
717  * happens, consider this a failure
718  */
719  msg(M_WARN, "Unexpected invalid algorithm used with "
720  "--verify-hash (%i)", opt->verify_hash_algo);
721  ret = FAILURE;
722  goto cleanup;
723  }
724 
725  if (memcmp(BPTR(&ca_hash), opt->verify_hash, BLEN(&ca_hash)))
726  {
727  msg(D_TLS_ERRORS, "TLS Error: level-1 certificate hash verification failed");
728  goto cleanup;
729  }
730  }
731 
732  /* save common name in session object */
733  if (cert_depth == 0)
734  {
735  set_common_name(session, common_name);
736  }
737 
738  session->verify_maxlevel = max_int(session->verify_maxlevel, cert_depth);
739 
740  /* export certificate values to the environment */
741  verify_cert_set_env(opt->es, cert, cert_depth, subject, common_name,
742  opt->x509_track);
743 
744  /* export current untrusted IP */

788  ret = SUCCESS;
789 
790 cleanup:
791

1274  *
1275  * Will set session->ks[KS_PRIMARY].authenticated according to
1276  * result of the username/password verification
1277  */

1279 verify_user_pass(struct user_pass *up, struct tls_multi *multi,
1280  struct tls_session *session)
1281 {

1283  bool s2 = true;
1284  struct key_state *ks = &session->key[KS_PRIMARY]; /* primary key */
1285

1287  int man_def_auth = KMDA_UNDEF;
1288 
1289  if (management_enable_def_auth(management))
1290  {
1291  man_def_auth = KMDA_DEF;
1292  }
1293 #endif
1294 
1295  /* enforce character class restrictions in username/password */
1296  string_mod_remap_name(up->username);
1297  string_mod(up->password, CC_PRINT, CC_CRLF, '_');
1298 
1299  /*
1300  * If auth token succeeds we skip the auth
1301  * methods unless otherwise specified
1302  */
1303  bool skip_auth = false;
1304 
1305  /*
1306  * If server is configured with --auth-gen-token and the client sends
1307  * something that looks like an authentication token, this
1308  * round will be done internally using the token instead of
1309  * calling any external authentication modules.
1310  */
1311  if (session->opt->auth_token_generate && is_auth_token(up->password))
1312  {
1313  ks->auth_token_state_flags = verify_auth_token(up, multi, session);
1314  if (session->opt->auth_token_call_auth)
1315  {
1316  /*
1317  * we do not care about the result here because it is
1318  * the responsibility of the external authentication to
1319  * decide what to do with the result
1320  */
1321  }
1322  else if (ks->auth_token_state_flags == AUTH_TOKEN_HMAC_OK)
1323  {
1324  /*
1325  * We do not want the EXPIRED or EMPTY USER flags here so check
1326  * for equality with AUTH_TOKEN_HMAC_OK
1327  */
1328  msg(M_WARN, "TLS: Username/auth-token authentication "
1329  "succeeded for username '%s'",
1330  up->username);
1331  skip_auth = true;
1332  }
1333  else
1334  {
1335  wipe_auth_token(multi);
1336  ks->authenticated = KS_AUTH_FALSE;
1337  msg(M_WARN, "TLS: Username/auth-token authentication "
1338  "failed for username '%s'", up->username);
1339  return;
1340  }
1341  }
1342 
1343  /* Set the environment variables used by all auth variants */
1344  if (!set_verify_user_pass_env(up, multi, session))
1345  {
1346  skip_auth = true;
1347  s1 = OPENVPN_PLUGIN_FUNC_ERROR;
1348  }

1351  if (!skip_auth)
1352  {
1353 #ifdef MANAGEMENT_DEF_AUTH
1354  if (man_def_auth==KMDA_DEF)
1355  {
1356  man_def_auth = verify_user_pass_management(session, multi, up);
1357  }
1358 #endif

1360  {
1361  s1 = verify_user_pass_plugin(session, multi, up);
1362  }
1363 
1364  if (session->opt->auth_user_pass_verify_script)
1365  {
1366  s2 = verify_user_pass_script(session, multi, up);
1367  }
1368  }
1369

1371  if ((session->opt->ssl_flags & SSLF_USERNAME_AS_COMMON_NAME)
1372  && strlen(up->username)>TLS_USERNAME_LEN)
1373  {
1374  msg(D_TLS_ERRORS,
1375  "TLS Auth Error: --username-as-common name specified and username is longer than the maximum permitted Common Name length of %d characters",
1376  TLS_USERNAME_LEN);
1377  s1 = OPENVPN_PLUGIN_FUNC_ERROR;
1378  }
1379  /* auth succeeded? */
1380  if ((s1 == OPENVPN_PLUGIN_FUNC_SUCCESS
1381 #ifdef PLUGIN_DEF_AUTH
1382  || s1 == OPENVPN_PLUGIN_FUNC_DEFERRED
1383 #endif
1384  ) && s2
1385 #ifdef MANAGEMENT_DEF_AUTH
1386  && man_def_auth != KMDA_ERROR
1387 #endif
1388  && tls_lock_username(multi, up->username))
1389  {
1390  ks->authenticated = KS_AUTH_TRUE;
1391 #ifdef PLUGIN_DEF_AUTH
1392  if (s1 == OPENVPN_PLUGIN_FUNC_DEFERRED)
1393  {
1394  ks->authenticated = KS_AUTH_DEFERRED;
1395  }
1396 #endif
1397 #ifdef MANAGEMENT_DEF_AUTH
1398  if (man_def_auth != KMDA_UNDEF)
1399  {
1400  if (skip_auth)
1401  {
1402  ks->mda_status = ACF_DISABLED;
1403  }
1404  else
1405  {
1406  ks->authenticated = KS_AUTH_DEFERRED;
1407  }
1408  }
1409 #endif
1410  if ((session->opt->ssl_flags & SSLF_USERNAME_AS_COMMON_NAME))
1411  {
1412  set_common_name(session, up->username);
1413  }
1414 
1415  if ((session->opt->auth_token_generate))
1416  {
1417  /*
1418  * If we accepted a (not expired) token, i.e.
