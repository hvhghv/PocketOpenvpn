231 bool
232 crypto_pem_encode(const char *name, struct buffer *dst,
233  const struct buffer *src, struct gc_arena *gc)
234 {
235  /* 1000 chars is the PEM line length limit (+1 for tailing NUL) */
236  char header[1000+1] = { 0 };
237  char footer[1000+1] = { 0 };
238 
239  if (!openvpn_snprintf(header, sizeof(header), "-----BEGIN %s-----\n", name))
240  {
241  return false;
242  }
243  if (!openvpn_snprintf(footer, sizeof(footer), "-----END %s-----\n", name))
244  {
245  return false;
246  }
247 
248  size_t out_len = 0;
249  if (MBEDTLS_ERR_BASE64_BUFFER_TOO_SMALL !=
250  mbedtls_pem_write_buffer(header, footer, BPTR(src), BLEN(src),
251  NULL, 0, &out_len))
252  {
253  return false;
254  }
255 
256  /* We set the size buf to out_len-1 to NOT include the 0 byte that
257  * mbedtls_pem_write_buffer in its length calculation */
258  *dst = alloc_buf_gc(out_len, gc);
259  if (!mbed_ok(mbedtls_pem_write_buffer(header, footer, BPTR(src), BLEN(src),
260  BPTR(dst), BCAP(dst), &out_len))
261  || !buf_inc_len(dst, out_len-1))
262  {
263  CLEAR(*dst);
264  return false;
265  }
266 
267  return true;
268 }

270 bool
271 crypto_pem_decode(const char *name, struct buffer *dst,
272  const struct buffer *src)
273 {
274  /* 1000 chars is the PEM line length limit (+1 for tailing NUL) */
275  char header[1000+1] = { 0 };
276  char footer[1000+1] = { 0 };
277 
278  if (!openvpn_snprintf(header, sizeof(header), "-----BEGIN %s-----", name))
279  {
280  return false;
281  }
282  if (!openvpn_snprintf(footer, sizeof(footer), "-----END %s-----", name))
283  {
284  return false;
285  }
286 
287  /* mbed TLS requires the src to be null-terminated */
288  /* allocate a new buffer to avoid modifying the src buffer */
289  struct gc_arena gc = gc_new();
290  struct buffer input = alloc_buf_gc(BLEN(src) + 1, &gc);
291  buf_copy(&input, src);
292  buf_null_terminate(&input);
293 
294  size_t use_len = 0;
295  mbedtls_pem_context ctx = { 0 };
296  bool ret = mbed_ok(mbedtls_pem_read_buffer(&ctx, header, footer, BPTR(&input),
297  NULL, 0, &use_len));
298  if (ret && !buf_write(dst, ctx.buf, ctx.buflen))
299  {
300  ret = false;
301  msg(M_WARN, "PEM decode error: destination buffer too small");
302  }
303 
304  mbedtls_pem_free(&ctx);
305  gc_free(&gc);
306  return ret;
307 }

318 /*
319  * Initialise the given ctr_drbg context, using a personalisation string and an
320  * entropy gathering function.
321  */
322 mbedtls_ctr_drbg_context *
323 rand_ctx_get(void)
324 {
325  static mbedtls_entropy_context ec = {0};
326  static mbedtls_ctr_drbg_context cd_ctx = {0};
327  static bool rand_initialised = false;
328 
329  if (!rand_initialised)
330  {
331  struct gc_arena gc = gc_new();
332  struct buffer pers_string = alloc_buf_gc(100, &gc);
333 
334  /*
335  * Personalisation string, should be as unique as possible (see NIST
336  * 800-90 section 8.7.1). We have very little information at this stage.
337  * Include Program Name, memory address of the context and PID.
338  */
339  buf_printf(&pers_string, "OpenVPN %0u %p %s", platform_getpid(), &cd_ctx, time_string(0, 0, 0, &gc));
340 
341  /* Initialise mbed TLS RNG, and built-in entropy sources */
342  mbedtls_entropy_init(&ec);
343 
344  mbedtls_ctr_drbg_init(&cd_ctx);
345  if (!mbed_ok(mbedtls_ctr_drbg_seed(&cd_ctx, mbedtls_entropy_func, &ec,
346  BPTR(&pers_string), BLEN(&pers_string))))
347  {
348  msg(M_FATAL, "Failed to initialize random generator");
349  }
350 
351  gc_free(&gc);
352  rand_initialised = true;
353  }
354 
355  return &cd_ctx;
356 }

981 int
982 memcmp_constant_time(const void *a, const void *b, size_t size)
983 {
984  /* mbed TLS has a no const time memcmp function that it exposes
985  * via its APIs like OpenSSL does with CRYPTO_memcmp
986  * Adapt the function that mbedtls itself uses in
987  * mbedtls_safer_memcmp as it considers that to be safe */
988  volatile const unsigned char *A = (volatile const unsigned char *) a;
989  volatile const unsigned char *B = (volatile const unsigned char *) b;
990  volatile unsigned char diff = 0;
991 
992  for (size_t i = 0; i < size; i++)
993  {
994  unsigned char x = A[i], y = B[i];
995  diff |= x ^ y;
996  }
997
