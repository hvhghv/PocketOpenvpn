362 /*
363  * Return the io_wait() flags appropriate for

366 static inline unsigned int
367 p2p_iow_flags(const struct context *c)
368 {
369  unsigned int flags = (IOW_SHAPER|IOW_CHECK_RESIDUAL|IOW_FRAG|IOW_READ|IOW_WAIT_SIGNAL);
370  if (c->c2.to_link.len > 0)
371  {
372  flags |= IOW_TO_LINK;
373  }

375  {

377  }

380  {

382  }
383 #endif
384  return flags;

387 /*
388  * This is the core I/O wait function, used for all I/O waits except
389  * for TCP in server mode.
390  */
391 static inline void
392 io_wait(struct context *c, const unsigned int flags)
393 {
394  void io_wait_dowork(struct context *c, const unsigned int flags);
395 
396  if (c->c2.fast_io && (flags & (IOW_TO_TUN|IOW_TO_LINK|IOW_MBUF)))
397  {
398  /* fast path -- only for TUN/TAP/UDP writes */
399  unsigned int ret = 0;
400  if (flags & IOW_TO_TUN)
401  {
402  ret |= TUN_WRITE;
403  }
404  if (flags & (IOW_TO_LINK|IOW_MBUF))
405  {
406  ret |= SOCKET_WRITE;
407  }
408  c->c2.event_set_status = ret;
409  }
410  else
411  {
412 #ifdef _WIN32
413  bool skip_iowait = flags & IOW_TO_TUN;
414  if (flags & IOW_READ_TUN)
415  {
416  /*
417  * don't read from tun if we have pending write to link,
418  * since every tun read overwrites to_link buffer filled
419  * by previous tun read
420  */
421  skip_iowait = !(flags & IOW_TO_LINK);
422  }
423  if (tuntap_is_wintun(c->c1.tuntap) && skip_iowait)
424  {
425  unsigned int ret = 0;
426  if (flags & IOW_TO_TUN)
427  {
428  ret |= TUN_WRITE;
429  }
430  if (flags & IOW_READ_TUN)
431  {
432  ret |= TUN_READ;
433  }
434  c->c2.event_set_status = ret;
435  }
436  else
437 #endif /* ifdef _WIN32 */
438  {
439  /* slow path */
440  io_wait_dowork(c, flags);
441  }
442  }
443 }
