232 static void
233 plugin_init_item(struct plugin *p, const struct plugin_option *o)
234 {
235  struct gc_arena gc = gc_new();
236  bool rel = false;
237 
238  p->so_pathname = o->so_pathname;
239  p->plugin_type_mask = plugin_supported_types();
240 
241 #ifndef _WIN32
242 
243  p->handle = NULL;
244 
245  /* If the plug-in filename is not an absolute path,
246  * or beginning with '.', it should use the PLUGIN_LIBDIR
247  * as the base directory for loading the plug-in.
248  *
249  * This means the following scenarios are loaded from these places:
250  * --plugin fancyplug.so -> $PLUGIN_LIBDIR/fancyplug.so
251  * --plugin my/fancyplug.so -> $PLUGIN_LIBDIR/my/fancyplug.so
252  * --plugin ./fancyplug.so -> $CWD/fancyplug.so
253  * --plugin /usr/lib/my/fancyplug.so -> /usr/lib/my/fancyplug.so
254  *
255  * Please note that $CWD means the directory OpenVPN is either started from
256  * or the directory OpenVPN have changed into using --cd before --plugin
257  * was parsed.
258  *
259  */
260  if (!platform_absolute_pathname(p->so_pathname)
261  && p->so_pathname[0] != '.')
262  {
263  char full[PATH_MAX];
264 
265  openvpn_snprintf(full, sizeof(full), "%s/%s", PLUGIN_LIBDIR, p->so_pathname);
266  p->handle = dlopen(full, RTLD_NOW);
267  }
268  else
269  {
270  rel = !platform_absolute_pathname(p->so_pathname);
271  p->handle = dlopen(p->so_pathname, RTLD_NOW);
272  }
273  if (!p->handle)
274  {
275  msg(M_ERR, "PLUGIN_INIT: could not load plugin shared object %s: %s", p->so_pathname, dlerror());
276  }
277 
278 #define PLUGIN_SYM(var, name, flags) libdl_resolve_symbol(p->handle, (void *)&p->var, name, p->so_pathname, flags)
279 
280 #else /* ifndef _WIN32 */
281 
282  WCHAR *wpath = wide_string(p->so_pathname, &gc);
283  WCHAR normalized_plugin_path[MAX_PATH] = {0};
284  /* Normalize the plugin path, converting any relative paths to absolute paths. */
285  if (!GetFullPathNameW(wpath, MAX_PATH, normalized_plugin_path, NULL))
286  {
287  msg(M_ERR, "PLUGIN_INIT: could not load plugin DLL: %ls. Failed to normalize plugin path.", wpath);
288  }
289 
290  if (!plugin_in_trusted_dir(normalized_plugin_path))
291  {
292  msg(M_FATAL, "PLUGIN_INIT: could not load plugin DLL: %ls. The DLL is not in a trusted directory.", normalized_plugin_path);
293  }
294 
295  p->module = LoadLibraryW(normalized_plugin_path);
296  if (!p->module)
297  {
298  msg(M_ERR, "PLUGIN_INIT: could not load plugin DLL: %ls", normalized_plugin_path);
299  }
300 
301 #define PLUGIN_SYM(var, name, flags) dll_resolve_symbol(p->module, (void *)&p->var, name, p->so_pathname, flags)
302 
303 #endif /* ifndef _WIN32 */

320  msg(M_FATAL, "PLUGIN: symbol openvpn_plugin_open_vX is undefined in plugin: %s", p->so_pathname);
321  }
322

324  {
325  msg(M_FATAL, "PLUGIN: symbol openvpn_plugin_func_vX is undefined in plugin: %s", p->so_pathname);
326  }
327 
328  /*
329  * Verify that we are sufficiently up-to-date to handle the plugin
330  */
331  if (p->min_version_required)
332  {
333  const int plugin_needs_version = (*p->min_version_required)();
334  if (plugin_needs_version > OPENVPN_PLUGIN_VERSION)
335  {

337  plugin_needs_version,
338  OPENVPN_PLUGIN_VERSION,
339  p->so_pathname);
340  }
341  }
342 
343  if (p->initialization_point)
344  {
345  p->requested_initialization_point = (*p->initialization_point)();
346  }
347  else
348  {
349  p->requested_initialization_point = OPENVPN_PLUGIN_INIT_PRE_DAEMON;
350  }
351 
352  if (rel)
353  {
354  msg(M_WARN, "WARNING: plugin '%s' specified by a relative pathname -- using an absolute pathname would be more secure", p->so_pathname);
355  }
356 
357  p->initialized = true;
358 
359  gc_free(&gc);
360 }

362 static void
363 plugin_vlog(openvpn_plugin_log_flags_t flags, const char *name, const char *format, va_list arglist)
364 {
365  unsigned int msg_flags = 0;
366 
367  if (!format)
368  {
369  return;
370  }
371 
372  if (!name || name[0] == '\0')
373  {
374  msg(D_PLUGIN_DEBUG, "PLUGIN: suppressed log message from plugin with unknown name");
375  return;
376  }
377 
378  if (flags & PLOG_ERR)
379  {
380  msg_flags = M_INFO | M_NONFATAL;
381  }
382  else if (flags & PLOG_WARN)
383  {
384  msg_flags = M_INFO | M_WARN;
385  }
386  else if (flags & PLOG_NOTE)
387  {
388  msg_flags = M_INFO;
389  }
390  else if (flags & PLOG_DEBUG)
391  {
392  msg_flags = D_PLUGIN_DEBUG;
393  }
394 
395  if (flags & PLOG_ERRNO)
396  {
397  msg_flags |= M_ERRNO;
398  }
399  if (flags & PLOG_NOMUTE)
400  {
401  msg_flags |= M_NOMUTE;
402  }
403 
404  if (msg_test(msg_flags))
405  {
406  struct gc_arena gc;
407  char *msg_fmt;
408 
409  /* Never add instance prefix; not thread safe */
410  msg_flags |= M_NOIPREFIX;
411 
412  gc_init(&gc);
413  msg_fmt = gc_malloc(ERR_BUF_SIZE, false, &gc);
414  openvpn_snprintf(msg_fmt, ERR_BUF_SIZE, "PLUGIN %s: %s", name, format);
415  x_msg_va(msg_flags, msg_fmt, arglist);
416 
417  gc_free(&gc);
418  }
419 }

449 static void
450 plugin_open_item(struct plugin *p,
451  const struct plugin_option *o,
452  struct openvpn_plugin_string_list **retlist,
453  const char **envp,
454  const int init_point)
455 {
456  ASSERT(p->initialized);
457 
458  /* clear return list */
459  if (retlist)
460  {
461  *retlist = NULL;
462  }
463 
464  if (!p->plugin_handle && init_point == p->requested_initialization_point)
465  {
466  struct gc_arena gc = gc_new();
467 
468  dmsg(D_PLUGIN_DEBUG, "PLUGIN_INIT: PRE");
469  plugin_show_args_env(D_PLUGIN_DEBUG, o->argv, envp);
470 
471  /*
472  * Call the plugin initialization
473  */
474  if (p->open3)
475  {
476  struct openvpn_plugin_args_open_in args = { p->plugin_type_mask,
477  (const char **const) o->argv,
478  (const char **const) envp,
479  &callbacks,
480  SSLAPI,
481  PACKAGE_VERSION,
482  OPENVPN_VERSION_MAJOR,
483  OPENVPN_VERSION_MINOR,
484  _OPENVPN_PATCH_LEVEL};
485  struct openvpn_plugin_args_open_return retargs;
486 
487  CLEAR(retargs);
488  retargs.return_list = retlist;
489  if ((*p->open3)(OPENVPN_PLUGINv3_STRUCTVER, &args, &retargs) == OPENVPN_PLUGIN_FUNC_SUCCESS)
490  {
491  p->plugin_type_mask = retargs.type_mask;
492  p->plugin_handle = retargs.handle;
493  }
494  else
495  {
496  p->plugin_handle = NULL;
497  }
498  }
499  else if (p->open2)
500  {
501  p->plugin_handle = (*p->open2)(&p->plugin_type_mask, o->argv, envp, retlist);
502  }
503  else if (p->open1)
504  {
505  p->plugin_handle = (*p->open1)(&p->plugin_type_mask, o->argv, envp);
506  }
507  else
508  {
509  ASSERT(0);
510  }
511 
512  msg(D_PLUGIN, "PLUGIN_INIT: POST %s '%s' intercepted=%s %s",
513  p->so_pathname,
514  print_argv(o->argv, &gc, PA_BRACKET),
515  plugin_mask_string(p->plugin_type_mask, &gc),
516  (retlist && *retlist) ? "[RETLIST]" : "");
517 
518  if ((p->plugin_type_mask | plugin_supported_types()) != plugin_supported_types())
519  {
520  msg(M_FATAL, "PLUGIN_INIT: plugin %s expressed interest in unsupported plugin types: [want=0x%08x, have=0x%08x]",
521  p->so_pathname,
522  p->plugin_type_mask,
523  plugin_supported_types());
524  }
525 
526  if (p->plugin_handle == NULL)
527  {
528  msg(M_FATAL, "PLUGIN_INIT: plugin initialization function failed: %s",
529  p->so_pathname);
530  }
531 
532  gc_free(&gc);
533  }
534 }

536 static int
537 plugin_call_item(const struct plugin *p,
538  void *per_client_context,
539  const int type,
540  const struct argv *av,
541  struct openvpn_plugin_string_list **retlist,
542  const char **envp,
543  int certdepth,
544  openvpn_x509_cert_t *current_cert
545  )
546 {
547  int status = OPENVPN_PLUGIN_FUNC_SUCCESS;
548 
549  /* clear return list */
550  if (retlist)
551  {
552  *retlist = NULL;
553  }
554 
555  if (p->plugin_handle && (p->plugin_type_mask & OPENVPN_PLUGIN_MASK(type)))
556  {
557  struct gc_arena gc = gc_new();
558  struct argv a = argv_insert_head(av, p->so_pathname);
559 
560  dmsg(D_PLUGIN_DEBUG, "PLUGIN_CALL: PRE type=%s", plugin_type_name(type));
561  plugin_show_args_env(D_PLUGIN_DEBUG, (const char **)a.argv, envp);
562 
563  /*
564  * Call the plugin work function
565  */
566  if (p->func3)
567  {
568  struct openvpn_plugin_args_func_in args = { type,
569  (const char **const) a.argv,
570  (const char **const) envp,
571  p->plugin_handle,
572  per_client_context,
573  (current_cert ? certdepth : -1),
574  current_cert };
575 
576  struct openvpn_plugin_args_func_return retargs;
577 
578  CLEAR(retargs);
579  retargs.return_list = retlist;
580  status = (*p->func3)(OPENVPN_PLUGINv3_STRUCTVER, &args, &retargs);
581  }
582  else if (p->func2)
583  {
584  status = (*p->func2)(p->plugin_handle, type, (const char **)a.argv, envp, per_client_context, retlist);
585  }
586  else if (p->func1)
587  {
588  status = (*p->func1)(p->plugin_handle, type, (const char **)a.argv, envp);
589  }
590  else
591  {
592  ASSERT(0);
593  }
594 
595  msg(D_PLUGIN, "PLUGIN_CALL: POST %s/%s status=%d",
596  p->so_pathname,
597  plugin_type_name(type),
598  status);
599 
600  if (status == OPENVPN_PLUGIN_FUNC_ERROR)
601  {
602  msg(M_WARN, "PLUGIN_CALL: plugin function %s failed with status %d: %s",
603  plugin_type_name(type),
604  status,
605  p->so_pathname);
606  }
607 
608  argv_free(&a);
609  gc_free(&gc);
610  }
611  return status;
612 }

798 int
799 plugin_call_ssl(const struct plugin_list *pl,
800  const int type,
801  const struct argv *av,
802  struct plugin_return *pr,
803  struct env_set *es,
804  int certdepth,
805  openvpn_x509_cert_t *current_cert
806  )
807 {
808  if (pr)
809  {
810  plugin_return_init(pr);
811  }
812 
813  if (plugin_defined(pl, type))
814  {
815  struct gc_arena gc = gc_new();
816  int i;
817  const char **envp;
818  const int n = plugin_n(pl);
819  bool success = false;
820  bool error = false;
821  bool deferred_auth_done = false;
822 
823  setenv_del(es, "script_type");
824  envp = make_env_array(es, false, &gc);
825 
826  for (i = 0; i < n; ++i)
827  {
828  const int status = plugin_call_item(&pl->common->plugins[i],
829  pl->per_client.per_client_context[i],
830  type,
831  av,
832  pr ? &pr->list[i] : NULL,
833  envp,
834  certdepth,
835  current_cert
836  );
837  switch (status)
838  {
839  case OPENVPN_PLUGIN_FUNC_SUCCESS:
840  success = true;
841  break;
842 
843  case OPENVPN_PLUGIN_FUNC_DEFERRED:
844  if ((type == OPENVPN_PLUGIN_AUTH_USER_PASS_VERIFY)
845  && deferred_auth_done)
846  {
847  /*
848  * Do not allow deferred auth if a deferred auth has
849  * already been started. This should allow a single
850  * deferred auth call to happen, with one or more
851  * auth calls with an instant authentication result.
852  *
853  * The plug-in API is not designed for multiple
854  * deferred authentications to happen, as the
855  * auth_control_file file will be shared across all
856  * the plug-ins.
857  *
858  * Since this is considered a critical configuration
859  * error, we bail out and exit the OpenVPN process.
860  */
861  error = true;
862  msg(M_FATAL,
863  "Exiting due to multiple authentication plug-ins "
864  "performing deferred authentication. Only one "
865  "authentication plug-in doing deferred auth is "
866  "allowed. Ignoring the result and stopping now, "
867  "the current authentication result is not to be "
868  "trusted.");
869  break;
870  }
871  deferred_auth_done = true;
872  break;
873 
874  default:
875  error = true;
876  break;
877  }
878  }
879 
880  if (pr)
881  {
882  pr->n = i;
883  }
884 
885  gc_free(&gc);
886 
887  if (type == OPENVPN_PLUGIN_ENABLE_PF && success)
888  {
889  return OPENVPN_PLUGIN_FUNC_SUCCESS;
890  }
891  else if (error)
892  {
893  return OPENVPN_PLUGIN_FUNC_ERROR;
894  }
895  else if (deferred_auth_done)
896  {
897  return OPENVPN_PLUGIN_FUNC_DEFERRED;
898  }
899  }
900 
901  return OPENVPN_PLUGIN_FUNC_SUCCESS;
902 }
